%======== Abschnitt 3.2 ============
%Orthogonale Basen

\begin{bem}
Im Folgenden werden nur noch separable Hilberträume $H$ betrachtet, d.h. solche für die eine abzählbare, dichte Teilmenge existiert (Vergleiche dazu Definition 2.1.5). Der Vorteil dieser Einschränkung ist die Möglichkeit Reihen einfacher zu formulieren.
\end{bem}
\medskip

\begin{bsp}
Die Räume $(\C^n, \norm{\cdot}_2)$, $l_2$ und $L_2(\Omega)$ (für $\Omega \subset \R^n$ separabel) sind alle separabel.
\end{bsp}
\medskip

\begin{bem}
$(x_i)_{i\in I} \subset H$ für $I$ überabzählbar heißt summierbar \gdwdef $\exists x\in H$ und für alle $\epsilon > 0$ existiert eine endliche Indexmenge $J(\epsilon)\subset I$, sodass für alle endlichen Indexmengen $J$ mit $J(\epsilon) \subset J \subset I$ stets $\norm{x- \sum_{i\in J} x_i}<\epsilon$ gilt.\\
Solche Folgen / Reihen muss man betrachen, wenn man auch nicht separable Hilberträume $H$ betrachten will.
\end{bem}
\medskip

\begin{defi}
Die Folge $(e_j)_{j=1}^{\infty} \subset H$ (oder eventuell auch nur endlich viele) heißt \highl[Orthonormalsystem (ONS)]{Orthonormalsystem} (kurz \highl[ONS|see{Orthonormalsystem}]{ONS}) \gdwdef Für alle $j,k \in \N$ gilt:
\begin{gather*}
(e_j,e_k)=\delta_{jk}=
\begin{cases}
1, & j=k\\
0, & j \neq k
\end{cases}
\end{gather*}
\end{defi}
\medskip

\begin{bem}
Das verschwinden des Skalarprodukts für verschiedene Elemente rechtfertigt die Bezeichnung "`ortho"' und die Normierung $\norm{e_j}^2=(e_j,e_j)=1$ ($\Rightarrow \norm{e_j}=1$) rechtfertigt die Bezeichnung "`normal"'.\\
Das Ziel ist es, ein ONS zu finden, welches gleichzeitig Basis des Hilbertraumes ist.
\end{bem}
\medskip

\begin{satz}
Sei $H$ ein (Prä-) Hilbertraum und $x \in H$. Dann gilt:\\
(i) Ist $E:=\{ e_1, \ldots, e_J \} \subset H$ ein endliches ONS, so
\begin{gather*}
\exists! y \in \spn{E} \text{ mit } \norm{x-y}=\inf_{z \in \spn{E}} \norm{x-z}
\end{gather*}
(ii) Ist $(e_j)_{j=1}^{\infty} \subset H$ ein ONS, so gilt die \highl[Ungleichung!von Bessel]{Bessel'sche Ungleichung}:
\begin{gather*}
\sum_{j=1}^{\infty} |(x,e_j)|^2 \leq \norm{x}^2
\end{gather*}
\end{satz}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{bem}
Mithilfe des Pythagoras ist leicht zu sehen, dass für $e_k \bot e_j$, für $j \neq k$ und $\lambda_k, \lambda_j \in \C$ gilt $\norm{\sum_{j=1}^J \lambda_j e_j}^2=\sum_{j=1}^J \norm{\lambda_j e_j}^2 = \sum_{j=1}^J |\lambda_j|^2$.
\end{bem}
\medskip

\begin{lemmas}
Ist $(e_j)_{j=1}^{\infty}$ ein ONS eines (Prä-) Hilbertraumes $H$ und $x \in H$. Dann gilt:
\begin{gather*}
x = \sum_{j=1}^{\infty} (x,e_j) e_j \gdw \sum_{j=1}^{\infty} |(x,e_j)|^2 = \norm{x}^2 \quad \text{(\highl[Gleichung!Parseval'sche]{Parseval'sche Gleichung})}
\end{gather*}
\end{lemmas}
\begin{proof}
Folgt aus der Herleitung im Beweis des Satzes 3.2.1 (ii).
\end{proof}
\medskip

\begin{defis}
Ist $(e_j)_{j=1}^{\infty}$ ein ONS eines (Prä-) Hilbertraumes $H$ und $x \in H$, dann heißt\\
(i) \highl[$(x,e_j)$|see{Fourier}]{$(x,e_j)$} $\in \C$ \highl[Fourier!Fourierkoeffizient $(x,e_j)$]{Fourierkoeffizient} von $x$ bezüglich $e_j$,\\
(ii) \highl[$\sum_{j=1}^{\infty} (x,e_j) e_j$|see{Fourier}]{$\sum_{j=1}^{\infty} (x,e_j) e_j$} \highl[Fourier!Fourierreihe]{Fourierreihe} von $x$.\\
\end{defis}
\medskip

\begin{satz}
Ist $(e_j)_{j=1}^{\infty}$ ein ONS eines separablen, unendlich-dimensionalen Hilbertraumes ($\{ e_1, \ldots, e_J \}$ im endlich-dimensionalen Hilbertraum) $H$, so sind folgende Aussagen äquivalent:
\begin{align*}
\text{(i) }   & x=\sum_{j=1}^{\infty} (x,e_j)e_j, \quad \forall x\in H \\
\text{(ii) }  & \sum_{j=1}^{\infty} |(x,e_j)|^2 = \norm{x}^2, \quad \forall x\in H \\
\text{(iii) } & (x,y)=\sum_{j=1}^{\infty} (x,e_j) \overline{(y,e_j)}, \quad \forall x,y\in H \\
\text{(iv) }  & \spn{\{ e_j \sep j\in \N \}} \text{ ist dicht in } H \\
\text{(v) }   & \{ e_j \sep j \in \N \}^{\bot} = \{ 0 \} \\
\text{(vi) }  & \text{Die Abbildung } H \nach l_2 \text{ mit } x \mapsto \left( (x,e_j) \right)_{j=1}^{\infty} \text{ ist bijektiv und normerhaltend} \\
							& \text{(Isometrie)} \\
\text{(vii) }  & (x,e_j)=0 \quad \forall j \in \N \Rightarrow x=0.
\end{align*}
\end{satz}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{bem}
Hat man also ein ONS, für welches eine der Aussagen in Satz 3.2.2 gilt, so ist dieser unendlich-dimensionale separable Hilbertraum isometrisch isomorph zum Raum $l_2$.
\end{bem}
\medskip

\begin{defi}
Ein ONS $(e_j)_{j=1}^{\infty} \subset H$ heißt \highl[Basis!Orthonormalbasis (ONB)]{Orthonormalbasis} (kurz \highl[ONB|see{Orthonormalbasis}]{ONB}) eines separablen Hilbertraumes $H$ (oder auch \highl[vollständig!Orthonormalsystem|see{Orthonormalbasis}]{vollständiges ONS}, kurz \highl[VONS|see{Orthonormalbasis}]{VONS}) \gdwdef Für alle $x\in H$ gilt die Parseval'sche Gleichung (bzw. eine der äquivalenten Aussagen aus Satz 3.2.2).
\end{defi}
\medskip

\begin{satz}
In jedem Hilbertraum existiert eine Orthonormalbasis.
\end{satz}
\begin{proof}

Hier, für den Spezialfall separabler Hilberträume, wird ein konstruktiver Beweis mit Hilfe des \highl[orthogonal!Orthogonalisierungsverfahren|see{Gram-Schmidt-Verfahren}]{Orthogonalisierungsverfahrens} von \highl[Gram-Schmidt-Verfahren]{Gram-Schmidt} geführt:\\
Siehe Vorlesung.
\end{proof}
\medskip

\begin{bsp}
In $L_2([-1,1])$ liegen die stetigen Funktionen $C([-1,1])$ dicht. In diesen liegen wiederum die Polynome $\spn{\{ 1, t, t^2, t^3, \ldots \}}$ dicht. Die Menge $F:=\{ 1, t, t^2, t^3, \ldots \}$ entspricht also der, aus dem zweiten Beweisschritt in Satz 3.2.3. Die Elemente sind noch nicht orthogonal zueinander, denn es ist beispielsweise
\begin{gather*}
(1,t^2)=\int_{-1}^1 t^2 dt = \frac{2}{3} \neq 0.
\end{gather*}
Hier muss also noch das Gram-Schmidt-Verfahren angewendet werden (siehe dazu Übungsserie 10).
\end{bsp}
\medskip

\begin{bsp}[für Orthonormalbasen]
(i) In $(\C^n, \norm{\cdot}_2)$ ist $(e_j)_{j=1}^n$ mit $e_j = (0,\ldots,0,1,0,\ldots,0)$ eine ONB. Analog dazu bildet $(e_j)_{j=1}^\infty$ eine Orthonormalbasis des $l_2$.\\
\\
(ii) Im Raum $L_2(-\pi,\pi)$ bilden die Basisfunktionen der klassischen Fourierreihe\\
$\{ \frac{1}{\sqrt{2\pi}} \, e^{ikt} \sep k \in \Z \}$ bzw. $\{ \frac{1}{\sqrt{2\pi}}, \frac{\cos{t}}{\sqrt{\pi}}, \frac{\sin{t}}{\sqrt{\pi}}, \frac{\cos{2t}}{\sqrt{\pi}}, \frac{\sin{2t}}{\sqrt{\pi}}, \ldots \}$ eine ONB. Sie sind Lösungen der Differentialgleichung $u'' - \lambda u = 0$ mit $u(\pi)=u(-\pi)$.\\
\\
(iii) Beispiel (ii) lässt sich auf $L_2(0,a)$ verallgemeinern. Hier ist $\{ \frac{1}{\sqrt{a}} \, e^{2\pi i \frac{k}{a} t} \sep k \in \Z \}$ eine Orthonormalbasis.\\
\\
(iv) Auf $L_2(-1,1)$ lässt sich aber neben der klassischen Fourierbasis auch eine Orthonormalbasis aus Polynomen angeben: $\{ \eta_n(t) := \sqrt{\frac{2n+1}{2}} \, P_n(t) \sep n \in \N_0 \}$, wobei die Funktionen $P_n$ \highl[Polynome!Legendre'sche $P_n$]{Legendre'sche Polynome} heißen und wie folgt definiert sind:\\
$P_n(t) = \frac{1}{2^n n!} \frac{d^n}{dt^n} \left[ (t^2-1)^n \right]$. Es handelt sich hierbei um Polynome vom Grad $n$ mit Vorfaktor 1.\\
Die ONB erhält man einerseits mittels eines Potenzreihenansatzes als Lösung der Differentialgleichung $(1-t)^2 u'' - 2tu' +\lambda u = 0$ mit $|u(1)|,|u(-1)|<\infty$ oder durch das Gram-Schmidt-Verfahren, angewendet auf die Menge der Polynome $\{ 1, t, t^2, \ldots\}$, denn es gilt $\overline{\spn{ \{ 1, t, t^2, \ldots \} }} = L_2(-1,1)$.\\
Die Legendre-Polynome sind Lösungen der \highl[Differentialgleichung!Legendre'sche]{Legendre'schen Differentialgleichung}\\
$(1-t^2) P''_n - 2t P'_n + n(n+1)P_n=0$ mit $|P_n(1)|,|P_n(-1)|<\infty$.\\
\\
(v) In $L_2(\R)$ ist $\{ \Psi_n(t):= \frac{1}{\sqrt{2^n n! \sqrt{\pi}}} \, e^{- \frac{t^2}{2}} \, H_n(t) \sep n \in \N_0 \}$ mit den \highl[Polynome!Hermite'sche $H_n$]{Hermite'schen Polynomen} (vom Grad $n$, mit Vorfaktor 1) $H_n(t):= (-1)^n \, e^{t^2} \, \frac{d^n}{dt^n} \left[ e^{-t^2} \right]$ eine ONB. Wegen dem $L_2$-Skalarprodukt $(f,g)=\int_\R f(t) \overline{g(t)}dt$ ist $\Psi_n \in L_2(\R)$.\\
Wählt man den Raum $L_{2,w}(\R)$ mit dem Skalarprodukt $(f,g)_{L_{2,w}}:=\int_\R f(t) \overline{g(t)} w(t) dt$ und $w(t):=e^{-t^2}$, so sind auch die Hermite'schen Polynome Elemente von $L_{2,w}(\R)$. Sie sind Lösungen der \highl[Differentialgleichung!Hermite'sche]{Hermite'schen Differentialgleichung} $H''_n - 2tH'_n+2nH_n=0$.\\
\\
(vi) Auf $L_2((0,\infty))$ ist $\{ \phi_n(t):= \frac{1}{n!} \, e^{- \frac{t}{2}} \, L_n(t) \sep n \in \N_0 \}$ eine Orthonormalbasis. Die Funktionen $L_n(t):= e^t \frac{d^n}{dt^n} \left[ t^n e^{-t} \right]$ heißen \highl[Polynome!Laguerr'sche $L_n$]{Laguerr'sche Polynome} (vom Grad $n$). Man erhält sie durch eine Potenzreihenansatz aus der \highl[Differentialgleichung!Laguerr'sche]{Laguerr'schen Differentialgleichung} $t L''_n(t) +(1-t) L'_n(t) + n L_n(t) = 0$ für $0<t<\infty$.\\
\\
(vii) Im Raum $L_2(\times_{j=1}^n(-\pi,\pi))$, wobei $\times_{j=1}^n(-\pi,\pi)$ als $n$-dimensionaler Würfel mit der Seitenlänge $2\pi$ angesehen werden kann, bildet $\{ \frac{1}{(2\pi)^{\frac{n}{2}}} \, e^{ikx} \sep k \in \Z^n \}$ eine ONB, wenn man $kx:=k_1x_1+\ldots+k_nx_n$ vereinbart.\\
\\
(viii) Die Menge $w_3 = S^2 := \{ x\in \R^3 \sep r=|x|=1 \}$ bezeichne die Oberfläche der Einheitskugel im $\R^3$ mit Kugelkoordinaten. Der Raum $L_2(w_3)$ mit dem Skalarprodukt $(f,g)_{L_2(w_3)}:=\int_{w_3} f(\phi, \vartheta) \overline{g(\phi, \vartheta)} \, dS_{w_3} = \int_0^{2\pi} \int_0^{\pi} f(\phi, \vartheta) \overline{g(\phi, \vartheta)} \sin{\vartheta} \, d\vartheta d\phi$ ist ein Hilbertraum. Man definiert die zugeordneten Legendre'schen Polynome für $m=0,\ldots,l$ durch $P_m^l(t):=(1-t^2)^{\frac{m}{2}} \frac{d^{l+m}}{dt^{l+m}} \left[ (1-t^2)^l\right]$ (die gewöhnlichen Legendre Polynome erhält man mit $m=0$). \\
Dann bilden die Kugelflächenfunktionen $\{ Y_{l,m} \sep l \in \N_0, m=0,\ldots,l \}$, sowie\\
$\{ \widetilde{Y}_{l,m} \sep l \in \N_0, m=1,\ldots,l\}$ eine Orthonormalbasis des $L_2(w_3)$ mit $(\cdot,\cdot)_{L_2(w_3)}$. Dabei ist
$Y_{l,m}(\phi, \vartheta):=\frac{1}{2^l l!} \sqrt{\frac{2(l+1)(l-m)!}{2\pi(l+m)!}} P_l^m(\cos{\vartheta}) \cos{(m\phi)}$ und\\ $\widetilde{Y}_{l,m}(\phi, \vartheta):=\frac{1}{2^l l!} \sqrt{\frac{2(l+1)(l-m)!}{2\pi(l+m)!}} P_l^m(\sin{\vartheta}) \sin{(m\phi)}$.\\
\\
(ix) Im $L_2((0,1))$ sind die \highl[Funktionen!Haar'sche $h_n^{(k)}$]{Haar'schen Funktionen} $\{ h_n^{(k)}(t) \sep n \in N_0, k=1,\ldots,2^n\}$ eine Orthonormalbasis (vgl. auch \highl{Wavelets}). Dabei ist
\begin{align*}
h_0^{(0)}(t) & :=1, \\
h_n^{(k)}(t) & :=
\begin{cases}
2^{\frac{n}{2}}, & \frac{k-1}{2^n} \leq t < \frac{k-\frac{1}{2}}{2^n} \\
-2^{\frac{n}{2}}, & \frac{k-\frac{1}{2}}{2^n} \leq t < \frac{k}{2^n} \\
0, & \text{sonst}
\end{cases}
\end{align*}
\\
\\
(x) Die \highl[Funktionen!Rademacher'sche $r_n$]{Rademacher'schen Funktionen} $\{ r_n(t) \sep n \in \N_0\}$ mit
\begin{align*}
r_0(t) & := h_0^{(0)}(t) \\
r_1(t) & := h_1^{(0)}(t) \\
r_n(t) & := \frac{1}{\sqrt{2^n}} \sum_{k=1}^{2^n} h_n^{(k)}(t), \quad n \geq 2 \\
\end{align*}
bilden nur ein Orthonormalsystem des $L_2((0,1))$, sind aber nicht vollständig (also keine ONB).\\
\\
(xi) Die \highl[Funktionen!Walsch'sche $w_n$]{Walsch'schen Funktionen} $\{ w_n(t) \sep n \in \N_0 \}$ hingegen sind eine Orthonormalbasis des $L_2((0,1))$ mit $w_0(t):=1$ und $w_n(t):=r_{\vartheta_1+1}(t) \cdot \ldots \cdot r_{\vartheta_p+1}(t)$ für $n \geq 1$. Dabei ist $n=2^{\vartheta_1}+\ldots+2^{\vartheta_p}$ die Binärdarstellung von $n$, mit $\vartheta_1 < \vartheta_2 < \ldots < \vartheta_p$.
\end{bsp}

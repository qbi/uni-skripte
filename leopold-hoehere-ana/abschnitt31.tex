%Grundbegriffe

\begin{bem}
In diesem Kapitel wollen wir den Begriff des Banachraums weiter spezifizieren. Es ergibt sich folgende Rangordnung (vom Allgemeinen zum Speziellen):\\\\
Topologischer Raum\\
Metrischer Raum\\
Normierter Raum\\
Banachraum\\
Hilbertraum
\end{bem}
\medskip

\begin{bem}[unübliche Definition]
Ein Hilbertraum ist ein Banachraum in dem für beliebige Elemente $x,y$ stets die \highl[Gleichung!Parallelogrammgleichung]{Parallelogrammgleichung} $\norm{x+y}^2 + \norm{x-y}^2 = 2 (\norm{x}^2 + \norm{y}^2)$ gilt.
\end{bem}
\medskip

\begin{defi}
Sei $H$ ein linearer Raum über $\C$ (oder $\R$). Eine Abbildung \highl[$(\cdot,\cdot)$|see{Skalarprodukt}]{$(\cdot,\cdot)$} $\colon H \times H \nach \C$ heißt \highl[Skalarprodukt!Skalarprodukt $(\cdot, \cdot)$]{Skalarprodukt} \gdwdef $\forall x,y,x_1,x_2 \in H$ und $\lambda_1,\lambda_2 \in \C$ (bzw. $\R$) gilt:
\begin{align*}
\text{(i) } & (x,x) > 0 \text{ für } x \neq 0 & \text{(insbesondere reell)} \\
\text{(ii) } & (x,y) = \overline{(y,x)} & (\overline{\cdot} \text{ konjugiert komplex)} \\
\text{(iii) } & (\lambda_1 x_1 + \lambda_2 x_2, y) = \lambda_1 (x_1,y) + \lambda_2 (x_2,y) &
\end{align*}
\end{defi}
\medskip

\begin{bem}
Ein Skalarprodukt ist also eine positiv definite, hermitische Sesquilinearform. Im Falle eines Vektorraums über $\R$ ist das Skalarprodukt eine positiv definite, symmetrische Bilinearform (Abb. $H \times H \nach \R$).
\end{bem}
\medskip

\begin{lemmas}
Es gilt für alle $x,y_1,y_2 \in H$ und $\lambda_1,\lambda_2 \in \C$ (bzw. $\R$):
\begin{align*}
\text{(i) }   & (0,x)=(0x,x)=0(x,x)=0 \\
\text{(ii) }  & (x,\lambda_1 y_1 + \lambda_2 y_2) = \overline{(\lambda_1 y_1 + \lambda_2 y_2, x)} = \overline{\lambda_1 (y_1,x) + \lambda_2 (y_2, x)} = \overline{\lambda_1} (x,y_1) + \overline{\lambda_2} (x,y_2) \\
\text{(iii) } & (x,x)=\overline{(x,x)} \in \R
\end{align*}
\end{lemmas}
\begin{proof}
Trivial.
\end{proof}
\medskip

\begin{satz}
Im Vektorraum $H$ wird mittels der Abbildung $\norm{\cdot}=\sqrt{(\cdot,\cdot)}$ eine Norm definiert.
\end{satz}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{bem}
Durch die vom Skalarprodukt \highl[Skalarprodukt!induzierte Norm]{induzierte Norm} ist $(H, \norm{\cdot})$ ein normierter Raum.
\end{bem}
\medskip

\begin{lemma}[Cauchy-Schwarz'sche Ungleichung]\index{Lemma!Cauchy-Schwarz'sche Ungleichung}\index{Ungleichung!von Cauchy-Schwarz|see{Lemma}}
Sei $H$ ein normierter Raum, $x,y \in H$ und $\norm{\cdot}=\sqrt{(\cdot,\cdot)}$. Dann gilt stets
\begin{gather*}
\abs{(x,y)} \leq \norm{x} \cdot \norm{y}.
\end{gather*}
Gleichheit gilt nur für ($x=0$ oder $y=0$ oder) $x=\lambda y$ mit $\lambda \in \C$.
\end{lemma}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{folgs}
Das Skalarprodunkt ist stetig, d.h. für $(x_n)_{n=1}^{\infty}, (y_n)_{n=1}^{\infty} \subset \C$, sowie  $x, y \in H$ mit $x_n \nach x$ und $y_n \nach y$ gilt $(x_n, y_n) \nach (x,y)$ (in $\C$).
\end{folgs}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{satz}
Sei $X$ ein normierter Raum. Dann gilt:\\
Es existiert auf $X$ ein Skalarprodunkt mit $\norm{\cdot} = \sqrt{(\cdot, \cdot)}$ \gdw für alle $x,y \in X$ gilt die \highl[Gleichung!Parallelogrammgleichung]{Parallelogramm-Gleichung}: $\norm{x+y}^2 + \norm{x-y}^2 = 2 (\norm{x}^2 + \norm{y}^2)$.
\end{satz}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{defi}
(i) Ein linearer Raum $X$ mit Skalarprodukt heißt \highl[Raum!Prähilbertraum]{Prähilbertraum}.\\
(ii) Ist der durch $\norm{\cdot} := \sqrt{(\cdot, \cdot)}$ bestimmte normierte Raum $(X, \norm{\cdot})$ vollständig (d.h. Banachraum), so heißt $H:=X$ \highl[Raum!Hilbertraum]{Hilbertraum} (mit diesem Skalarprodukt).
\end{defi}
\medskip

\begin{bsp}
(i) $H:=\R^n$ oder $\C^n$ mit $l_2^n$-Norm ist ein Hilbertraum mit \highl[Skalarprodukt!in $(\R^n,\norm{\cdot \sep l_2^n})$]{$(x,y)$} $:= \sum_{k=1}^n \xi_k \overline{\eta_k}$, für $x:=(\xi_k)_{k=1}^n, y:=(\eta_k)_{k=1}^n \in H$\\
\\
(ii) $H:=\R^n$ oder $\C^n$ mit $l_p^n$-Norm ($p \neq 2$): kein Skalarprodukt möglich. Sei dazu beispielsweise $p=1$, $x=(1,0,\ldots)$ und $y=(0,1,0,\ldots)$. Dann ist $\norm{x}_1=\norm{y}_1=1$, sowie $\norm{x+y}_1=\norm{x-y}_1=2$. Damit ist die Parallelogrammgleichung verletzt: $2^2+2^2 \neq 2(1+1)$.\\
\\
(iii) $H:=l_2$ ist Hilbertraum mit \highl[Skalarprodukt!in $l_2$]{$(x,y)$} $:= \sum_{k=1}^{\infty} \xi_k \overline{\eta_k}$, für $x:=(\xi_k)_{k=1}^{\infty}, y:=(\eta_k)_{k=1}^{\infty} \in H$, das ist sinnvoll, da nach der Hölder'schen Ungleichung gilt:
\begin{gather*}
\abs{\sum_{k=1}^{\infty} \xi_k \overline{\eta_k}} \leq \left( \sum_{k=1}^{\infty} \abs{\xi_k}^2 \right)^{\frac{1}{2}} \left( \sum_{k=1}^{\infty} \abs{\overline{\eta_k}}^2 \right)^{\frac{1}{2}}
\end{gather*}
Die Norm sieht dann so aus: $\norm{x} = \left( \sum_{k=1}^{\infty} \xi_k \overline{\xi_k} \right)^\frac{1}{2} = \left( \sum_{k=1}^{\infty} \abs{\xi_k}^2 \right)^\frac{1}{2}$ und $l_2$ ist damit vollständig.\\
\\
(iv) Für $\Omega \subset \R^n$ ist $L_2(\Omega)$ (Äquivalenzklassen quadratisch integrierbarer Funktionen von $\Omega$ nach $\R$, die sich nur auf Nullmengen unterscheiden) ein Hilbertraum mit der Norm
\begin{gather*}
\norm{f \sep L_2(\Omega)}=\left( \int_\Omega \abs{f(x)}^2 \, d\mu(x) \right)^{\frac{1}{2}}
\end{gather*}
und folgendem Skalarprodukt:
\begin{gather*}
\text{\highl[Skalarprodukt!in $L_2(\Omega)$]{$(f,g)$}} := \int_\Omega f(x) g(x) \, d\mu (x)
\end{gather*}
Denn $(L_2(\Omega), \norm{\cdot})$ ist nach Lemma 2.1.3 ein Banachraum und $(\cdot, \cdot) \colon L_2 \times L_2 \nach \R$ ist eine symmetrische, offensichtlich linkslineare Abbildung (da das Integral linear ist) für die gilt $(f,f)>0$ für $f \in L_2 \setminus \{0\}$, denn wegen $f \neq 0 \Rightarrow \exists M \subset \Omega$ mit $\mu (M) \neq 0$, sodass gilt $\forall x \in M: f(x) \neq 0 \quad \Rightarrow \quad 0 < \int_M f(x)^2 d\mu(x) \leq \int_\Omega f(x)^2 d\mu(x) = (f,f)$.
\end{bsp}
\medskip

\begin{defi}
Sei $H$ ein (Prä-) Hilbertraum.\\
(i) $x,y \in H$ heißen \highl[orthogonal!Elemente $x \bot y$]{orthogonal} (schreibe \highl[$x \bot y$|see{orthogonal}]{$x \bot y$}) \gdwdef $(x,y)=0$.\\
(ii) $M_1, M_2 \subset H$ heißen \highl[orthogonal!Mengen $M_1 \bot M_2$]{orthogonal} (schreibe \highl[$M_1 \bot M_2$|see{orthogonal}]{$M_1 \bot M_2$}) \gdwdef $\forall x_1 \in M_1, x_2\in M_2$ gilt $x_1 \bot x_2$.\\
(iii) \highl[$M^\bot$|see{orthogonal}]{$M^\bot$} $:= \{ x\in H \sep \forall y \in M: (x,y)=0\}$ heißt das \highl[orthogonal!Komplement $M^\bot$]{orthogonale Komplement} von $M \subset H$.
\end{defi}
\medskip

\begin{folgs}
Sei $H$ ein (Prä-) Hilbertraum.\\
(i) $x,y \in H, x \bot y \Rightarrow \norm{x}^2+\norm{y}^2=\norm{x+y}^2$ (Pythagoras).\\
(ii) $M^\bot$ ist ein abgeschlossener linearer Teilraum von $H$.
\end{folgs}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{folgs}
Sei $H$ ein (Prä-) Hilbertraum. Dann gilt stets:\\
(i) $M \subset (M^{\bot}){^\bot}$ \\
(ii) $M^\bot = \overline{\spn{M}}^{\bot}$
\end{folgs}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{defis}
Eine Teilmenge $M$ eines (reellen) linearen Raumes heißt \highl[Menge!konvex]{konvex} \gdwdef für alle $x,y\in M$ und $t \in [0,1]$ ist $tx+(1-t)y \in M$ (alle Elemente der Verbindungsgerade zwischen $x$ und $y$ aus $M$ gehören ebenfalls zur Menge).
\end{defis}
\medskip

\begin{lemma}
Jede abgeschlossene und konvexe Teilmenge $M$ eines Hilbertraumes $H$ besitzt genau ein Element minimaler Norm, d.h. $\exists ! x^* \in M$ mit $\norm{x^*}=\min_{x\in M} \norm{x}$.
\end{lemma}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{folgs}[Bestapproximation]\index{Bestapproximation|see{Lemma}}\index{Lemma!Bestapproximation}
Ist $H$ ein (Prä-) Hilbertraum und $M \subset H$ eine konvexe, abgeschlossene (und vollständige) Teilmenge. Dann existiert zu jedem $x_0 \in H$ genau ein $y \in M$ mit
\begin{gather*}
\norm{y-x_0}=\min_{z\in M} \norm{z-x_0}.
\end{gather*}
\end{folgs}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{defis}
Man nennt \highl[$\alpha(x_0,M)$|see{Abstand}]{$\alpha(x_0,M)$} $:=\min_{z\in M} \norm{z-x_0}$ den \highl[Abstand $\alpha(x_0,M)$]{Abstand} des Punktes $x_0$ von der Menge $M$.
\end{defis}
\medskip

\begin{bem}
Die Folgerung gilt auch in vielen Banachräumen, aber nicht in allen.
\end{bem}
\medskip

\begin{bsp}
Gültig für $l_p$ mit $1<p<\infty$.\\
Ungültig für $l_1, l_\infty$. Beispielsweise besitzt die abgeschlossene Kugel $K_1((0,2))$ des $l_\infty^{2}$ unendlich viele Punkte mit minimalem Abstand zum Ursprung.
\end{bsp}
\medskip

\begin{satz}
Sei $H$ ein (Prä-) Hilbertraum und $H_1$ ein abgeschlossener (und vollständiger) linearer Teilraum von $H$. Dann kann jedes $x \in H$ eindeutig in der Form $x=x_1+x_2$ mit $x_1 \in H_1$ und $x_2 \in H_1^\bot$ dargestellt werden.
\end{satz}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{defis}
$x_1$ aus Satz 3.1.3 heißt \highl[orthogonal!Projektion]{orthogonale Projektion} von $x$ auf $H_1$.
\end{defis}
\medskip

\begin{folgs}
Ist $H_1$ ein abgeschlossener, linearer Teilraum eines Hilbertraumes $H$, so gilt $(H_1^\bot)^\bot=H_1$.
\end{folgs}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{bem}
Die Vollständigkeit (welche für einen Prähilbertraum separat für den Unterraum gefordert werden muss) ist wichtig für die Existenz der orthogonalen Projektion, wie folgendes Beispiel zeigt:
\end{bem}
\medskip

\begin{bsp}
Die Menge $\widetilde{H}:=\{ f \in C([0,1]) \sep f(1)=0 \}$ mit dem Skalarprodukt aus $C([0,1])$, also $(f,g):=\int_0^1 f(t)g(t)dt$ ist ein Hilbertraum, welcher nicht vollständig ist.\\
$T:=\{ f \in \widetilde{H} \sep \int_0^1 f(t)dt=0 \}$ ist ein in $\widetilde{H}$ abgeschlossener, linearer Teilraum (ohne Beweis). Es gilt $T^\bot = \{ 0 \}$, aber $T \subsetneqq \widetilde{H}$.
\end{bsp}
\medskip

\begin{defi}
Sei $H$ ein Hilbertraum, dann definiert man:\\
(i) die \highl[orthogonal!Summe $H_1 \oplus H_2$]{orthogonale Summe} zweier zueinander orthogonaler linearer Teilräume $H_1 \bot H_2$ von $H$ durch:
\begin{gather*}
\text{\highl[$H_1 \oplus H_2$|see{orthogonal}]{$H_1 \oplus H_2$}} := \{ x \in H \sep x=x_1+x_2, \quad x_1 \in H_1, x_2\in H_2\}
\end{gather*}
(ii) die \highl[orthogonal!Summe $\bigoplus_{j=1}^{J} H_j$]{orthogonale Summe} endlich vieler paarweise orthogonaler linearer Teilräume $H_1,\ldots,H_J$ von $H$ ($H_i\bot H_j,\quad i\neq j$) durch:
\begin{gather*}
\text{\highl[$\bigoplus_{j=1}^{J} H_j$|see{orthogonal}]{$\bigoplus_{j=1}^{J} H_j$}} := \{ x \in H \sep x=\sum_{j=1}^{J} x_j, \quad x_j \in H_j\}
\end{gather*}
(iii) die \highl[orthogonal!Summe $\bigoplus_{j=1}^{\infty} H_j$]{orthogonale Summe} abzählbar unendlich vieler paarweise orthogonaler linearer Teilräume $(H_j)_{j=1}^{\infty}$ von $H$ ($H_i\bot H_j,\quad i\neq j$) durch:
\begin{gather*}
\text{\highl[$\bigoplus_{j=1}^{\infty} H_j$|see{orthogonal}]{$\bigoplus_{j=1}^{\infty} H_j$}} := \bigcap \widetilde{H}
\end{gather*}
wobei die $\widetilde{H}$ lineare Teilräume von $H$ sind, welche alle $\bigoplus_{j=1}^{J} H_j$ für $J\in \N$ enthalten.
\end{defi}
\medskip

\begin{lemmas}
Sind $H_1$ und $H_2$ lineare Teilräume eines Hilbertrames $H$ mit $H_1 \bot H_2$, so ist $H_1 \cap H_2 = \{ 0 \}$.
\end{lemmas}
\begin{proof}
Klar.
\end{proof}
\medskip

\begin{bem}
In Banachräumen gilt:\\
$H_1 \cap H_2=\{ 0\} \Rightarrow H_1 \dot{+} H_2 = \{ x \in H \sep x=x_1+x_2, \quad x_1 \in H_1, x_2 \in H_2 \}$ (direkte Summe).
\end{bem}
\medskip

\begin{lemmas}
Die Darstellung von $x\in H_1 \oplus H_2$ ist eindeutig bestimmt.
\end{lemmas}
\begin{proof}
Angenommen man hätte zwei Darstellungen\\
$x=x_1+x_2=x'_1+x'_2$ mit $x_1,x'_1\in H_1$ und $x_2,x'_2\in H_2$. Dann ist $x_1-x'_1=x'_2-x_2$, wobei $x_1-x'_1 \in H_1$ und $x_2-x'_2\in H_2$ ist (da $H_1$ und $H_2$ lineare Teilräume sind). Wegen dem vorrigen Lemma ist $x_1-x'_1=x_2-x'_2=0$, also $x_1=x'_1$ und $x_2=x'_2$.
\end{proof}
\medskip

\begin{lemma}
Sei $H$ ein Hilbertraum, dann gilt:\\
(i) Sind $H_1 \bot H_2$ lineare Teilräume von $H$, so ist $H_1 \oplus H_2$ abgeschlossen \gdw $H_1$ und $H_2$ sind abgeschlossen.\\
(ii) Sind $H_1, \widetilde{H}$ abgeschlossene, lineare Teilräume von $H$, so existiert genau ein abgeschlossener Teilraum $H_2$ von $H$ mit $\widetilde{H}=H_1 \oplus H_2$. (man schreibt auch $H_2 = \widetilde{H} \ominus H_1$)
\end{lemma}
\begin{proof}
Siehe Vorlesung.
\end{proof}

\end{bem}

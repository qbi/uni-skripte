\begin{defis}
Eine nichtleere Menge $X$ heiﬂt \highl[Raum!linearer Raum]{linearer Raum} oder \highl[Raum!Vektorraum|see{linearer Raum}]{Vektorraum} ¸ber dem Kˆrper $\K$ \gdwdef Es existieren Abbildungen $+ \colon X \times X \nach X$, sodass f¸r alle $x,y,z \in X$ gilt:
\begin{gather*}
x+y=y+x \\
(x+y)+z = x+(y+z) \\
\exists 0_X \in X \text{ sodass } x+0_X=x \\
\exists (-x) \in X \text{ sodass } x+(-x)=0_X
\end{gather*}
und $\cdot \colon \K \times X \nach X$, sodass f¸r alle $x,y \in X$ und $\lambda, \mu \in \K$ gilt:
\begin{gather*}
\lambda (x+y) = \lambda x + \lambda y \\
(\lambda + \mu) x = \lambda x + \mu x \\
(\lambda \mu ) x = \lambda (\mu x) \\
1_\K x = x
\end{gather*}
\end{defis}

\begin{defi}
Ein (reeller oder komplexer) linearer Raum $X$ heiﬂt \highl[Raum!normierter Raum]{normierter Raum} \gdwdef Es existiert eine Abbildung $p \colon X \nach [0,\infty)$, sodass f¸r alle $x \in X$ und $\lambda \in \R$ (oder $\C$) gilt:
\begin{align*}
\text{(i) } & x \neq 0 \Rightarrow p(x)>0 & \text{(Definitheit)} \\
\text{(ii) } & p(\lambda x) = \abs{\lambda} p(x) & \text{(Homogenit‰t)} \\
\text{(iii) } & p(x+y) \leq p(x) + p(y) & \text{(Dreiecksungleichung)}
\end{align*}
$p$ heiﬂt dann \highl{Norm}.
\end{defi}

\begin{bem}

Der normierte Raum selbst wird sowohl mit $(X, \norm{\cdot})$, als auch nur mit $X$ beschrieben (falls die verwendete Norm klar ist).
\end{bem}

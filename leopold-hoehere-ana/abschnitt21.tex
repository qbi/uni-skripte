%Grundbegriffe

\begin{bem}
Im Folgenden werden insbesondere die \highl[Körper $\K$]{Körper} \highl[$\K$|see{Körper}]{$\K$} $= \R$ oder $\C$ betrachtet.
\end{bem}
\medskip

\begin{defis}
Eine nichtleere Menge $X$ heißt \highl[Raum!linearer Raum]{linearer Raum} oder \highl[Raum!Vektorraum|see{linearer Raum}]{Vektorraum} über dem Körper $\K$ \gdwdef Es existieren Abbildungen $+ \colon X \times X \nach X$, sodass für alle $x,y,z \in X$ gilt:
\begin{gather*}
x+y=y+x \\
(x+y)+z = x+(y+z) \\
\exists 0_X \in X \text{ sodass } x+0_X=x \\
\exists (-x) \in X \text{ sodass } x+(-x)=0_X
\end{gather*}
und $\cdot \colon \K \times X \nach X$, sodass für alle $x,y \in X$ und $\lambda, \mu \in \K$ gilt:
\begin{gather*}
\lambda (x+y) = \lambda x + \lambda y \\
(\lambda + \mu) x = \lambda x + \mu x \\
(\lambda \mu ) x = \lambda (\mu x) \\
1_\K x = x
\end{gather*}
\end{defis}
\medskip

\begin{defis}
Eine nichtleere Teilmenge $E$ eines Vektorraums $X$ heißt \highl[Raum!Unterraum|see{linearer Teilraum}]{Unterraum} oder \highl[Raum!linearer Teilraum]{linearer Teilraum} \gdwdef Für alle $x,y \in E$ und $\lambda, \mu \in \K$ gilt $\lambda x + \mu y \in E$.
\end{defis}
\medskip

\begin{defis}
Ist $M \subset X$, so bezeichnet
\begin{gather*}
\text{\highl[$\spn{M}$|see{lineare Hülle}]{$\spn{M}$}} := \{ y\in X \sep y = \sum_{j=1}^n \lambda_j y_j \text{ mit } y_j \in M \text{ und } \lambda_j \in \K \text{ sowie } n \in \N\}
\end{gather*}
den \highl[Spann|see{lineare Hülle einer Menge}]{Spann von $M$} bzw. die \highl[lineare Hülle]{lineare Hülle von $M$} in $X$.
\end{defis}
\medskip

\begin{defis}
(i) Die endliche Menge $\{ x_1, \ldots, x_N \} \subset X$ heißt \highl[Menge!linear unabhängig (endl.)]{linear unabhängig} \gdwdef Aus \\ $\sum_{j=1}^N \lambda_j x_j = 0$ folgt $ \lambda_j = 0$ für $j=1 \ldots, N$. \\
(ii) $M \subset X$ heißt \highl[Menge!linear unabhängig (unendl.)]{linear unabhängig} \gdwdef Jede endliche Teilmenge aus $M$ ist linear unabhängig.
\end{defis}
\medskip

\begin{bem}
Die leere Menge ist per Definition linear unabhängig.\\
Jede Menge die den Nullvektor enthält muss linear abhängig sein.
\end{bem}
\medskip

\begin{defis}
Eine Teilmenge $B \subset X$ heißt \highl[Basis!algebraische]{(algebraische) Basis} oder auch \highl[Basis!Hamelbasis|see{algebraische}]{Hamelbasis} des Vektorraums $X$ \gdwdef $B$ ist linear unabhängig und $\spn{B}=X$.
\end{defis}
\medskip

\begin{bem}
Jedes Element eines Vektorraums kann also durch eine endliche Linearkombination von Basiselementen mit Koeffizienten des Körpers dargestellt werden, und zwar eindeutig!\\
Neben der Hamelbasis gibt es weitere (nicht zwingend algebraische) Basisbegriffe, wie zum Beispiel die Schauderbasis eines Banachraumes. Dabei sind dann auch abzählbare Linearkombinationen möglich. Ein Beispiel ist die Standardbasis der Einheitsvektoren in $l_p$, welche ganz offensichtlich keine Hamelbasis sein kann.
\end{bem}

\end{proof}
\end{satzs}

\begin{proof}

\end{proof}
\end{folgs}

\begin{defi}
Ein (reeller oder komplexer) linearer Raum $X$ heißt \highl[Raum!normierter Raum]{normierter Raum} \gdwdef Es existiert eine Abbildung $p \colon X \nach [0,\infty)$, sodass für alle $x \in X$ und $\lambda \in \R$ (oder $\C$) gilt:
\begin{align*}
\text{(i) } & x \neq 0 \Rightarrow p(x)>0 & \text{(Definitheit)} \\
\text{(ii) } & p(\lambda x) = \abs{\lambda} p(x) & \text{(Homogenität)} \\
\text{(iii) } & p(x+y) \leq p(x) + p(y) & \text{(Dreiecksungleichung)}
\end{align*}
$p$ heißt dann \highl{Norm}.
\end{defi}
\medskip

\begin{bem}
Man schreibt statt $p(\cdot)$ auch $\norm{\cdot}_X$, bzw. $\norm{\cdot | X }$ oder auch nur $\norm{\cdot}$ (falls der zugehörige lineare Raum klar ist) .\\
Der normierte Raum selbst wird sowohl mit $(X, \norm{\cdot})$, als auch nur mit $X$ beschrieben (falls die verwendete Norm klar ist).
\end{bem}
\medskip

\begin{defis}
Gilt in Definition 2.1.1\\
(a) nur (ii) und (iii), so heißt $p$ \highl[Norm!Halbnorm]{Halbnorm}. \\
(b) statt (iii) nur $p(x+y) \leq d(p(x) + p(y))$ mit einer Konstanten $d > 1$, so heißt $p$ \highl[Norm!Quasinorm]{Quasinorm}. \\
(c) statt (ii) $p(\lambda x) = \abs{\lambda}^q p(x)$ mit einer Konstanten $q \neq 1$ so heißt $p$ \highl[Norm!$q$-homogene Norm]{$q$-homogene Norm}.\\
Die zugehörigen Räume heißen in diesen Fällen \highl[Raum!halbnormierter Raum]{halbnormiert}, \highl[Raum!quasinormierter Raum]{quasinormiert} bzw. \highl[Raum!$q$-homogen-normierter Raum]{$q$-homogen-normiert}.
\end{defis}

\begin{defi}
Sei $X$ halbnormierter Raum und $(x_j)_{j=1}^{\infty} \subset X$ eine Folge darin.\\
(i) $(x_j)_{j=1}^{\infty}$ heißt \highl[Folge!konvergente Folge (norm. Raum)]{konvergent} \gdwdef $\exists x \in X$ und $\forall \epsilon >0 \, \exists j_0 (\epsilon)$ mit $\norm{x_j-x}<\epsilon$ für alle $j \geq j_0 (\epsilon)$.\\
(ii) $(x_j)_{j=1}^{\infty}$ heißt \highl[Folge!Cauchy-Folge (norm. Raum)]{Cauchy-Folge} (CF) \gdwdef $\forall \epsilon >0 \, \exists j_1 (\epsilon)$ mit $\norm{x_j-x_m}<\epsilon$ für alle $j,m \geq j_1 (\epsilon)$.\\
(iii) $X$ heißt \highl[vollständig!normierter Raum]{vollständig} \gdwdef Jede Cauchy-Folge konvergiert in $X$.\\
(iv) $X$ heißt \highl[Raum!Banachraum]{Banachraum} \gdwdef $X$ ist vollständiger normierter Raum.
\end{defi}

\begin{bsp}
(i) $\Q, \R$ oder auch $\C$ mit $\norm{\cdot}=\abs{\cdot}$ sind normierte Räume ($\R, \C$ sogar vollständig).\\
\\
(ii) Die diskrete Metrik auf einem Vektorraum besitzt kein Gegenstück.\\
\\
(iii) Die Arcus-Tangens-Metrik auf $\R$ ($d(x,y)=\abs{\arctan(x)-\arctan(y)}$) besitzt wegen der Homogenität der Norm auch kein Gegenstück.\\
\\
(iv) Das Gegenstück zu $\C$ mit $d(z,w)=\abs{\Re(z) - \Re(w)}$ ist die Halbnorm $\norm{\cdot}:=\abs{\Re(\cdot)}$.\\
\\
(v) $\R^n$ oder auch $\C^n$ wird für $1 \leq p <\infty$ mit \index{Norm!$p$-Norm des $\R^n$ $\norm{\cdot}_p$}\highl[$\norm{\cdot}_p$|see{Norm}]{$\norm{x}_p$} $= \norm{x \sep l_p^n} := \left( \sum_{k=1}^{n} \abs{\xi_k }^p \right)^{\frac{1}{p}}$ zum Banachraum. Die Funktion \index{Norm!Maximumsnorm des $\R^n$ $\norm{\cdot}_\infty$}\highl[$\norm{\cdot}_\infty$|see{Norm}]{$\norm{x}_\infty$} $= \norm{x \sep l_{\infty}^n} = \max_{k=1,\ldots,n} \abs{\xi_k}$ bildet die Maximumsnorm des $\R^n$ bzw. $\C^n$. Dabei ist wie üblich $x=(\xi_1,\ldots,\xi_n)$.\\
\\
(vi)+ (vii) Auch für die metrischen Räume mit der Urwald-Metrik bzw. der Eisenbahn-Metrik existiert keine Norm.\\
\\
(viii) Die Menge aller Folgen $s = \{ x \sep x=(\xi_k)_{k=1}^{\infty} \}$ ist mit komponentenweiser Addition bzw. Multiplikation mit einem Skalar ein Vektorraum. Wegen der Homogenität, welche eine Norm erfüllen müsste, existiert auch zum metrischen Raum $(s,d)$ mit $d(x,y) = \sum_{k=1}^{\infty} \frac{1}{2^k} \frac{\abs{\xi_k - \eta_k}}{1+ \abs{\xi_k - \eta_k}}$ keine Norm.\\
$l_p = \{ x \sep \norm{x|l_p} < \infty \}$ ist für $1 \leq p < \infty$ mit der Norm \index{$\norm{\cdot \sep l_p}$|see{Norm}}\highl[Norm!$l_p$-Norm $\norm{x \sep l_p}$]{$\norm{x \sep l_p}$} $= \left( \sum_{k=1}^{\infty} \abs{\xi_k}^p \right)^{\frac{1}{p}}$ ein vollständiger normierter Raum.\\
$l_\infty=\{ x \sep \norm{x|l_\infty}<\infty \}$ ist ein normierter Raum durch \index{Norm!$l_\infty$-Norm $\norm{\cdot \sep l_\infty}$}\highl[$\norm{\cdot \sep l_\infty$}|see{Norm}]{$\norm{x \sep l_\infty}$} $=\sup_k \abs{\xi_k}$.\\
Die Mengen der konvergenten Folgen $c$, bzw. der Nullfolgen $c_0$ sind lineare Teilräume von $l_\infty$.\\
Für $0<p<1$ ist $\left( \sum_{k=1}^{\infty} \abs{\xi_k}^p \right)^{\frac{1}{p}}$ nur eine Quasinorm auf $l_p$. Es gilt die scharfe Abschätzung $\norm{x+y|l_p} \leq 2^{\frac{1}{p}-1} (\norm{x|l_p}+\norm{y|l_p})$.\\
\\
(ix) Die stetigen Funktionen $C([a,b])$ sind mit der Norm \index{$\norm{f}_{\infty}$|see{Norm}}\highl[Norm!sup-Norm stetiger Fkt.'n $\norm{f}_{\infty}$]{$\norm{f}_{\infty}$} $=\sup_{t \in [a,b]} \abs{f(t)}<\infty$ (da stetige Funktionen auf abgeschlossenen Mengen beschränkt sind) vollständig.\\
Die Funktion \index{$\norm{f}_p$|see{Norm}}\highl[Norm!int-Norm setiger Fkt.'n $\norm{f}_p$]{$\norm{f}_p$} $:= \left( \int_a^b \abs{f(t)}^p dt \right)^{\frac{1}{p}}$ ist für $1 \leq p <\infty$ ebenfalls eine Norm auf $C([a,b])$. Mit ihr ist der Raum jedoch nicht vollständig.\\
\\
(x) Die Menge \highl[$C^1(M)$|see{Funktionenräume}]{$C^1([a,b])$} der \highl[Funktionenräume!stetig diff'bare Fkt.'n $C^1(M)$]{stetig differenzierbaren} Funktionen über $[a,b]$ ist ein Untervektorraum von $C([a,b])$. Mit der Supremumsnorm aus (ix) ist der Raum jedoch nicht vollständig. Dazu sei beispielsweise die Funktionenfolge $(f_j)_{j=1}^{\infty} \subset C^1([-1,1])$ mit $f_j(t):= \left( t^2 + \frac{1}{j} \right)^{\frac{1}{2}}$ gegeben. Bezüglich der Supremumsnorm handelt es sich dabei um eine Cauchy-Folge, deren Grenzfunktion $f(t)=\abs{t}$ zwar stetig, aber nicht differenzierbar in $t=0$ ist.\\
Man kann auf $C^1([a,b])$ weitere Normen definieren: $\norm{f}:=\max \{ \norm{f}_{\infty}, \norm{f'}_{\infty} \}$ oder auch $\abs{\norm{f}}:=\norm{f}_{\infty}+\norm{f'}_{\infty}$. Diese Normen sind äquivalent mit $\norm{f} \leq \abs{\norm{f}} \leq 2 \norm{f}$ und $C^1([a,b])$ ist damit jeweils vollständig. Sei dazu $(f_j)_{j=1}^{\infty}$ eine beliebige Cauchy-Folge in $(C^1([a,b]), \norm{\cdot})$. Dann sind die Folgen $(f_j)_{j=1}^{\infty}$ und $(f'_j)_{j=1}^{\infty}$ Cauchy-Folgen in $(C([a,b]), \norm{\cdot}_{\infty})$. Da dieser Raum vollständig ist (dazu siehe (ix)) existieren Funktionen $f,g \in C([a,b])$ mit $f_j \nach f$ und $f'_j \nach g$ (gleichmäßig, d.h. bezüglich $\norm{\cdot}_{\infty}$). In den Vorlesungen Analysis 1 und 2 wurde gezeigt, dass dann $g=f'$ gilt, d.h. $f \in C^1([a,b])$.\\
\\
(xi) Sei $\Omega \subset \R^n$ offen und $k \in \N_0$. Dann ist
\begin{align*}
C^k(\overline{\Omega}) := \{ & \phi \sep \phi \colon \Omega \nach \C \text{ mit } \phi \, k \text{-mal stetig diff'bar und für } \abs{\alpha} \leq k \\
                             & \text{ ist } D^{\alpha} \phi \text{ stetig auf } \overline{\Omega} \text{ fortsetzbar, sowie } \norm{\phi}<\infty \}
\end{align*}
ein Vektorraum und mit $\norm{\phi | C^k(\overline{\Omega})}:= \sum_{\abs{\alpha} \leq k} \norm{D^{\alpha} \phi}_{\infty} = \sum_{\abs{\alpha} \leq k} \sup_{x \in \overline{\Omega}} \abs{D^{\alpha} \phi(x)}$ ein vollständiger normierter Raum (Banachraum).\\
\\
(xii) Bezeichne $\mathbb{D}:= \{ z \in \C \sep \abs{z}<1 \}$ den offenen komplexen Einheitskreis und $H^{\infty}$ die Menge aller beschränkten, holomorphen Funktionen $\mathbb{D} \nach \C$. Mit der Norm $\norm{f}_{\infty}:=\sup_{\abs{z}<1} \abs{f(z)} < \infty$ (da beschränkt) ist $(H^{\infty}, \norm{\cdot}_{\infty})$ ein Banachraum. Zum Beweis wähle man eine Cauchy-Folge $(f_j)_{j=1}^{\infty} \subset H^{\infty}$. Dann gibt es eine stetige Funktion $f$, mit $f_j \nach f$ gleichmäßig (in $\norm{\cdot}_\infty$). Mit Hilfe der Funktionentheorie kann man zeigen, dass $f$ dann holomorph ist und somit in $H^{\infty}$ liegt.
\end{bsp}

\begin{bem}
Zur Betrachtung der $L_p$-Räume nun ein paar Definitionen aus der Maß- und Integrationstheorie.
\end{bem}
\medskip

\begin{defis}
(i) Ist die Menge $T \neq \leer$, so heißt $\Sigma \subset \Pot(T)$ \highl[$\sigma$-Algebra $\Sigma$]{$\sigma$-Algebra} über $T$ \gdwdef
\begin{align*}
\text{(a) } & \leer \in \Sigma \\
\text{(b) } & E \in \Sigma \Rightarrow T \setminus E \in \Sigma\\
\text{(c) } & E_j \in \Sigma, \forall j \in \N \Rightarrow \bigcup_{j=1}^{\infty} E_j \in \Sigma
\end{align*}
(ii) Für eine $\sigma$-Algebra $\Sigma$ heißt eine Abbildung $\mu \colon \Sigma \nach [0, \infty]$ \highl{Maß} \gdwdef
\begin{align*}
\text{(a) } & \mu(\leer) = 0 \\
\text{(b) } & \mu \text{ ist \highl[Maß!$\sigma$-additiv]{$\sigma$-additiv} } \left( \text{d.h. } E_j \in \Sigma \text{ paarweise disjunkt } \Rightarrow \mu(\bigcup_{j=1}^{\infty} E_j) = \sum_{j=1}^{\infty} \mu(E_j) \right)
\end{align*}
(iii) $(T, \Sigma, \mu)$ heißt \highl[Maßraum!Maßraum $(T, \Sigma, \mu)$]{Maßraum} \gdwdef $\Sigma$ ist $\sigma$-Algebra über der Menge $T \neq \leer$ und $\mu$ ein Maß auf $\Sigma$.\\
(iv) Ein Maßraum (bzw. das Maß selbst) heißt \highl[Maßraum!endlich]{endlich} \gdwdef $\mu(T) < \infty$.\\
(v) Ein Maßraum (bzw. das Maß selbst) heißt \highl[Maßraum!$\sigma$-endlich]{$\sigma$-endlich} \gdwdef $\exists (E_j)_{j=1}^{\infty} \subset \Sigma$ sodass für alle $j\in \N$ gilt: $\mu(E_j) < \infty$ und $\bigcup_{j=1}^{\infty} E_j = T$.
\end{defis}
\medskip

\begin{bem}
Ist $(T,d)$ ein metrischer oder $(T,\tau)$ ein topologischer Raum, so existieren dort offene Mengen. Das kleinste Mengensystem mit den Eigenschaften einer $\Sigma$-Algebra, welches alle offenen Mengen enthält bezeichnet man als $\sigma$-Algebra der Borelmengen. Als Maß wird dann häufig das Lebesgue-Maß $\lambda$ verwendet.
\end{bem}
\medskip

\begin{defis}
Sei $(T, \Sigma, \mu)$ ein Maßraum und $1 \leq p < \infty$.\\
(i) Eine Funktion $f \colon T \nach \R$ (oder $\C$) heißt \highl[meßbare Funktion]{meßbar} \gdwdef für alle $A \subset \R$ gilt $f^{-1}(A) \in \Sigma$ (vollst. Urbild)\\
(ii) Man definiert das \highl{Lebesgue-Integral} solcher Funktionen: $\int_T f(x) d\mu(x)$ (über Treppenfunktionen).\\
(iii) \highl{$\widehat{L_p}(T, \Sigma, \mu, \C)$} $:=\{ f \colon T \nach \C \sep f$ meßbar mit $\int_T \abs{f(x)}^p d\mu(x) < \infty\}$
\end{defis}
\medskip

\begin{bem}
Die Menge $\widehat{L_p}:=\widehat{L_p}(T, \Sigma, \mu, \C)$ bildet für $1 \leq p < \infty$ einen linearen Raum (Vektorraum) und die Abbildung $\norm{\cdot}_{p}^{*}:= \left( \int_T \abs{\cdot(x)}^p d\mu(x) \right)^{\frac{1}{p}} < \infty$ definiert eine Halbnorm darüber, sodass $(\widehat{L_p}, \norm{\cdot}^*)$ ein vollständiger halbnormierter Raum ist. Definiert man $N=N_p$ wie in Lemma 2.1.3 so bilden die Räume \index{Funktionenräume!$p$-integrierbare Fkt.'n $L_p$} \highl[$L_p$|see{Funktionenräume}]{$L_p$} $:= \widehat{L_p}/_{N_p}$ mit der Norm \index{$\norm{\cdot \sep L_p}$|see{Norm}}\highl[Norm!$L_p$-Norm $\norm{\cdot \sep L_p}$]{$\norm{[\cdot]}$} $:=\left( \int_T \abs{\cdot (x)}^p d\mu(x) \right)^{\frac{1}{p}}$ für $1 \leq p < \infty$ einen Banachraum, wobei $f,g \in [f] \in L_p \gdw f=g$  $\mu$-f.ü. Man schreibt oft $f \in L_p$, meint aber eigentlich einen Vertreter $f \in [f]$ einer Äquivalenzklasse aus $L_p$.
\end{bem}
\medskip

\begin{defis}
Sei $(T, \Sigma, \mu)$ ein Maßraum und $f \colon T \nach \C$ meßbar. Dann bezeichnet
\begin{gather*}
\text{\highl[$\esssup{x \in T} \abs{f(x)}$|see{wesentliches Supremum}]{$\esssup{x \in T} \abs{f(x)}$}} := \inf_{\substack{N \in \Sigma,\\ \mu(N)=0}} \{ \sup_{x\in T\setminus N} \abs{f(x)} \, \}
\end{gather*}
das \highl[wesentliches Supremum]{wesentliche Supremum} (essentielles Supremum).
\end{defis}
\medskip

\begin{bem}
Für (abschnittsweise) stetige Funktionen ergibt sich die Identität mit dem klassischen Supremum.\\
Im Grunde ist das wesentliche Supremum die kleinste obere Schranke für den Funktionswert einer Menge von Funktionen, welche sich nur auf einer Nullmenge unterscheiden.\\
Definiert man $\norm{\cdot}_{\infty}^* := \esssup{x\in T} \abs{\cdot (x)}$, so kann man analog zum obigen Vorgehen auch den Raum \highl[$L_{\infty}$|see{Funktionenräume}]{$L_{\infty}$}\index{Funktionenräume!$L_{\infty}$} definieren.
\end{bem}
\medskip

\begin{bsp}
(i) $T=\R$ oder $T=I \subset \R$ (Intervall) und $\Sigma$ die $\sigma$-Algebra der Borelmengen, sowie $\mu = \lambda$ das Lebesgue-Maß.\\
\\
(ii) $T=\Omega \subset \R^n$ und wieder $\Sigma$ die $\sigma$-Algebra der Borelmengen, sowie $\mu = \lambda_n$ das $n$-dimensionale Lebesgue-Maß.\\
\\
(iii) $T=\N$, $\Sigma = \Pot(\N)$, sowie $\mu(\cdot) = \#(\cdot)$ das Zähl-Maß. Dann ist $L_p(\N, \Pot(\N), \mu)=l_p$.\\
\\
(iv) $L_{\infty}$ ist Banachraum.\\
\\
(v) Für $0<p<1$ ist $L_p$ nur Quasi-Banachraum.
\end{bsp}

\begin{defi}
Sei $X$ ein linearer Raum und $\norm{\cdot}_1, \norm{\cdot}_2$ (halb-) Normen auf $X$.\\
(i) $\norm{\cdot}_1$ heißt \highl[Norm!schwächer]{schwächer} als $\norm{\cdot}_2$ \gdwdef $\exists c>0$, sodass $\forall x\in X$ gilt: $\norm{x}_1 \leq c \norm{x}_2$.\\
(ii) $\norm{\cdot}_1$ ist \highl[Norm!äquivalent]{äquivalent} zu $\norm{\cdot}_2$ \gdwdef $\exists c_1,c_2>0$, sodass $\forall x\in X$ gilt:\\
$c_1 \norm{x}_2 \leq \norm{x}_1 \leq c_2 \norm{x}_2$
\end{defi}
\medskip

\begin{bem}
Normen sind äquivalent, falls jeweils die eine schwächer ist als die andere.\\
Damit hat man eine Halbordnung bzw. eine Äquivalenzrelation auf Normen definiert.
\end{bem}
\medskip

\begin{satz}
Alle Normen auf einem endlich-dimensionalen linearen Raum sind äquivalent.
\begin{proof}
Siehe Vorlesung.
\end{proof}
\end{satz}

\begin{bem}
Sie führen also zur gleichen Konvergenz und zur gleichen Topologie auf $X$.
\end{bem}

\begin{defi}
Sei $X$ ein normierter / metrischer / topologischer Raum.\\
(i) $M \subset X$ heißt \highl[Raum!dicht (metr./norm./topol. Raum)]{dicht} in $X$ \gdwdef $\overline{M}=X$.\\
(ii) $X$ heißt \highl[Raum!separabel (metr./norm./topol. Raum)]{separabel} \gdwdef Es existiert eine abzählbare, dichte Teilmenge $M\subset X$.
\end{defi}
\medskip

\begin{lemmas}
Sei $X$ ein normierter / metrischer Raum. Dann sind folgende Aussagen äquivalent:\\
(i) $M$ liegt dicht in $X$.\\
(ii) $\overline{M}=X.$\\
(iii) $\forall x\in X$ existiert $(y_j)_{j=1}^{\infty} \subset M$ mit $y_j \nach x$.\\
(iv) $\forall x \in X \, \forall \epsilon >0$ existiert stets $y(x,\epsilon) \in M$ mit $d(x,y)<\epsilon$.
\end{lemmas}
\begin{proof}
Klar.
\end{proof}

\begin{bsp}
$(\R^n, \norm{\cdot}_p), (\C^n, \norm{\cdot}_p)$ oder jeder andere beliebige endlich-dimensionale Raum $X$ mit beliebiger Norm ist separabel. Zum Beweis wähle eine Basis $B:=\{ y_1, \ldots y_n \}$ von $\R^n$ und definiere $M:=\{ x \in \R^n \sep x = \sum_{k=1}^{n} \lambda_k y_k$ mit $\lambda_k \in \Q \}$. Da $\Q$ und $\Q^n$ abzählbar sind und dicht liegen, ist auch $M$ abzählbar und liegt dicht im $\R^n$.
\end{bsp}

$X$ ist separabel \gdw Es existiert eine abzählbare Menge $A \subset X$ mit $\overline{\spn{A}}=X$.
\end{lemma}
\begin{proof}
Siehe Vorlesung.
\end{proof}

\begin{bsp}
(i) Für $1 \leq p < \infty$ ist $(l_p, \norm{\cdot}_p)$ separabel.\\
Zum Beweis setze $A:=\{ e_j \sep e_j=(0,\ldots,0,1,0,\ldots) \quad j$-ter Einheitsvektor $\}$. Dann ist $\overline{\spn{A}}=l_p$ (Abschluss in $l_p$), denn es ist
\begin{gather*}
x=(\xi_k)_{k=1}^{\infty} \in l_p \gdw \norm{x-\sum_{k=1}^N \xi_k e_k}_p = \left( \sum_{k=N+1}^{\infty} \abs{\xi_k}^p \right)^{\frac{1}{p}} \nach 0, \text{ für } N \nach \infty
\end{gather*}
Also ist $l_p$ separabel nach Lemma 2.1.5. Sucht man noch eine abzählbare, dichte Teilmenge von $l_p$, so bietet sich die Menge $M\subset F$ der finiten Folgen, mit rationalen Komponenten an.\\
\\
(ii) $(c_0, \norm{\cdot}_{\infty})$ ist separabel (analog).\\
\\
(iii) $(l_{\infty}, \norm{\cdot}_{\infty})$ ist nicht separabel.\\
\\
(iv) $(C([a,b]), \norm{\cdot}_{\infty})$ ist ein separabeler Banachraum.\\
Zeige dazu, dass die Menge der endlichen Linearkombinationen von Polynomen\\
$\spn{\{ x^n | n \in \N \}}$ dicht in $C([a,b])$ liegt (Bernsteinpolynome, Weierstrass'scher Approximationssatz).\\
\\
(v) $(L_{\infty}([a,b]), \norm{\cdot}_{\infty})$ ist nicht separabel.\\
\\
(vi) Für $1\leq p < \infty$ ist $(L_p([a,b]), \norm{\cdot}_p)$ separabel.\\
Dazu zeigt man, dass $C([a,b])$ dicht in $L_p([a,b])$ liegt und nach (iv) selbst separabel ist.
\end{bsp}

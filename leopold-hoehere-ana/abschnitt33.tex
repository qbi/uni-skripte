%======== Abschnitt 3.3 ============
%Lineare Operatoren im Hilbertraum

\begin{bem}
Lineare und beschränkte (stetige) Operatoren in Banachräumen wurden bereits in Abschnitt 2.3 betrachtet.\\
Abschnitt 2.4 beschäftigte sich mit linearen Funktionalen $x' \in X'=\L(X,\C)$. Insbesondere waren Aussagen der Form $(l_p)' \cong l_{p'}$ mit $\frac{1}{p}+\frac{1}{p'}=1$ für $1 \leq p < \infty$ von Interesse.\\
Die folgenden Betrachtungen gelten wieder auch für nicht separable Hilberträume.
\end{bem}
\medskip

\begin{satz}[Darstellungssatz von Frechét-Riesz]\index{Satz!von Frechét-Riesz (Darstellungssatz)}
Sei $l$ ein lineares und stetiges Funktional über einem Hilbertraum $H$. Dann gilt:\\
(i) $\exists! y \in H$ mit $l(x) = (x,y)$, für alle $x \in H$ und $\norm{l \sep \L(H, \C)} = \norm{y \sep H}$.\\
(ii) Die Abbildung $T \colon H \nach H'=\L(H,\C)$ mit $y \mapsto (\cdot,y)$ ist bijektiv, isometrisch und konjugiert linear (d.h. $T(\lambda y) = \overline{\lambda} T(y)$).
\end{satz}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{bem}
Damit gilt stets $H \cong H'$. Insbesondere sind daher alle Hilberträume reflexiv.
\end{bem}
\medskip

\begin{defi}
Sei $H$ ein Hilbertraum über $\C$.\\
(i) Eine Abbildung $L \colon H \times H \nach \C$ heißt beschränkte / stetige \highl[Sesquilinearform!beschränkte / stetige]{Sesquilinearform} \gdwdef Für alle $x,x_1,x_2,y,y_1,y_2\in H$ und $\lambda_1,\lambda_2,\mu_1,\mu_2\in \C$ gilt:
\begin{align*}
\text{(a) } & L(\lambda_1 x_1+\lambda_2 x_2, y) = \lambda_1 L(x_1,y) + \lambda_2 L(x_2,y) & \text{(Linearität im 1. Argument)} \\
\text{(b) } & L(x,\mu_1 y_1+\mu_2 y_2) = \overline{\mu_1} L(x,y_1) + \overline{\mu_2} L(x,y_2) & \text{(konj. Linearität im 2. Argument)} \\
\text{(c) } & \exists c \geq 0 \text{ mit } |L(x,y)| \leq c \norm{x} \norm{y} & \text{(Beschränktheit)}
\end{align*}
(ii) Die \highl[Sesquilinearform!Norm $\norm{L}$|see{Norm}]{Norm einer} \highl[Norm!Sesqulinearform $\norm{L}$]{Sesquilinearform} $L$ ist definiert durch
\begin{gather*}
\text{\highl[$\norm{L}$|see{Norm}]{$\norm{L}$}}:=\sup_{\stackrel{\norm{x}\leq 1}{\norm{y}\leq 1}} |L(x,y)|.
\end{gather*}
\end{defi}
\medskip

\begin{bem}
Eine mögliche Verallgemeinerung wäre eine Abbildung $L \colon H_1 \times H_2 \nach \C$, für Hilberträume $H_1,H_2$.\\
"`Sesqui"' steht für $1\frac{1}{2}$ - zwischen Linearform (lineare Funktionale) und Bilinearform.
\end{bem}
\medskip

\begin{bsp}
Jedes Skalarprodukt ist eine Sesquilinearform.
\end{bsp}
\medskip

\begin{lemma}
Sei $L$ eine stetige Sesquilinearform auf einem Hilbertraum $H$. Dann gibt es zwei eindeutig bestimmte Operatoren $T,S\in\L(H)$ mit:
\begin{align*}
& L(x,y) = (Tx,y) = (x,Sy), \quad \forall x,y\in H\\
& \norm{L} = \norm{T \sep \L(H)} = \norm{S \sep \L(H)}.
\end{align*}
\end{lemma}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{defi}
Ist $H$ ein Hilbertraum und $T\in \L(H)$. Dann heißt \highl[$T^*$|see{Operator}]{$T^*$} $\in \L(H)$ der zu $T$ \highl[Operator!adjungiert $T^*$]{adjungierte Operator} \gdwdef Es gilt $(Tx,y)=(x,T^*y)$ für alle $x,y\in H$.
\end{defi}
\medskip

\begin{lemmas}
Ist $H$ ein Hilbertraum, so existiert zu jedem $T\in \L(H)$ stets der adjungierte Operator $T^*\in \L(H)$ und ist eindeutig bestimmt.
\end{lemmas}
\begin{proof}
Ist der Operator $T\in \L(H)$ vorgegeben, so definiere die stetige Sesquilinearform $L$ durch $L(x,y):=(Tx,y)$, für $x,y\in H$ (stetig, da $T$ und $(\cdot,\cdot)$ stetig). Nach Lemma 3.3.1 existiert dann ein eindeutiges $S \in \L(H)$, mit $L(x,y)=(Tx,y)=(x,Sy)$ für alle $x,y\in H$. Setze $T^*:=S$.
\end{proof}
\medskip

\begin{bem}
Die Definition setzt vorraus, dass $T$ auf ganz $H$ definiert und dort beschränkt ist.
\end{bem}
\medskip

\begin{defis}
Ein Operator $T\in \L(H)$ über einem Hilbertraum $H$ heißt \highl[Operator!selbstadjungiert]{selbstadjungiert} \gdwdef $T=T^*$
\end{defis}
\medskip

\begin{bem}
Unbeschränkte Operatoren sind nur sinnvoll auf Teilmengen eines Hilbertraumes. Beispielsweise $\frac{d^2}{dt^2} \colon f \mapsto f''$ mit $D(\frac{d^2}{dt^2})=\{ f \in L_2(\R) \sep f''\in L_2(\R) \} \subset L_2(\R)$. Für $(\frac{d^2}{dt^2} f,g)=(f,\frac{d^2}{dt^2} g)$ muss $f\in D(\frac{d^2}{dt^2})$ und $g\in D(\frac{d^2}{dt^2})$ mit Randbedingungen gelten. Dies ist dann nur "`Symmetrie"', die Selbstadjungiertheit verlangt noch gleiche Definitionsgebiete.
\end{bem}
\medskip

\begin{bem}
Ist $T\in \L(H_1,H_2)$ für Hilberträume $H_1,H_2$, so ist die Definition eines adjungierten Operators $T^*\in \L(H_2,H_1)$ mit $(Tx,y)_{H_2} = (x,T^*y)_{H_1}$ für $x\in H_1, y\in H_2$ möglich.
\end{bem}
\medskip

\begin{satz}[Eigenschaften des adjungierten Operators]
Sind $T,T_1,T_2$ lineare, beschränkte Operatoren über einem Hilbertraum $H$ und $T^*,T_1^*,T_2^*$ die zugehörigen adjungierten Operatoren. Dann gilt für $\lambda_1,\lambda_2\in \C$:
\begin{align*}
\text{(i) }   & \norm{T}=\norm{T^*} \\
\text{(ii) }  & (\lambda_1 T_1+ \lambda_2 T_2)^* = \overline{\lambda_1} T_1^* + \overline{\lambda_2} T_2^* \\
\text{(iii) } & (T_1 \circ T_2)^* = T_2^* \circ T_1^* \\
\text{(iv) }  & \exists T^{-1} \in \L(H) \Rightarrow \exists (T^*)^{-1}=(T^{-1})^* \in \L(H) \\
\text{(v) }   & (T^*)^* = T
\end{align*}
\end{satz}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{satz}
Sei $H$ ein Hilbertraum und $T\in \L(H)$. Dann gilt:
\begin{gather*}
H=\overline{\im T} \oplus \ker{T^*}=\overline{\im{T^*}} \oplus \ker T.
\end{gather*}
\end{satz}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{bem}
Für selbstadjungierte Operatoren gilt nach Satz 3.3.3 $H=\overline{\im T} \oplus \ker T$.
\end{bem}
\medskip

\begin{defi}[Projektions-Operatoren / Orthogonale Projektion]
Sei $H_1$ ein abgeschlossener, linearer Teilraum eines Hilbertraumes $H$ mit $\{0\} \subsetneqq H_1 \subsetneqq H$. Dann existieren eindeutig bestimmte $x_1\in H_1, x_2\in H_1^\bot$ mit $x=x_1+x_2, \quad \forall x \in H$.\\
Definiere die (\highl[orthogonal!Projektion]{orthogonale}) \highl[Operator!orthogonale Projektion $Px$]{Projektion} $Px$ von $x$ auf $H_1$ durch $Px:=x_1, \quad \forall x \in H$.
\end{defi}
\medskip

\begin{bem}
In den Sonderfällen $H_1=\{ 0 \}$ oder $H_1=H$ folgt unmittelbar $P=$ \highl[$\0$|see{Nulloperator}]{$\0$} (\highl[Operator!Nulloperator $\0$]{Nulloperator}) bzw. $P=I_H$ (Identität). Damit ist die orthogonale Projektion für alle abgeschlossenen Unterräume $H_1$ eines Hilbertraumes $H$ definiert.
\end{bem}
\medskip

\begin{lemmas}
Sei $H_1$ ein abgeschlossener, linearer Teilraum eines Hilbertraumes $H$. Die orthogonale Projektion $P$ auf $H_1$ ist eindeutig bestimmt und es gilt:
\begin{align*}
\text{(i) }   & P \text{ ist linear}\\
\text{(ii) }  & P \text{ ist beschränkt mit } \norm{P \sep \L(H)}=1 \\
\text{(iii) } & \im P = H_1 \text{ und } \ker P = H_1^\bot
\end{align*}
\end{lemmas}
\begin{proof}
Die Fälle der obigen Bemerkung sind trivial, daher sei o.B.d.A. $\{0\} \subsetneqq H_1 \subsetneqq H$.\\
(i) Seien $x=x_1+x_2$ und $y=y_1+y_2$ mit $x_1,y_1\in H_1$, $x_2,y_2\in H_1^\bot$ (eindeutig) und $\lambda,\mu\in \C$. Dann ist $\lambda x+\mu y = (\lambda x_1+\mu y_1)+(\lambda x_2+\mu y_2) \in H_1 \oplus H_1^\bot$. Offensichtlich gilt damit $P(\lambda x+\mu y)=\lambda x_1+\mu y_1=\lambda P(x)+\mu P(y)$.\\
(ii) Mit dem Satz des Pythagoras folgt\\
$\norm{x}^2=\norm{x_1+x_2}^2=\norm{x_1}^2+\norm{x_2}^2=\norm{Px}^2+\norm{x_2}^2 \geq \norm{Px}^2$. Daher ist $\norm{P} \leq 1$. Also ist $P$ beschränkt.\\
Andererseits ist für $x_1\in H_1\setminus \{ 0 \}$ die Darstellung $x_1=x_1+0$ eindeutig. Daher folgt $\norm{Px_1}=\norm{x_1}$, also $\norm{P}\geq 1$. Zusammengefasst gilt daher $\norm{P \sep \L(H)}=1$.\\
(iii) gilt offensichtlich.
\end{proof}

\end{gather*}
\end{satz}

\end{proof}
\medskip

\begin{lemmas}
Ist $H$ ein Hilbertraum und $P\in \L(H)$, so gilt:
\begin{gather*}
P \text{ ist Orthogonalprojektion} \gdw (I-P) \text{ ist Orthogonalprojektion}.
\end{gather*}
\end{lemmas}
\begin{proof}
Ist $P$ eine Projektion bezüglich $\widetilde{H}$, so ist $(I-P)$ eine Projektion auf $\widetilde{H}^\bot$ (siehe Beweis Satz 3.3.4). Für die Rückrichtung setze $\widehat{P}:=I-P$ und wende bereits Bewiesenes an. Damit folgt $I-\widehat{P}=I-(I-P)=I-I+P=P$ ist Projektion.
\end{proof}

\begin{bem}
Weitere Eigenschaften von orthogonalen Projektionsoperatoren finden sich in Übungsserie 11 / Aufgabe 1.\\
Betrachte im Folgenden finite (ausgeartete) Operatoren $K$ auf Hilberträumen $H$, d.h. $K\in \L(H)$ mit $\dim(\im K)<\infty$.
\end{bem}

\begin{lemma}
Sei $K$ ein finiter Operator über einem Hilbertraum $H$. Dann ist $K^*$ ebenfalls finit und es gilt $\dim(\im K)=\dim(\im{K^*}):=L$. Weiter existiert eine ONB $\{ e_1, \ldots e_L \}$ von $\im K$, sowie linear unabhängige Elemente $\{ e_1^*,\ldots,e_L^*\}$ mit $\spn{\{ e_1^*,\ldots,e_L^*\}}=\im{K^*}$ (Basis). Es gilt ferner
\begin{align*}
Kx=\sum_{l=1}^L (x,e_l^*) e_l \quad \text{ und } \quad K^*y=\sum_{l=1}^L (y,e_l) e_l^*, \quad \text{ für } x,y \in H
\end{align*}
\end{lemma}

\end{proof}

\begin{bem}
Die Elemente $\{ e_1^*,\ldots,e_L^*\}$ in Lemma 3.3.2 müssen nicht mehr orthogonal sein, wie folgendes Beispiel zeigt.
\end{bem}
\medskip

\begin{bsp}
Gegeben sei der Operator $K \colon \R^3 \nach \R^3$, definiert durch die Matrix 
$\begin{pmatrix}
2 & 0 & 2 \\
1 & 0 & 0 \\
0 & 0 & 0
\end{pmatrix}$.\\
Es ist $Kx=(2 \xi_1 + 2 \xi_3, \xi_1, 0)^T$, also gilt $\dim K = 2$ und $\ker K = \spn{\{ (0,1,0)^T \}}$. Die Vektoren $e_1=(1,0,0)^T, e_2=(0,1,0)^T$ bilden eine ONB in $\im K$.\\
Der zu $K$ adjungierte Operator $K^* \colon \R^3 \nach \R^3$ ist definiert durch die Matrix 
$\begin{pmatrix}
2 & 1 & 0 \\
0 & 0 & 0 \\
2 & 0 & 0
\end{pmatrix}$ (transponierte Matrix).
Es ist $K^*y=(2 \eta_1 + \eta_2, 0, 2 \eta_1)^T$, also gilt $\dim{K^*} = 2$ und $\ker{K^*} = \spn{\{ (0,0,1)^T \}}$. Die Vektoren $e_1^*=K^*e_1=(2,0,2)^T$ und\\
$e_2^*=K^*e_2=(1,0,0)^T$ sind dann zwar linear unabhängig und bilden eine Basis in $\im{K^*}$, aber sie sind nicht orthogonal zueinander.
\end{bsp}

Sei $H$ ein Hilbertraum, dann ist\\
(i) $K\in \L(H)$ kompakt \gdw $\exists (K_j)_{j=1}^\infty \subset \L(H)$ finit mit $\norm{K-K_j \sep \L(H)} \nach 0$\\
(ii) $K\in \L(H)$ kompakt \gdw $K^*$ kompakt.
\end{satz}

\end{proof}

\begin{defi}
Sei $H$ ein Hilbertraum und $T\in\L(H)$.\\
(i) $T$ heißt \highl[Operator!unitär]{unitär} \gdwdef $T$ ist invertierbar und es gilt $T^{-1}=T^*$\\
(ii) $T$ heißt \highl[Operator!selbstadjungiert]{selbstadjungiert} \gdwdef Es gilt $T=T^*$ \\
(iii) $T$ heißt \highl[Operator!normal]{normal} \gdwdef Es gilt $T\circ T^*=T^*\circ T$
\end{defi}
\medskip

\begin{bem}
Auch andere Definitionen sind für diese Begriffe möglich. Sie sollen hier als Folgerung notiert werden.
\end{bem}
\medskip

\begin{folgs}
Sei $H$ Hilbertraum und $T\in \L(H)$. Dann gilt:
\begin{align*}
T \text { unitär} & \gdw (Tx,Ty)=(x,y) \quad \forall x,y\in H \\
T \text{ selbstadjungiert} & \gdw (Tx,y)=(x,Ty) \quad \forall x,y\in H\\
T \text { normal} & \gdw (Tx,Ty)=(T^*x,T^*y) \quad \forall x,y\in H
\end{align*}
\end{folgs}
\begin{proof}
Seien $x,y$ beliebig in $H$. Wegen den Eigenschaften des adjungierten Operators gilt:\\
(i) $(Tx,Ty)=(x,T^*Ty)=(x,y)$, wegen $T^*=T^{-1}$.\\
(ii) $(Tx,y)=(x,T^*y)=(x,Ty)$, wegen $T^*=T$.\\
(iii) $(Tx,Ty)=(x,T^*Ty)=(x,TT^*y)=(T^*x,T^*y)$, wegen $TT^*=T^*T$.
\end{proof}
\medskip

\begin{bem}
Offensichtlich sind sowohl unitäre, als auch selbstadjungierte Operatoren normal. Die Normalität eines Operators ist also die schwächste dieser drei Eigenschaften. Überdies sind unitäre Operatoren isometrisch (norm- und winkelerhaltend).
\end{bem}

\begin{defis}
Sei $H$ ein Hilbertraum und $T\in \L(H)$ selbstadjungiert.\\
$T$ heißt \highl[Operator!positiv $T \geq 0$]{positiv} (schreibe \highl[$T \geq 0$|see{Operator}]{$T \geq 0$}) \gdwdef $(Tx,x) \geq 0\quad \forall x\in H$
\end{defis}
\medskip

\begin{lemmas}
Jede orthogonale Projektion $P$ über einem Hilbertraum $H$ ist positiv.
\end{lemmas}
\begin{proof}
Sei $x\in H$ beliebig. Dann ist $(Px,x)=(P^2x,x)=(Px,P^*x)=(Px,Px)\geq 0$, wegen $P=P^2$ und $P=P^*$.
\end{proof}

\begin{bem}
Auf der Menge $M$ der selbstadjungierten Operatoren und Projektionen über einem Hilbertraum $H$ kann man nun eine Halbordnung definieren:
\begin{gather*}
T \leq S \gdwdef \forall x \in H \text{ gilt: } (Tx,x)\leq (Sx,x)
\end{gather*}
Äquivalent dazu ist $S-T \geq 0$.
\end{bem}
\medskip

Sei $H$ ein Hilbertraum und $T\in \L(H)$ selbstadjungiert (allgemeiner: normal). Dann gilt:
\begin{gather*}
\norm{T\sep \L(H)} = \sup_{\norm{x}\leq 1} |(Tx,x)|=\sup_{\norm{x}= 1} |(Tx,x)|
\end{gather*}
\end{satz}
\begin{proof}
Siehe Vorlesung.
\end{proof}

\begin{folgs}
Sei $H$ ein Hilbertraum und $T\in \L(H)$ selbstadjungiert. Gilt für alle $x\in H$ stets $(Tx,x)=0$, so ist $T=\0$ (Nulloperator).
\end{folgs}
\begin{proof}
Entweder mithilfe von Satz 3.3.7, oder mit dem Argument $(Tx,x)=0 \quad \forall x \in H$\\
$\Rightarrow \quad Tx \in H^\bot=\{0\} \quad \Rightarrow \quad T=\0$.
\end{proof}

\begin{defis}
Ist $T\in \L(H)$ ein selbstadjungierter Operator über einem Hilbertraum $H$, so heißt
\begin{align*}
\text{\highl[$m_T$|see{Operator}]{$m_T$}} :=\inf_{\norm{x}=1} (Tx,x) & \text{ \highl[Operator!untere Grenze $m_T$]{untere Grenze} des Operators } T \text{ und} \\
\text{\highl[$M_T$|see{Operator}]{$M_T$}} :=\sup_{\norm{x}=1} (Tx,x) & \text{ \highl[Operator!obere Grenze $M_T$]{obere Grenze} des Operators } T.
\end{align*}
\end{defis}
\medskip

\begin{bem}
Diese Definition ist sinnvoll, da $(Tx,x)$ nach Satz 3.3.6 stets reell ist. Es gilt (wegen Vorzeichen): $\norm{T \sep \L(H)} = \max(|m_T|,|M_T|)$.\\
Warum $m_T$ und $M_T$ obere bzw. untere Grenze genannt wird, begründet die nächste Folgerung: 
\end{bem}
\medskip

\begin{folgs}
Für selbstadjungierte Operatoren $T\in \L(H)$ über einem Hilbertraum $H$ gilt stets
\begin{gather*}
m_T I_H \leq T \leq M_T I_H
\end{gather*}
\end{folgs}
\begin{proof}
Sei $x\in H$.\\
1. Fall $x=0$: Offensichtlich gilt: $0 = (m_T I_H x,x) \leq (Tx,x) \leq (M_T I_H x,x)=0$.\\
2. Fall $x\neq 0$: setze $y:=\frac{x}{\norm{x}}$. Damit ist $x=\norm{x} y$ und $\norm{y}=1$. Es gilt:
\begin{align*}
(m_T I_H x,x) & = (m_T I_H (\norm{x}y),\norm{x}y) = \norm{x}^2 (m_T I_H y,y) = \norm{x}^2 m_T \norm{y}^2 = \norm{x^2} m_T \\
              & \leq \norm{x}^2 (Ty,y) = (T(\norm{x}y),\norm{x}y) = (Tx,x) \\
              & \leq \norm{x}^2 M_T = \norm{x}^2 M_T \norm{y}^2 = \norm{x}^2 (M_T I_H y,y) = (M_T I_H (\norm{x}y),\norm{x}y) \\
              & = (M_T I_H x,x)
\end{align*}
Zusammenfassend ist also $(m_T I_H x,x) \leq (Tx,x) \leq (M_T I_H x,x) \quad \forall x\in H$.
\end{proof}

\begin{lemmas}
Ist $H$ ein Hilbertraum und $T\in \L(H)$ normal, so ist für $\alpha \in \C$ auch $(T-\alpha I_H) \in \L(H)$ ein normaler Operator.
\end{lemmas}
\begin{proof}
Es gilt wegen $I_H = I_H^*$ und $TT^*=T^*T$ die folgende Umrechnung:
\begin{align*}
(T-\alpha I_H)(T-\alpha I_H)^* & = (T-\alpha I_H)(T^*-\overline{\alpha} I_H^*) \\
															 & = (T-\alpha I_H)(T^*-\overline{\alpha} I_H) \\
															 & = TT^* - \alpha I_H T^* - T \overline{\alpha} I_H + \alpha I_H \overline{\alpha} I_H \\
															 & = TT^* - \alpha T^* - \overline{\alpha} T + \alpha \overline{\alpha} I_H \\
															 & = T^*T - T^* \alpha I_H - \overline{\alpha} I_H T + \overline{\alpha} I_H \alpha I_H \\
															 & = (T^*-\overline{\alpha} I_H)(T-\alpha I_H) \\
															 & = (T-\alpha I_H)^*(T-\alpha I_H)
\end{align*}
Daher ist auch der Operator $T-\alpha I_H$ normal.
\end{proof}

%========== Abschnitt 1.1 =============
%Metrische Räume

\begin{defi}
Sei $X$ eine nicht leere Menge.\\
(i) Eine Abbildung $d \colon X \times X \nach [0,\infty)$ heißt \highl{Metrik} oder Abstandsfunktion \gdwdef für alle $x,y,z \in X$ gilt:
\begin{align*}
\text{(M1) } & x \neq y \Rightarrow d(x,y) \neq 0 & \text{(Definitheit)} \\
\text{(M2) } & d(x,x) = 0 & \\
\text{(M3) } & d(x,y)=d(y,x) & \text{(Symmetrie)} \\
\text{(M1) } & d(x,z) \leq d(x,y) + d(y,z) & \text{(Dreiecksungleichung)}
\end{align*}
Das Tupel \highl[$(X,d)$|see{Raum}]{$(X,d)$} heißt dann \highl[Raum!metrischer Raum $(X,d)$]{metrischer Raum}.\\
(ii) Gelten für $d$ nur (M2) - (M4), so spricht man von einer \highl[Metrik!Pseudometrik]{Pseudometrik} oder auch \highl[Metrik!Halbmetrik|see{Pseudometrik}]{Halbmetrik} bzw. von einem \highl[Raum!pseudometrischer Raum]{pseudometrischen Raum}.
\end{defi}
\medskip

\begin{bsp}
(i) $\Q$, $\R$ oder $\C$ versehen mit der \highl[Metrik!Betragsmetrik $d_{|\cdot|}$]{Betragsmetrik} \highl[$d_{|\cdot|}$|see{Metrik}]{$d_{|\cdot|}(x,y)$}$:=|y-x|$ sind metrische Räume.\\
\\
(ii) Jede beliebige nichtleere Menge $X$ wird mit der \highl[Metrik!diskrete Metrik $d_{dis}$]{diskreten Metrik} 
\begin{gather*}
\highl[$d_{dis}$|see{Metrik}]{$d_{dis}(x,y)$}:=
\begin{cases}
0, & x=y \\
1, & \text{sonst}
\end{cases}
\end{gather*}
zu einem metrischen Raum. \\
\\
(iii) $\R$ mit der \highl[Metrik!Arcustangensmetrik $d_{arc}$]{Arcustangensmetrik} \highl[$d_{arc}$|see{Metrik}]{$d_{arc}(x,y)$}$:=|\arctan(x)-\arctan(y)|$ ist metrischer Raum (zum Beweis siehe Übungsserie 1).\\
\\
(iv) $\C$ mit $d(z,w):=|\Re(z)-\Re(w)|$ ist ein peudometrischer Raum, da $d$ hier nur Pseudometrik ist. Denn für $z:=1+i$ und $w:=1+2i$ gilt $d(z,w)=0$, obwohl $z \neq w$.\\
\\
(v) Seien $x=(\xi_1, \ldots, \xi_n), y=(\eta_1, \ldots, \eta_n) \in \R^n$ oder $\C^n$. Dann definieren alle folgenden Funktionen Metriken auf diesen beiden Mengen:
\begin{align*}
\text{\highl[Metrik!$d_1$]{$d_1(x,y)$}} & :=\sum_{k=1}^n |\xi_k - \eta_k| \\ 
\text{\highl[Metrik!euklidische Metrik $d_2$]{$d_2(x,y)$}} & :=\left( \sum_{k=1}^n |\xi_k - \eta_k|^2 \right)^{\frac{1}{2}} \\
\text{\highl[Metrik!Metrik des $\R^n$ $d_p$]{$d_p(x,y)$}} & :=\left( \sum_{k=1}^n |\xi_k - \eta_k|^p \right)^{\frac{1}{p}} \text{, für } 1 \leq p < \infty\\
\text{\highl[Metrik!Maximumsmetrik des $\R^n$ $d_\infty$]{$d_\infty(x,y)$}} & := \max_{k=1, \ldots, n} |\xi_k - \eta_k|
\end{align*}
Summiert man dagegen nur bis $n-1$, so werden sie zu Pseudometriken.\\
\\
(vi) $(\R^2, d_M)$ ist ein metrischer Raum mit
\begin{gather*}
\text{\highl[$d_M$|see{Metrik}]{$d_M(x,y)$}}:=
\begin{cases}
\sqrt{(\xi_1 - \eta_1)^2 + (\xi_2 - \eta_2)^2},        & x=\lambda y, \text{ mit } \lambda \in \R \\
\sqrt{\xi_1^2 + \xi_2^2} + \sqrt{\eta_1^2 + \eta_2^2}, & \text{sonst}
\end{cases}
\end{gather*}
Hier ist der Abstand zweier linear abhängiger Punkte der euklidische und falls die Punkte nicht linear abhängig sind die Summe der Abstände zu einem zentralen Punkt (Nullpunkt). Die Metrik wird in Anlehnung an das französische Eisenbahnnetz / Pariser U-Bahnnetz auch \highl[Metrik!Metro-Metrik $d_M$]{Metro-Metrik} genannt.\\
\\
(vii) $(\R^2, d_u)$ mit
\begin{gather*}
\text{\highl[$d_u$|see{Metrik}]{$d_u(x,y)$}}:=
\begin{cases}
|\xi_1 - \eta_1| + |\xi_2| + |\eta_2|, & \xi_1 \neq \eta_1 \\
|\xi_2 - \eta_2|,                      & \xi_1 = \eta_1
\end{cases}
\end{gather*}
ist ebenfalls ein metrischer Raum. Die Metrik wird auch als \highl[Metrik!Urwaldmetrik $d_u$]{Urwaldmetrik} bezeichnet.
\end{bsp}
\medskip

\begin{defis}
Man bezeichnet die Menge aller (unendlichen) \highl[Folgenräume!Folgen $s$]{Folgen} über einem Körper $\K$ ($\R$ oder $\C$) mit \highl[$s$|see{Folgenräume}]{$s$}$:=\{ x=(\xi_k)_{k=1}^{\infty} \sep \xi_k \in \K \}$.\\
Man definiert darüber hinaus spezielle Mengen von Folgen mit gewissen Eigenschaften (Teilmengen von $s$):
\begin{align*}
\text{\highl[Folgenräume!$p$-summierbare Folgen $l_p$]{$l_p$}} :=      & \{ x \in s \sep \left( \sum_{k=1}^{\infty} |\xi_k|^p \right)^{\frac{1}{p}} <\infty \} & \text{( für } 1 \leq p < \infty \text{\highl[$l_p$|see{Folgenräume}]{})} \\
\text{\highl[Folgenräume!beschränkte Folgen $l_\infty$]{$l_\infty$}} := & \{ x \in s \sep \sup_k |\xi_k| < \infty \}                                           & \text{( Raum der \highl[$l_\infty$|see{Folgenräume}]{beschränkten Folgen})}\\
\text{\highl[Folgenräume!konvergente Folgen $c$]{$c$}} :=              & \{ x \in s \sep \lim_{k \nach \infty} \xi_k \text{ existiert} \}                      & \text{( Raum der \highl[$c$|see{Folgenräume}]{konvergenten Folgen})}\\
\text{\highl[Folgenräume!Nullfolgen $c_0$]{$c_0$}} :=                  & \{ x \in s \sep \lim_{k \nach \infty} \xi_k =0 \}                                     & \text{( Raum der \highl[$c_0$|see{Folgenräume}]{Nullfolgen})}\\
\text{\highl[Folgenräume!finite Folgen $F$]{$F$}} :=        & \{ x \in s \sep x=(\xi_1, \ldots, \xi_k, 0,\ldots) \}                                 & \text{( Raum der \highl[$F$|see{Folgenräume}]{finiten Folgen})}.
\end{align*}
\end{defis}
\medskip

\begin{bsp}[Fortsetzung]
(viii) $s$ bildet mit $d(x,y):= \sum_{k=1}^{\infty} \frac{1}{2^k} \frac{|\xi_k - \eta_k|}{1+|\xi_k - \eta_k|}$ einen metrischen Raum. Dabei ist wieder $x=(\xi_k)_{k=1}^{\infty}$ und $y=(\eta_k)_{k=1}^{\infty}$. Da der zweite Bruch stets $\leq 1$ ist und die Reihe über $\frac{1}{2^k}$ konvergiert ist $d$ stets endlich. Zum Beweis der Dreiecksungleichung nutze man die strenge Monotonie der Funktion $f(t)=\frac{t}{1+t}$ auf $[0,\infty)$.\\
Die Funktionen $d^{(k)}(x,y):= | \xi_k - \eta_k |$ bilden für alle $k \in \N$ ein System von Pseudometriken (Halbnormen) auf $s$. Durch obigen Ansatz führt ein solches System stets auf eine Metrik.\\\\
$l_p$ bildet mit der \highl[Metrik!$l_p$-Metrik $d_p$]{$l_p$-Metrik} \highl[$d_p$|see{Metrik}]{$d_p(x,y)$}$:= \left( \sum_{k=1}^{\infty} |\xi_k - \eta_k|^p \right)^{\frac{1}{p}}$ einen metrischen Raum. Zum Beweis der Gültigkeit der Metrikaxiome verwendet man vom endlich-dimensionalen Fall ausgehend einen Grenzübergang, sowie die Minkowski'sche Ungleichung für die Dreiecksungleichung.\\\\
Aus den Mengen $l_\infty, c$ und $c_0$ erhält man einen metrischen Raum üblicherweise durch die Wahl der \highl[Metrik!sup-Metrik auf $l_\infty$ $d_\infty(x,y)$]{Supremumsmetrik}: \highl[$d_\infty$|see{Metrik}]{$d_\infty(x,y)$}$:= \sup_k |\xi_k - \eta_k |$.
\end{bsp}
\medskip

\begin{defis}
(i) Für ein Gebiet $M$ (in der Regel $M=[a,b]$) bezeichnet \highl[$C(M)$|see{Funktionenräume}]{$C(M)$} die Menge der \highl[Funktionenräume!stetige Funktionen über $M$ $C(M)$]{stetigen Funktionen} $f \colon M \nach \R$.\\
(ii) Ist $X$ eine beliebige Menge, so heißt \highl[$B(X)$|see{Funktionenräume}]{$B(X)$}$:= \{ f \colon X \nach \C \sep \sup_{x \in X} |f(x)|<\infty \}$ die Menge der \highl[Funktionenräume!Beschränkte Funktionen über $X$ $B(X)$]{beschränkten Funktionen} über $X$.
\end{defis}
\medskip

\begin{bsp}[Fortsetzung]
(ix) \highl[Metrik!sup-Metik auf $C(M)$ $d_\infty(f,g)$)]{$d_\infty(f,g)$}$:= \sup_{x \in [a,b]} |f(x)-g(x)|$ definiert die Metrik auf $C([a,b])$.\\
Auf $B(X)$ ist durch $d_\infty(f,g) := \sup_{x \in X} |f(x)-g(x)|$ eine Metrik gegeben. Wählt man für alle Teilmengen $Y \subset X$ die Pseudometrik $d_{\infty, Y}(f,g) := \sup_{x \in Y} |f(x)-g(x)|$, so kann man auch aus diesem, $X$ ausschöpfenden System durch obigen Trick eine Metrik auf ganz $X$ konstruieren.\\
\\
(x) Für $1 \leq p < \infty$ bilden auch die Funktionen
\begin{gather*} 
\text{\highl[Metrik!Int-Metrik auf $C(M)$ $d_p(f,g)$]{$d_p(f,g)$}} := \left( \int_a^b |f(x)-g(x)|^p dx \right)^{\frac{1}{p}}
\end{gather*}
Metriken auf $C([a,b])$. Dabei ist $d_p$ offensichtlich eine symmetrische Abbildung\\
$C([a,b]) \times C([a,b]) \nach [0,\infty)$. Ist $f=g$ so folgt $d_p(f,g)=0$ unmittelbar, die Umkehrung beweist man indirekt mit dem Argument, dass die Funktion $f-g$ stetig sein muss. Für die Dreiecksungleichung nutzt man die gewöhnliche Dreiecksungleichung für Beträge und die Linearität des Integrals. 
\end{bsp}
\medskip

\begin{defi}
Sei $(X,d)$ ein metrischer Raum.\\
(i) Für $x\in X$ und $\epsilon > 0$ heißt \highl[$K_\epsilon$|see{offene Kugel}]{$K_\epsilon(x)$}$:=\{ y\in X \sep d(x,y)<\epsilon \}$ \highl{offene Kugel} mit Radius $\epsilon$ um $x$.\\
(ii) Ist $M \subset X$, so heißt $x \in M$ \highl[Punkt!innerer (metr. Raum)]{innerer Punkt} von $M$ und $M$ heißt \highl[Umgebung!Umgebung (metr. Raum)]{Umgebung} von $x$ \gdwdef $\exists \epsilon >0$ mit $K_\epsilon(x) \subset M$.
\end{defi}
\medskip

\begin{bsp}
(i) In $(\R, |\cdot|)$ ist $K_\epsilon(x)=(x-\epsilon, x+\epsilon)$ eine offene Kugel.\\
\\
(ii) Kreise in $(\R^2, d_2)$: $(\xi_1-\eta_1)^2+(\xi_2-\eta_2)^2<\epsilon^2$.\\
\\
(iii) $(X, d_{dis})$:
\begin{gather*}
K_\epsilon(x)=
\begin{cases}
X,     & \epsilon > 1 \\
\{x\}, & 0 < \epsilon \leq 1
\end{cases}
\end{gather*}
\end{bsp}
\medskip

\begin{defi}
Sei $(X,d)$ ein pseudometrischer Raum.\\
$O \subset X$ heißt \highl[Menge!offen (metr. Raum)]{offene Menge} (in $(X,d)$) \gdwdef $\forall x\in O$ gilt: $x$ ist innerer Punkt von $O$.
\end{defi}
\medskip

\begin{bsp}
Kreise ohne Rand im $\R^2$ sind offene Mengen
\end{bsp}
\medskip

\begin{bem}
Offene Kugeln $K_\epsilon(x)$ sind tatsächlich offene Mengen nach Definition 1.1.3.
\end{bem}
\medskip

\begin{satz}
Sei $(X,d)$ ein pseudometrischer Raum und $U \subset \Pot(X)$ das System aller offenen Mengen in $(X,d)$. Dann gilt:
\begin{align*}
\text{(i) }   & \leer, X \in U \\
\text{(ii) }  & O_1, O_2 \in U \Rightarrow O_1 \cap O_2 \in U \\
\text{(iii) } & O_i \in U \text{ für } i \in I \Rightarrow \bigcup_{i\in I} O_i \in U.
\end{align*}
\end{satz}
\begin{proof}
Offensichtlich.
\end{proof}
\medskip

\begin{bem}
Durch Induktion kann man (ii) noch auf endlich viele Mengen $O$ ausweiten.\\
$I$ aus (iii) ist eine beliebige Indexmenge (z.B.: $\N$ oder auch $\R$), d.h. auch die überabzählbare Vereinigung offener Mengen bleibt offen. Der Durchschnitt von abzählbar unendlich vielen offenen Mengen muss dagegen schon nicht mehr offen sein, wie nachfolgendes Beispiel zeigt:
\end{bem}
\medskip

\begin{bsp}
Wähle $(\R, |\cdot|)$ als metrischen Raum. Darin sind die Mengen $O_k:=(-\frac{1}{k}, \frac{1}{k})$ für alle $k \in \N$ offen, die Menge $\{0\}$ jedoch nicht. Dennoch gilt: $\bigcap_{k=1}^{\infty} O_k = \{ 0 \}$.
\end{bsp}

\begin{defi}
Sei $(X,d)$ ein metrischer Raum.\\
Die Menge $A \subset X$ heißt \highl[Menge!abgeschlossen (metr. Raum)]{abgeschlossen} \gdwdef $X \setminus A$ ist offen in $(X,d)$.
\end{defi}
\medskip

\begin{bem}
Abgeschlossen und offen sind nicht die einzigen Zustände. Es existieren auch Mengen die weder offen noch abgeschlossen sind.
\end{bem}
\medskip

\begin{folgs}
Sei $(X,d)$ ein pseudometrischer Raum. Dann gilt:
\begin{align*}
\text{(i) }   & X, \leer \text{ sind abgeschlossen} \\
\text{(ii) }  & A_1, A_2 \subset X \text{ abgeschlossen } \Rightarrow A_1 \cup A_2 \text{ abgeschlossen}\\
\text{(iii) } & A_i \subset X \text{ für } i \in I \text{ abgeschlossen } \Rightarrow \bigcap_{i\in I} A_i \text{ abgeschlossen}.
\end{align*}
\end{folgs}
\begin{proof}
Folgt leicht anhand von Definition 1.1.4, Satz 1.1.1 und den DeMorgan'schen Regeln.
\end{proof}

\begin{defi}
Sei $(X,d)$ ein metrischer Raum und $M\subset X$ eine beliebige Teilmenge. Dann heißt
\begin{align*}
\text{\highl[$\overline{M}$|see{Menge}]{$\overline{M}$}} & := \bigcap_{\stackrel{M \subset A}{A \text{ abgeschl.}}} A & \text{\highl[Menge!Abschluss $\overline{M}$ (metr. Raum)]{Abschluss von $M$}} \\
\text{\highl[$M^o$|see{Menge}]{$M^o$}} = \inner{M} & := \bigcup_{\stackrel{O \subset M}{O \text{ offen}}} O     & \text{\highl[Menge!Inneres $M^o$ (metr. Raum)]{Inneres von $M$}}
\end{align*}
\end{defi}
\medskip

\begin{bem}
$\overline{M}$ ist die kleinste abgeschlossene Menge, die $M$ enthält. Sei dazu angenommen es existiere $B \subset \overline{M}$ abgeschlossen mit $M\subset B \neq \overline{M}$, dann folgt sofort der Widerspruch $B \subset \overline{M} = \bigcap \{ A \sep M \subset A, A \text{ abgeschl.}\} \subset B$ zu $B \neq \overline{M}$.\\
Analog ist $M^o$ die größte offene Menge, welche ganz in $M$ enthalten ist.
\end{bem}
\medskip

\begin{defis}
Sei $(X,d)$ ein metrischer Raum und $M\subset X$ eine beliebige Teilmenge. Dann heißt $x \in X$ \highl[Punkt!Berührungspunkt (metr. Raum)]{Berührungspunkt} der Menge $M$ \gdwdef $\forall \epsilon > 0$ gilt $K_\epsilon(x) \cap M \neq \leer$.
\end{defis}
\medskip

\begin{satzs}
Sei $(X,d)$ ein metrischer Raum und $M\subset X$ eine beliebige Teilmenge. Dann gilt:
\begin{align*}
\text{(i) }   & \overline{M} = \{ x\in X\sep x \text{ ist Berührungspunkt von } M\} \\ 
\text{(ii) }  & M^o = \{ x\in X \sep x \text{ ist innerer Punkt von } M\} \\
\text{(iii) } & X \setminus M^o = \overline{X \setminus M} \\
\text{(iv) }  & X \setminus \overline{M} = (X \setminus M)^o \\
\text{(v) }   & \overline{S} \cup \overline{T} = \overline{S \cup T} \\
\text{(vi) }  & (S \cap T)^o = S^o \cap T^o.
\end{align*}
\end{satzs}
\begin{proof}
Siehe Übungsserie 1.
\end{proof}

\begin{defi}
Sei $(X,d)$ ein (pseudo-) metrischer Raum. Die Folge $(x_j)_{j=1}^{\infty} \subset X$ heißt \highl[Folge!konvergente Folge (metr. Raum)]{konvergent} in $X$ \gdwdef $\exists x\in X$ und $\forall \epsilon >0 \, \exists j_0(\epsilon)\in \N$ mit $d(x_j,x)<\epsilon$ für alle $j \geq j_0(\epsilon)$. Man schreibt in diesem Falle $x_j \nach x$, $\lim_{j \nach \infty} x_j = x$ oder äquivalent $\lim_{j \nach \infty} d(x,x_j)=0$.
\end{defi}
\medskip

\begin{bem}
Offensichtlich konvergiert die Folge, wenn für alle $j\geq j_0(\epsilon)$ gilt $x_j \in K_\epsilon(x)$.\\
Im metrischen Raum ist der Grenzwert einer konvergenten Folge stets eindeutig bestimmt. Zum Beweis wählt man $\epsilon := \frac{d(x,x')}{3}>0$, wobei $x \neq x'$ die angenommenen Grenzwerte bezeichnen. Ab der Stelle $j_0$ liegen dann alle $x_j$ in $K_\epsilon(x)$ und ab $j_1$ alle in $K_\epsilon(x')$, nach der Wahl des $\epsilon$ gilt jedoch $K_\epsilon(x) \cap K_\epsilon(x') = \leer$ (Widerspruch).\\
Für pseudometrische Räume ist das nicht mehr richtig. Beispielsweile konvergiert in $\C$ mit $d(z,w):=|\Re(z)-\Re(w)|$ die Folge $z_j=\frac{1}{j}$ gegen jedes $z=0+yi$ für $y\in \R$.
\end{bem}
\medskip

\begin{satz}
Ist $(X,d)$ ein metrischer Raum und $M \subset X$, so gilt:
$x\in \overline{M} \gdw \exists (x_j)_{j=1}^{\infty} \subset M$ mit $x_j \nach x$.
\end{satz}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{bem}
Ein (pseudo-) metrischer Raum induziert eine topologische Struktur auf $X$ (Siehe dazu Abschnitt 1.2), aber auch eine uniforme Struktur, die sich in den Begriffen der Cauchy-Folge, Vollständigkeit oder gleichmäßige Stetigkeit manifestiert.
\end{bem}
\medskip

\begin{defi}
Sei $(X,d)$ ein (pseudo-) metrischer Raum. Die Folge $(x_j)_{j=1}^{\infty} \subset X$ heißt \highl[Folge!Cauchy-Folge (metr. Raum)]{Cauchy-Folge} \gdwdef Für jedes $\epsilon>0$ existiert $j_1(\epsilon)\in \N$ mit $d(x_j,x_l)<\epsilon$ für alle $j,l \geq j_1(\epsilon)$.
\end{defi}
\medskip

\begin{bem}
Im (pseudo-) metrischer Raum ist jede konvergente Folge stets eine Cauchy-Folge, denn konvergiert $(x_j)_{j=1}^{\infty}$ gegen $x$, so folgt $d(x_j,x_l) \leq d(x_j,x)+d(x,x_l)< 2 \epsilon$ für $j,l \geq j_1(\epsilon)$.
\end{bem}
\medskip

\begin{defi}
Ein (pseudo-) metrischer Raum $(X,d)$ heißt \highl[vollständig!metrischer Raum]{vollständig} \gdwdef Jede Cauchy-Folge konvergiert in $X$.
\end{defi}
\medskip

\begin{bsp}
(i) Die Räume $(\R,|\cdot|)$ und $(\C,|\cdot|)$ sind vollständig.\\
\\
(ii) $(\R^n,d_p)$ und $(\C^n,d_p)$ sind vollständig, denn $d_p(x_j,x) \nach 0$ \gdw für alle $k=1,\ldots,n$ gilt $\xi_k^{(j)} \nach \xi_k$ in $\R$ (bzw. $\C$).\\
\\
(iii) Die Räume $(l_p,d_p)$ sind vollständig für $1 \leq p \leq \infty$.\\ 
Dazu sei $(x_j)_{j=1}^\infty$ eine Cauchy-Folge in $l_p$ ($1 \leq p < \infty$) und $\epsilon >0$. Dann existiert ein $j_0(\epsilon)$, sodass für alle $j,l \geq j_0(\epsilon)$ gilt 
\begin{gather*}
\left( \sum_{k=1}^\infty |\xi_k^j - \xi_k^l|^p \right)^{\frac{1}{p}} < \epsilon
\end{gather*}
Für beliebiges $\delta > 0$ und $k \in \N$ ist damit $|\xi_k^j - \xi_k^l|<\delta$, für $j,l\geq j_1(\delta,k)$. Also bildet jede Komponente jeweils eine Cauchy-Folge $(\xi_k^j)_{j=1}^\infty$ in $\C$. Da $\C$ vollständig ist, existiert für jedes $k \in \N$ ein $\xi_k \in \C$ mit $\xi_k^j \nach \xi_k$. Man setzt $x:=(\xi_k)_{k=1}^\infty$. Mit obiger Ungleichung folgt für fixiertes $N \in \N$
\begin{align*}
								                                  & \sum_{k=1}^N |\xi_k^j - \xi_k^l|^p < \widetilde{\epsilon}    & j,l \geq j_0(\widetilde{\epsilon})\\
\stackrel{l \nach \infty}{\longrightarrow} \quad  & \sum_{k=1}^N |\xi_k^j - \xi_k|^p < \widetilde{\epsilon} 		 & j \geq j_0(\widetilde{\epsilon})\\
\stackrel{N \nach \infty}{\longrightarrow} \quad  & \sum_{k=1}^\infty |\xi_k^j - \xi_k|^p < \widetilde{\epsilon} & j \geq j_0(\widetilde{\epsilon})
\end{align*}
Damit konvergiert die Folge $(x_j)_{j=1}^\infty$ gegen $x$ in $l_p$. Dass $x=(\xi_k)_{k=1}^\infty$ selbst in $l_p$ liegt zeigt folgende Abschätzung mithilfe der Minkowski'schen Ungleichung:
\begin{gather*}
\sum_{k=1}^\infty |\xi_k|^p = \sum_{k=1}^\infty |\xi_k - \xi_k^j + \xi_k^j|^p \leq \sum_{k=1}^\infty |\xi_k - \xi_k^j|^p + \sum_{k=1}^\infty |\xi_k^j|^p < \infty
\end{gather*}
Die Vollständigkeit von $l_{\infty}$ kann wie folgt nachgewiesen werden:\\
Sei $(x_j)_{j=1}^{\infty} \subset l_\infty$ eine beliebige Cauchy-Folge. Dann existiert für jedes $\epsilon > 0$ ein $j_1(\epsilon)$, sodass für alle $j,l \geq j_1(\epsilon)$ gilt $\sup_k |\xi_k^{(j)} - \xi_k^{(l)}| = d_\infty(x_j,x_l)<\epsilon$. Daher ist für jedes $k$ die Folge $(\xi_k^{(j)})_{j=1}^{\infty}$ eine Cauchy-Folge in $\R$ bzw. $\C$. Da diese Räume vollständig sind, existiert für jedes $k$ ein $\xi_k$ mit $\xi_k^{(j)} \nach \xi_k$ für $j \nach \infty$. Bezeichne nun $x:=(\xi_k)_{k=1}^{\infty}$ und zeige, dass $d_\infty(x_j,x) \nach 0$ gilt.
\begin{align*}
|\xi_k^{(j)}-\xi_k| & \leq |\xi_k^{(j)}-\xi_k^{(l)}| + |\xi_k^{(l)}-\xi_k| \\
                    & \leq d_\infty(x_j,x_l) + \frac{\epsilon}{2}, \text{ für } l \geq l_0(\frac{\epsilon}{2}, k) \\
                    & \leq \frac{\epsilon}{2} + \frac{\epsilon}{2} = \epsilon, \text{ für } j,l \geq j_1(\frac{\epsilon}{2})
\end{align*}
Damit ist für alle $k$: $|\xi_k^{(j)}-\xi_k| < \epsilon$, für $j \geq j_1(\frac{\epsilon}{2})$, denn es werden nur Bedingungen an $j$ gestellt. Also gilt $\sup_k |\xi_k^{(j)}-\xi_k| \leq \epsilon$, für $j \geq j_1(\frac{\epsilon}{2})$. Also ist $l_\infty$ ein vollständiger metrischer Raum.\\
\\
(iv) Der Raum $C([a,b])$ ist vollständig bezüglich der Supremumsmetrik, d.h. bzgl. $d_{\infty}(f,g)= \sup_{x \in [a,b]} |f(x)-g(x)|$, aber unvollständig bezüglich der $d_1$-Metrik. Betrachte dazu die Folge $(f_j)_{j=1}^{\infty}$ mit
\begin{gather*}
f_j(x):=
\begin{cases}
0, & -1 \leq x \leq -\frac{1}{j} \\
\frac{jx+1}{2}, & -\frac{1}{j} < x < \frac{1}{j} \\
1, & \frac{1}{j} \leq x \leq 1
\end{cases}
\end{gather*}
welche eine nicht in $C([-1,1])$ konvergente Cauchy-Folge bzgl. $d_1$ darstellt (denn es ist $d_1(f_j,f_k) < \max(\frac{1}{k},\frac{1}{j})<\epsilon$ für $k,j \geq j_1(\epsilon)$). Bezüglich $d_{\infty}$ ist $(f_j)_{j=1}^{\infty}$ nicht mal eine Cauchy-Folge, denn es ist $\sup_{x \in [-1,1]} |f_j(x)-f_k(x)|=\frac{1-\frac{j}{k}}{2}$ (o.B.d.A $j<k$).
\end{bsp}
\medskip

\begin{bem}
Verschiedene Metriken auf einer Menge $X$ können zwar die gleiche Topologie aber unterschiedliche uniforme Strukturen erzeugen, wie folgende Beispiele zeigen:
\end{bem}
\medskip

\begin{bsp}
(i) $X=\R$ mit den Metriken $d_{|\cdot|}(x,y)=|x-y|$ und $d_{arc}=|\arctan(x)-\arctan(y)|$. Wie in Übungsserie 1 gezeigt stimmen hier die Systeme der offenen Mengen überein, das heißt es wird die gleiche Topologie induziert. In $(\R,|\cdot|)$ konvergiert jede Cauchy-Folge, daher ist dieser Raum vollständig. In $(\R,d_{arc})$ ist aber beispielsweise die Folge $x_j=j$ eine Cauchy-Folge (in $(\R,|\cdot|)$ nicht!) ohne Grenzwert in $\R$, daher ist $(\R,d_{arc})$ nicht vollständig.\\
\\
(ii) Wähle $X:=\{ \frac{1}{n+1} \sep n \in \N \}$ zusammen mit $d_{|\cdot|}$. Die Cauchy-Folge $x_j:=\frac{1}{j+1}$ liegt zwar in $X$, konvergiert dort aber nicht, denn $0 \neq X$. Daher ist $(X, d_{|\cdot|})$ unvollständig. Zusammen mit der diskreten Metrik ist die Folge jedoch keine Cauchy-Folge, denn bei dieser Metrik müssen Folgen ab einer gewissen Stelle konstant sein, um Cauchy-Folge zu sein. Diese sind dann sicher konvergent, daher ist $(X, d_{dis})$ vollständig. Die erzeugte Topologie der beiden Räume ist allerdings die gleiche.
\end{bsp}
\medskip

\begin{lemmas}
Ist $(X,d)$ ein (pseudo-) metrischer Raum und $M \subset X$ eine Teilmenge, so wird durch $d$ eine (pseudo-) Metrik auf $M$ induziert:
\begin{gather*}
d_M(x,y):=d(x,y) \quad \forall x,y \in M.
\end{gather*}
\end{lemmas}
\begin{proof}
Offensichtlich erfüllt $d_M$ alle Bedingungen für eine Metrik auf $M$.
\end{proof}
\medskip

\begin{satzs}
Zu jedem metrischen Raum $(X,d)$ gibt es einen vollständigen metrischen Raum $(\widehat{X},\widehat{d})$ und eine Isometrie $i\colon X \nach \widehat{X}$ (d.h.  $\forall x,y \in X$ gilt $d(x,y)=\widehat{d}(i(x),i(y))$), sodass gilt $\overline{i(X)}=\widehat{X}$.
\end{satzs}
\begin{proof}
Zur Idee siehe Vorlesung.
\end{proof}
\medskip

\begin{bsp}
$\Q$ kann eindeutig bis auf Isomorphie vervollständigt werden. Dabei entsteht $\R$. In diesem Sinne repräsentiert jede reelle Zahl eine Äquivalenzklasse von Cauchy-Folgen, die gegen diese konvergieren (siehe Beweisidee).
\end{bsp}
\medskip

\begin{defi}[Überdeckungskompaktheit]
Sei $(X,d)$ ein metrischer Raum und $M \subset X$ eine beliebige Teilmenge.\\
(i) $(X,d)$ heißt \highl[kompakt!überdeckungskompakt]{(überdeckungs-) kompakt} \gdwdef Aus jeder offenen Überdeckung\\
$X \subset \bigcup_{i \in I} O_i$ (mit $O_i$ offen) kann man endlich viele $O_i$ auswählen, welche bereits ganz $X$ überdecken ($X \subset \bigcup_{l=1}^L O_{i_l}$).\\
(ii) $M$ heißt \highl[kompakt!überdeckungskompakt]{(überdeckungs-) kompakt} \gdwdef Aus jeder offenen Überdeckung von $M$ lässt sich stets eine endliche Überdeckung von $M$ auswählen.
\end{defi}
\medskip

\begin{bem}
Sei $(X,d)$ ein metrischer Raum und $M \subset X$. Dann ist $(M,d_M)$ wieder ein metrischer Raum und die offenen Mengen $O_M$ in $(M,d_M)$ lassen sich jeweils darstellen als $O_M = O_X \cap M$. Die Frage ob $M$ überdeckungskompakt (gemäß (ii)) ist, ist nun äquivalent zur Frage ob $(M,d_M)$ überdeckungskompakt (nach (i)) ist.
\end{bem}
\medskip

\begin{bsp}
Wählt man $X=\R$ und $M=[1,2]$, so ist die Menge $[1,\frac{3}{2})$ offen in $(M,d_M)$, da der Raum dort endet.
\end{bsp}
\medskip

\begin{defi}[Präkompaktheit]
Sei $(X,d)$ ein metrischer Raum und $M \subset X$ eine beliebige Teilmenge. Dann heißt $M$ \highl[präkompakt!überdeckungspräkompakt]{(überdeckungs-) präkompakt} \gdwdef $\overline{M}$ ist kompakt.
\end{defi}
\medskip

\begin{lemmas}[Eigenschaften des Kompaktheitsbegriffs]
Ist $(X,d)$ ein metrischer Raum und $M \subset X$, so folgt:\\
(i) $M$ kompakt $\Rightarrow M$ präkompakt und abgeschlossen.\\
(ii) $(X,d)$ kompakt, $M\subset X \Rightarrow M$ präkompakt. 
\end{lemmas}
\begin{proof}
(i) mit dem Begriff der Folgenkompaktheit (siehe später).\\
(ii) Offensichtlich.
\end{proof}
\medskip

\begin{defi}[Folgenkompaktheit, $\epsilon$-Netz]
Sei $(X,d)$ ein metrischer Raum und $M \subset X$ eine beliebige Teilmenge.\\
(i) $M$ heißt \highl[kompakt!folgenkompakt]{(folgen-) kompakt} \gdwdef Aus jeder Folge $(x_j)_{j=1}^{\infty} \subset M$ kann eine konvergente Teilfolge ausgewählt werden, wobei der Grenzwert in $M$ liegt.\\
(ii) $M$ heißt \highl[präkompakt!folgenpräkompakt]{(folgen-) präkompakt} \gdwdef Aus jeder Folge $(x_j)_{j=1}^{\infty} \subset M$ kann eine konvergente Teilfolge ausgewählt werden.\\
(iii) $E \subset X$ heißt \highl[Netz!$\epsilon$-Netz]{$\epsilon$-Netz} für $M$ \gdwdef Für jedes $x\in M$ existiert stets $y_x \in E$ mit $d(x,y_x)<\epsilon$, d.h. $M \subset \bigcup_{y\in E} K_\epsilon(y)$.
\end{defi}
\medskip

\begin{bsp}
Wähle $(X,d)=(\R,|\cdot|)$ und $M=(a,b]$, dann ist $M$ nicht (folgen-) kompakt, aber (folgen-) präkompakt. 
\end{bsp}
\medskip

\begin{lemmas}
Ist $(X,d)$ ein metrischer Raum und $M \subset X$ (folgen-) präkompakt, so existiert für jedes $\epsilon >0$ stets ein endliches $\epsilon$-Netz für $M$, d.h. $\exists x_1,\ldots,x_L \in X$ mit $M \subset \bigcup_{l=1}^L K_\epsilon(x_l)$.
\end{lemmas}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{satz}[Satz von Heine-Borel]\index{Satz!von Heine-Borel}
Sei $(X,d)$ ein metrischer Raum und $M \subset X$. Dann ist\\
$M$ (überdeckungs-) kompakt \gdw $M$ ist (folgen-) kompakt.
\end{satz}
\begin{proof}
Siehe Vorlesung.
\end{proof}
\medskip

\begin{bem}
Das heißt es gibt im metrischen Raum nur DIE Kompaktheit, da beide Begriffe hier äquivalent sind.
\end{bem}
\medskip

\begin{lemmas}
Ist $(X,d)$ ein vollständiger metrischer Raum und $M \subset X$. Dann sind äquivalent:\\
(i) $M$ ist präkompakt.\\
(ii) $\forall \epsilon >0$ existiert ein endliches $\epsilon$-Netz für $M$.\\
(iii) $\forall \epsilon >0$ existiert ein präkompaktes $\epsilon$-Netz für $M$.
\end{lemmas}
\begin{proof}
(i) $\Rightarrow$ (ii): Lemma von oben.\\
(ii) $\Rightarrow$ (iii): trivial, da jede endliche Menge kompakt (Jede Folge enthält konvergente Teilfolge: da nur endliche viele Elemente da sind, muss mindestens eines unendlich oft vorkommen), also insbesondere auch präkompakt ist.\\
Rest: Siehe Vorlesung.
\end{proof}
\medskip

\begin{defis}
Ist $(X,d)$ ein metrischer Raum, so heißt $M \subset X$ \highl[Menge!beschränkt (metr. Raum)]{beschränkt} \gdwdef $\exists z \in X$, $\exists D>0$ mit $M \subset K_D(z)$.
\end{defis}
\medskip

\begin{lemmas}
Im metrischen Raum gilt:\\
(i) Jede endliche Menge ist präkompakt.\\
(ii) Jede präkompakte Menge ist beschränkt.
\end{lemmas}
\begin{proof}
(i) Endliche Mengen bilden selbst ihr eigenes $\epsilon$-Netz.\\
(ii) Ist eine Menge präkompakt, so existiert nach obigem Lemma ein endliches $\epsilon$-Netz $x_1,\ldots,x_l \in X$. Dann gilt $M \subset K_{2 l \epsilon}(x_i)$ für ein beliebiges $i\in \{ 1,\ldots,l\}$.
\end{proof}
\medskip

\begin{bem}
Man kann die Präkompaktheit einer Menge auch quantitativ messen:\\
Sei $M$ eine präkompakte Teilmenge eines metrischen Raumes. Dann gibt es für jedes $\epsilon>0$ ein endliches $\epsilon$-Netz für $M$. Man setzt
\begin{gather*}
N(\epsilon):=\min \{ N \in \N \sep M \text{ wird durch } N \text{ Kugeln mit Radius } \epsilon \text{ überdeckt} \} \\
\text{\highl[$e_k$|see{Menge, Operator}]{$e_k(M)$}}=E_{2^{k-1}}(M):=\inf \{ \epsilon > 0 \sep 2^{k-1} \text{ Kugeln vom Radius } \epsilon \text{ überdecken } M \}
\end{gather*}
als Maße der Präkompaktheit. Dabei heißen die $e_k(M)$ (dynamische) \highl[Menge!Entropiezahlen $e_k(M)$]{Entropiezahlen} der Menge $M$. Es gilt zum Beispiel:
\begin{gather*}
e_{k+l+1}(M) \leq e_k(M) \cdot e_l(M).
\end{gather*}
\end{bem}

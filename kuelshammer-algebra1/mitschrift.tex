%%%%%%%%%%%%%%%%%%%
% Satz 1.1
%%%%%%%%%%%%%%%%%%%
\index{Division mit Rest}
\begin{satz}[Division mit Rest]\label{satz:1.1}
Zu $a \in \Z$ und $m \in \N$ \marginpar{$\N = [1, \infty[$} existiert
stets eindeutig bestimmt $q, r \in \Z$ mit $a = qm + r$ und $0 \leq r < m$.
\begin{beweis}
Sei $R := \{a - qm \colon q \in \Z\}$. Dann $R \cap \N_{0} \neq
\varnothing$. Denn f"ur $a \geq 0$ ist $a - 0m = a \in \N_{0}$, und
f"ur $a < 0$ ist $a - am = a(1 - m) \in \N_{0}$ ein kleinstes
Element. Sei $r := a - qm := \min R \cap \N_{0}$ mit $q \in \Z$. Dann
$r \geq 0$, aber $r - m = a - (q + 1)m < 0$, d.\,h. $r < m$. Also $q,
r$ wie gew"unscht.

Seien auch $q'$, $r' \in \Z$ mit $a = q'm + r'$ und $0 \leq r' <
m$. Dann $r - r' = (q' - q)m$. Wegen $\abs{r - r'} < m$ folgt $q' - q
= 0$, sonst $\abs{(q' - q)m} \geq m$. Also $q' = q$, d.h. $r' = r$.
\end{beweis}
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Definition 1.1
%%%%%%%%%%%%%%%%%%%
\begin{defi}
Man bezeichnet $q$ als Quotient $r$ als Rest.
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Definition 1.2
%%%%%%%%%%%%%%%%%%%
\begin{defi}\label{defi:1.2}
  Man nennt $a \in \Z$ \addindex{Teiler} von $b \in \Z$ und schreibt
  $a|b$, wenn ein $c \in \Z$ mit $b = ac$ existiert. Man sagt auch $a$
  teilt $b$, $b$ ist Vielfaches von $a$, $b$ ist durch $a$ teilbar usw.
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Satz 1.2
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:1.2}
F"ur alle $a, b, c, x, y \in \Z$ gilt:
\begin{enumerate}[(i)]
	\item $a|0$, $1|a$, $a|a$
	\item $a|b \Rightarrow \pm a|\pm b$
	\item $a|b \wedge b|c \Rightarrow a|c$
	\item $a|b \wedge b|a \Rightarrow a = \pm b$
	\item $a|b \wedge a|c \Rightarrow a|bx + cy$
	\item $0|a \Rightarrow a = 0$
	\item $a|b \wedge a,b \in \N \Rightarrow a \leq b$
\end{enumerate}
\begin{beweis}
  Der Beweis wird nur f"ur den Punkt (v) geführt. Der Rest ist
  analog. Nach Voraussetzung existiert $g,h \in \Z$ mit $b = ag$, $c =
  ah$. Folglich:
  \[bx + cy = agx + ahy = \underbrace{a(gx + hy)}_{\in \Z}\] d.\,h. $a|gx + hy$.
\end{beweis}
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Definition 1.3
%%%%%%%%%%%%%%%%%%%
\begin{defi}\label{defi:1.3}
Seien $a_{1}, \ldots, a_{n}, t \in \Z$ mit $t|a_{1}, \ldots,
t|a_{n}$. Dann heißt $t$ \addindex{gemeinsamer Teiler} von $a_{1},
\ldots, a_{n}$. Setze $\gt(a_{1}, \ldots, a_{n}) = \left\{t \in \Z
  \colon t\text{ gemeinsamer Teiler von } a_{1}, \ldots, a_{n}\right\}$.
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\index{Euklidischer Algorithmus}
\begin{bem}\label{bem:1.3}
Wegen $\gt(a_{1}, \ldots, a_{n}) = \gt(\pm a_{1}, \ldots, \pm a_{n})$ kann man $a_{1}, \ldots, a_{n} \in \N_{0}$ annehmen. Da man $a_{1}, \ldots, a_{n}$ auch permutieren darf, kann man $a_{1} \geq \ldots \geq a_{n}$ annehmen, sogar $a_{1} > \ldots > a_{n} > 0$. Division mit Rest liefert $q, r \in \Z$ mit $a_{1} = qa_{n} + r$, $0 \leq r < a_{n}$. Aus Satz \ref{satz:1.2} folgt leicht:
\[\gt(a_{1}, \ldots, a_{n}) = \gt(a_{1} - qa_{n}, a_{2}, \ldots, a_{n})\]
mit $r + a_{2} + \ldots + a_{n} < a_{1} + a_{2} + \ldots + a_{2}$.\\
Iteration ergibt schlie"slich:\\
\[\gt(a_{1}, \ldots, a_{n}) = \gt(b).\]
Dieses Verfahren hei"st euklidischer Algorithmus (\textsc{Euklid} 365--300).
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Beispiel
%%%%%%%%%%%%%%%%%%%
\begin{bsp}
Das Beispiel zeigt die Anwendung des euklidischen Algorithmus'.
\begin{align*}
  \gt(45, 27, 12) & = \gt(\overbrace{45 - 3 \times 12}^{9}, 27, 12) =
  \gt(27, 12, 9) = \gt(27 - 3 \cdot 9, 12, 9)\\
  & = \gt(12, 9) = \gt(12 - 1 \cdot 9, 9) = \gt(9, 3) = \gt(9 - 3
  \cdot 3, 3)\\
  & = \gt(3)
\end{align*}
\end{bsp}

%%%%%%%%%%%%%%%%%%%
% Definition 1.4
%%%%%%%%%%%%%%%%%%%
%\index{Größter gemeinsamer Teiler}
\begin{defi}\label{defi:1.4}
  Seien $a_{1}, \ldots, a_{n} \in \Z$. Ein $d \in \N_{0} \cap
  \gt(a_{1}, \ldots, a_{n})$ heißt \addindex{größter gemeinsamer
    Teiler}\footnote{Bezeichnung: ggT} von $a_{1}, \ldots, a_{n}$,
  falls $d$ durch jeden gemeinsamen Teiler von $a_{1}, \ldots, a_{n}$
  teilbar ist.
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Satz 1.3
%%%%%%%%%%%%%%%%%%%
\begin{satz}[Existenz des größten gemeinsamen Teilers]\label{satz:1.3}
Zu $a_{1}, \ldots, a_{n} \in \Z$ existiert stets genau ein $\ggt d$.
\begin{beweis}
  Nach Bemerkung \ref{bem:1.3} ist $\gt(a_{1}, \ldots, a_{n}) =
  \gt(b)$ für ein $b \in \N_{0}$. Dann ist $b$ ein größter gemeinsamer
  Teiler von $a_{1}, \ldots, a_{n}$. Ist $b' \in \N_{0}$ ein weiterer
  größter gemeinsamer Teiler von $a_{1}, \ldots, a_{n}$, so
  gilt \[b|b'\ und\ b'|b,\] also $b = \pm b'$ nach
  Satz~\ref{satz:1.2}. Wegen $b, b' \in \N_{0}$ folgt $b = b'$.
\end{beweis}
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
  Man schreibt $d = \ggt(a_{1}, \ldots, a_{n})$ und zeigt leicht:\[d =
  \ggt(a_{1}, \ggt(a_{2}, \ldots, a_{n})).\] Daher genügt es meist,
  den größten gemeinsamen Teiler von zwei Zahlen zu berechnen.
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Satz 1.4
%%%%%%%%%%%%%%%%%%%
\index{Erweiterter Euklidischer Algorithmus}
\begin{satz}[Erweiterter Euklidischer Algorithmus]\label{satz:1.4}
  Seien $a, b \in \N$. Setze
  \begin{align*}
    (x_{0}, y_{0}, z_{0}) & := (1, 0, a) & (x_{1}, y_{1}, z_{1}) & :=
    (0, 1, b)
  \end{align*}
  und $i := 1$. Im Fall $z_{i} = 0$ brechen wir ab. Im Fall $z_{i}
  \neq 0$ liefert Division mit Rest $q_{i}, r_{i} \in \Z$ mit $z_{i-1}
  = q_{i}z_{i} + r_{i}, 0 \leq r_{i} < z_{i}$. Wir setzen
  \[(x_{i+1}, y_{i+1}, z_{i+1}) := (x_{i+1} - q_{i}x_{i}, y_{i-1} -
  q_{i}y_{i}, z_{i-1} - q_{i}z_{i}),\]
  erhöhen $i$ um 1 und iterieren. Dieses Verfahren bricht ab, und am
  Ende ist
  \[\ggt(a, b)  = z_{i-1} = x_{i-1}a + y_{i-1}b.\]
\begin{beweis}
Wegen $b = z_{1} > z_{2} > \ldots > z_{i-1} > z_{i} = 0$ bricht das
Verfahren ab.

Am Anfang ($i = 1$) ist $\ggt(a, b) = \ggt(z_{i-1}, z_{i})$, und das
bleibt während des Verfahrens erhalten. Beachte $\ggt(z_{i-1},
  z_{i} = \ggt(z_{i}, \underbrace{z_{i+1}}_{z_{i-1} -
    q_{i}z_{i}}))$. Am Ende ist $z_{i} = 0$, also $\ggt(a, b) =
\ggt(z_{i-1}, z_{i}) = z_{i-1}$. Am Anfang $(i = 0, 1)$ ist $x_{i}a +
y_{i}b = z_{i}$, und das bleibt w"ahrend des Verfahrens erhalten:

\[x_{i+1}a + y_{i+1}b = (x_{i-1} - q_{i}x_{i})a + (y_{i-1} -
q_{i}y_{i})b = \underbrace{x_{i-1}a + y_{i}b}_{z_{i-1}} -
q_{i}\underbrace{(x_{i}a + y_{i}b)}_{z_{i}} = z_{i+1}\] 
\end{beweis}
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beispiel
%%%%%%%%%%%%%%%%%%%
\begin{bsp}
$a = 143, b = 91$:\\
\begin{tabular}{c|c|c|c}
$x_{i}$ & $y_{i}$ & $z_{i}$ & $q_{i}$\\
\toprule
1 & 0 & 143\\
0 & 1 & 91 & 1\\
1 & -1 & 52 & 1\\
-1 & 2 & 39 & 1\\
2 & -3 & 13 & 3\\
& & 0\\
\bottomrule
\end{tabular}
Also ist $\ggt(143, 91) = 13 = 2 \cdot 143 - 3 \cdot 91$.
\end{bsp}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
  Der erweiterte euklidische Algorithmus liefert also nicht nur $d :=
  \ggt(a, b)$, somdern auch $x, y \in \Z$ mit $d = ax + by$. Ist man
  an $x, y$ nicht interessiert, so kann man bei der Iteration die
  $x_{i}, y_{i}$ weglassen.
\end{bem}

% Satz 1.5
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:1.5}
  F"ur $a_{1}, \ldots a_{n}, b \in \Z$ gilt:
  \[\exists\ x_{1}, \ldots, x_{n} \in \Z \colon b = x_{1}a_{1} +
  \ldots + x_{n}a_{n} \Leftrightarrow \ggt(a_{1}, \ldots, a_{n})|b.\]
\begin{beweis}
  \begin{itemize}
  \item "`$\Rightarrow$"' Sei $b = x_{1}a_{1} + \ldots + x_{n}a_{n}
    (x_{1}, \ldots, x_{n} \in \Z)$. Für $t \in \gt(a_{1}, \ldots,
    a_{n})$ gilt dann $t|x_{1}a_{1} + \ldots + x_{n}a_{n} =
    b$. Insbesondere $\ggt(a_{1}, \ldots, a_{n})|b$.
  \item "`$\Leftarrow$"' Sei $d := \ggt(a_{1}, \ldots, a_{n}|b)$, etwa
    $b = cd$. Es gen"ugt zu zeigen, dass $y_{1}, \ldots, y_{n} \in \Z$
    existieren mit $d = y_{1}a_{1} + \ldots + y_{n}a_{n}$, denn dann
    ist $b = cd = \underbrace{cy_{1}}_{x_{1}}a_{1} + \ldots +
    \underbrace{cy_{n}}_{x_{n}}a_{n}$. Die Existenz von $y_{1},
    \ldots, y_{n}$ zeigen wir induktiv. Im Fall $n = 1$ ist $d = \pm
    a_{1}$. Im Fall $n = 2$ folgt die Existenz von $y_{1}, y_{2}$ aus
    Satz~\ref{satz:1.4}. Im Fall $n > 2$ existieren nach Induktion
    $z_{1}, \ldots, z_{n} \in \Z$ mit $\ggt(a_{2}, \ldots, a_{n}) =
    z_{2}a_{2} + \ldots + z_{n}a_{n}$. Außerdem existieren $v_{1},
    v_{2} \in \Z$ mit $d = \ggt(a_{1}, \ggt(a_{2}, \ldots, a_{n}))$
    und sind nach Satz~\ref{satz:1.5} gleich $v_{1}a_{1} +
    v_{2}\ggt(a_{2}, \ldots, a_{n}) = v_{1}a_{1} + v_{2}z_{2}a_{2} +
    \ldots + v_{2}z_{n}a_{n}$.
  \end{itemize}
\end{beweis}
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
  Daher gilt: \[\ggt(a_{1}, \ldots, a_{n}) = 1 \Leftrightarrow
  \exists\ x_{1}, \ldots, x_{n} \in \Z \colon x_{1}a_{1} + \ldots +
  x_{n}a_{n} = 1.\] Gegebenenfalls heißen $a_{1}, \ldots, a_{n}$
  \addindex{teilerfremd}.
\end{bem}

% Definition 1.5

\begin{defi}\label{defi:1.5}
  Eine natürliche Zahl $p \neq 1$ heißt \addindex{Primzahl}, falls
  $\pm 1, \pm p$ ihre einzigen Teiler sind. Setze
  \[\PZ = \left\{p \in \N \colon p \ Primzahl\right\}\]
  Ist $a \in \Z$ und $p \in \PZ$ mit $p|a$, so hei"st $p$ \addindex{Primteiler}
  (\addindex{Primfaktor}) von $a$.
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
$\PZ = \{2, 3, 5, 7, 11, 13, \ldots\}$
\end{bem}

% Satz 1.6
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:1.6}
Jede nat"urliche Zahl $a \neq 1$ hat einen Primteiler.
\begin{beweis}
  Wegen $1 \neq a|a$ ist $D := \{d \in \N \colon 1 \neq d|a\} \neq
  \varnothing$. Folglich existiert $p := min \ D$. W"are $p \notin
  \PZ$, so h"atte $p$ einen Teiler $t \in \N$ mit $1 \neq t \neq
  p$. Dann w"are aber $t < p$ und $t \in D$ im Widerspruch zur Wahl
  von $p$. Also $p \in \PZ$.
\end{beweis}
\end{satz}

\begin{satz}\label{satz:1.8}
F"ur $a, b \in \Z$ ist jeder Primteiler $p$ von $ab$ ein Teiler von $a$ oder $b$.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Im Fall $p|a$ sind wir fertig.\\
Sei also $p \not| a.$\\ 
Dann: $1 = \ggt(p, a) = xp + ya$ mit geeigneten $x, y \in \Z.$\\
Folglich: $p|xpb + yab = (xp + ya)b = 1b = b$
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
Induktiv folgt leicht: Sind $a_1, \ldots, a_r \in \Z$ und $p \in \PZ$ mit $p|a_1, \ldots, a_r$, so ist $p|a_i$ f"ur ein $i \in \left\{1, \ldots, r\right\}.$
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Satz 1.9
%%%%%%%%%%%%%%%%%%%
\index{Primfaktorzerlegung}
\begin{satz}[Eindeutige Primfaktorzerlegung]\label{satz:1.9}
Zu jeder nat"urlichen Zahl $m \neq \ 1$ existieren $p_1, \ldots, p_r \in \PZ$ mit $m = p_1 \ldots p_r.$ Sind auch $q_1, \ldots, q_s \in \PZ$ mit $m = q_1 \ldots q_s,$ so ist $r = s,$ und nach Umnummerierung gilt $p_i = q_i$ f"ur $i = 1, \ldots, r.$
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
\emph{Existenz:} Nach Satz \ref{satz:1.6} hat $m$ einen Primteiler $p$. Im Fall $m = p$ sind wir fertig. Anderenfalls ist $\frac{m}{p} \in \N$ und $1 < \frac{m}{p} < m$. Induktiv k"onnen wir annehmen, dass $p_1, \ldots, p_{r-1} \in \PZ$ existiert mit $\frac{m}{p} = p_1 \ldots p_{r-1}$, d.h. $m = p_1 \ldots p_{r-1}p$.\\
\emph{Eindeutigkeit:} Aus $p_1 \ldots p_r = m = q_1 \ldots q_s$ folgt $p_r|q_1 \ldots q_s$, also $p_r|q_i$ f"ur ein $i \in \left\{1, \ldots, s\right\}$, o.B.d.A. $p_r|q_s$. Wegen $q_s \in \PZ$ folgt $p_r = q_s$, d.h. $p_1 \ldots p_{r-1} = q_1 \ldots q_{s-1}$. Induktiv ist $r-1 = s-1$ und $p_i = q_i$ f"ur $i = 1, \ldots, r-1$ (nach Umnummerierung).
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
\begin{enumerate}[(i)]
	\item Man nennt $m = p_1 \ldots p_r$ die Primfaktorzerlegung von $m$. Durch zusammenfassen mehrfach auftretender Faktoren kann man diese auch in der Form $m = q_1^{a_1} \ldots q_t^{a_t}$ mit paarweisen verschiedenen $q_1, \ldots, q_t \in \PZ$ und beliebigen $a_1, \ldots, a_t \in \N$ schreiben. Eine weitere Schreibweise ist $m = \prod\limits_{p \in \PZ}p^{b_p}$, wobei $b_p \in \N_0$ f"ur alle $p \in \PZ$ und $\left|\left\{p \in \PZ \colon b_p \neq 0\right\}\right| < \infty$ ist. So kann man auch $ 1 = \prod\limits_{p \in \PZ}p^0$ als Primfaktorzerlegung der 1 auffassen.
	\item Die Teiler von $m = \prod\limits_{p \in \PZ}p^{b_p}$ sind genau die Zahlen $\pm \prod\limits_{p \in \PZ}p^{t_p}$ mit $t_p \leq b_p$ f"ur alle $p \in \PZ$.
	\item Die gemeinsamen Teiler von $m = \prod\limits_{p \in \PZ}p^{b_p}$ und $n = \prod\limits_{p \in \PZ}p^{c_p}$ sind also genau die Zahlen $\pm \prod\limits_{p \in \PZ}p^{t_p}$ mit $t_p \leq min \left\{b_p, c_p\right\}$ f"ur alle $p \in \PZ$. Daher ist $\ggt(a, b) = \prod\limits_{p \in \PZ}p^{min\left\{b_p, c_p\right\}}$.
	\item Die Konstruktion gro"ser Primzahlen und die Primfaktorzerlegung gro"ser nat"urlicher Zahlen sind von erheblicher Bedeutung f"ur die Kryptographie, d.h. die Lehre vom Ver- sowie Entschl"usseln geheimer Nachrichten.
\end{enumerate}
\end{bem}

\chapter{Kongruenzen}
%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}\label{bem:2.1}
Bekanntlich ist eine \addindex{Relation} ein Tripel $(M, N, R)$, wobei $M, N$ Mengen $R \subseteq M \times N$ sind. Statt $(x, y) \in R$ schreibt man meist $xRy$.\\
Eine \addindex{Äquivalenzrelation} auf einer Menge $M$ ist eine Relation der Form $(M, M, R)$ mit folgenden Eigenschaften:
\begin{enumerate}[(i)]
  \item $x \in M \Rightarrow xRx$ (Reflexivit"at)
  \item $xRy \Rightarrow yRx$ (Symmetrie)
  \item $xRy \wedge yRz \Rightarrow xRZ$ (Transitivit"at)
\end{enumerate}
Wichtige Beispiele sind die Kongruenz von Dreiecken oder die "Ahnlichkeit von Matrizen. F"ur eine beliebige "Aquivalenzrelation $(M, M, R)$ und $x \in M$ hei"st $[x]_R := {y \in M \colon xRy}$ die "Aquivalenzklasse von $x$ bzgl. $R$. Mit $M/R := {[x]_R \colon x \in M}$ bezeichnet man die Menge aller "Aquivalenzklassen von $M$ bzgl. $R$. Bekanntlich sind f"ur $x, y \in M$ die folgenden Eigenschaften gleichbedeutent:
\begin{enumerate}[(1)]
  \item $xRy$
  \item $[x]_R = [y]_R$
  \item $[x]_R \cap [y]_R \neq \varnothing$
\end{enumerate}
Daher ist $M/R$ eine Partition von $M$, d.h. eine Menge nichtleerer, paarweise disjunkter Teilmengen von $M$ mit Vereinigung $M$.\\
Bekanntlich erh"alt man umgekehrt zu jeder Partition $\mathcal{P}$ einer Menge $M$ eine "Aquivalenzrelation $(M, M, R)$, wenn man definiert: \[xRy \colon \Leftrightarrow \exists \ P \in \mathcal{P} \colon x, y \in P.\] Ferner ist dann $M/R = \mathcal{P}$.\\
Daher sind "Aquivalenzrelationen auf einer Menge $M$ und Partitionen von $M$ verschiedene Sichtweisen des gleichen mathematischen Sachverhaltes.
\end{bem}

% Definition
%%%%%%%%%%%%%%%%%%%
\index{Modulo}
\begin{defi}
Seien $a, b \in \Z, m \in \N$ mit $m|a - b$. Dann nennt man $a$ kongruent zu $b \ modulo \ m$ und schreibt $a \equiv b\pmod m$.
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Satz 2.1
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:2.1}
Die Kongruenz $modulo \ m$ ist eine "Aquivalenzrelation auf $\Z$. Ferner ist sie mit Addition und Multiplikation \emph{vertr"aglich}, d.h. es gilt: \[a \equiv a'\pmod m \wedge b \equiv b'\pmod m \Rightarrow a \stackrel{+}{\cdot}\ \equiv a' \stackrel{+}{\cdot}b'\pmod m.\]
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
\emph{Reflexivität:} $a \in \Z \Rightarrow m|0 = a - a \Rightarrow a \equiv a\pmod m$\\
\emph{Symmetrie:} $a \equiv b\pmod m \Rightarrow m|a - b \Rightarrow m|-(a - b) = b - a \Rightarrow b \equiv a\pmod b$\\
\emph{Transivität:} $a \equiv b\pmod m \wedge b \equiv c\pmod m \Rightarrow m|a - b \wedge m|b - c \Rightarrow m|(a - b) + (b - c) = a - c \Rightarrow a \equiv c\pmod m$\\
\emph{Verträglichkeit:} Sei $a \equiv a'\pmod m$ und $b \equiv b'\pmod m$, d.h. $m|a - a'$ und $m|b - b'$. Dann ist $m|(a - a') + (b - b') = (a + b) - (a' + b')$, d.h. $a + b \equiv a' + b'\pmod m$. Außerdem ist $m|(a - a')b + a'(b - b') = ab - a'b'$, d.h. $ab \equiv a'b'\pmod m$.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
F"ur $a \in \Z$ und $m \in \N$ hei"st $a + m\Z = {a + mz \colon z \in \Z = {\ldots, a - 2m, a - m, a, a + m, a + 2m, \ldots}}$ die Restklassen von $a\ modulo\ m$. $\left( \text{z.B.: ist } 2 + 6\Z = \left\{\ldots, -10, -4, 2, 8, 14, \ldots\right\}\right)$ Die Bezeichnung "`\addindex{Restklasse}"' erklärt sich folgendermaßen: Division mit Rest liefert $q, r \in \Z$ mit $a = qm + r$ und $0 \leq r < m$. Wegen $m|qm = a - r$ ist dann $a \equiv r\pmod m$, d.h. $a + m\Z = r + m\Z$.\\ $\left[\text{offenbar ist } a + m\Z \text{ die Äquivalenzklasse von } a \text{ bzgl. der Kongruenz } modulo\ m\right]$\\
Die Menge aller Restklassen ist also \[\left\{a + m\Z \colon a \in \Z\right\} = \left\{0 + m\Z, 1 + m\Z, \ldots, (m - 1) + m\Z\right\} =: \ZmZ{m}\]
Sind $r, s \in \left\{0, \ldots, m - 1\right\}$ mit $r \equiv s\pmod m$, so ist $m| r - s$ und $\left|r - s\right| < m$, also $r - s = 0$, d.h. $r = s$.\\
Also: $\left|\ZmZ{m}\right| = m$.
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Satz 2.2
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:2.2}
F"ur $a, b, c \in \Z$ und $m \in \N$ gilt:\\
\begin{enumerate}[(i)]
  \item $m|ab \wedge \ggt(a, m) = 1 \Rightarrow m|b$.
  \item $ab \equiv ac\pmod m \Leftrightarrow b \equiv c \pmod{\frac{m}{\ggt(a, m)}}$.
\end{enumerate}
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
\begin{enumerate}[(i)]
  \item Sei $\ggt(a, m) = 1$. Dann existieren $x, y \in \Z$ mit $xa + ym = 1$, d.h. $xab + ymb = b$. Aus $m|ab$ folgt $m|b$.
  \item \underline{\glqq$\Rightarrow$\grqq}: Sei $ab \equiv ac\pmod m$, d.h. $m|ab - ac$, etwa $a(b - c) = mz$ f"ur ein $z \in \Z$. Dann $\frac{a}{d}\left(b - c\right) = \frac{m}{d}z$ mit $d := \ggt(a, m)$. Wegen $\ggt\left(\frac{a}{d}, \frac{m}{d}\right) = 1$ folgt aus (i): \[\frac{m}{d}|b - c \text{, d.h. } b \equiv c\pmod{\frac{m}{d}}\]
\underline{\glqq$\Leftarrow$\grqq}: Sei $b \equiv c\pmod{\frac{m}{d}}$, d.h. $b - c = z\frac{m}{d}$ f"ur ein $z \in \Z$.\\
Dann: $a(b - c) = zm\frac{a}{d}$, d.h. $ab \equiv ac\pmod m$.
\end{enumerate}
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Beispiel
%%%%%%%%%%%%%%%%%%%
\begin{bsp}[Vorsicht bei Division von Kongruenzen!]
$2 \cdot 6 \equiv 2 \cdot 2\pmod 8$, aber $6 \not\equiv 2\pmod 8$ siehe Satz \ref{satz:2.2}(ii)!
\end{bsp}

%%%%%%%%%%%%%%%%%%%
% Satz 2.3
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:2.3}
Seien $a, b \in \Z, m \in \N$ und $d := \ggt(a, m)$. Genau dann hat die Kongruenz $ax \equiv b\pmod m$ eine Lösung $x \in \Z$, wenn $d|b$ gilt.\\
Gegebenenfalls bilden die Lösungen \emph{eine Vereinigung} (disjunkte Vereinigung) von $d$ Restklassen $modulo \ m$.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
\underline{\glqq$\Rightarrow$\grqq}: Sei $x \in \Z$ mit $ax \equiv b\pmod m$. Dann existiert $y \in \Z$ mit $ax - b = my$.\\
Folglich: $d|ax - my = b$.\\
\underline{\glqq$\Leftarrow$\grqq}: Sei $d|b, a_0 := \frac{a}{d}, m_0 := \frac{m}{d}, b_0 := \frac{b}{d}$. Wegen $\ggt(a_0, m_0) = 1$ existiert $y_0, z_0 \in \Z$ mit $1 = a_0y_0 + m_0z_0$. Daher: $b = \underbrace{a_0b}_{= ab_0}y_0 + \underbrace{m_0b}_{= mb_0}z_0 \equiv ax_0\pmod{m}$ mit $x_0 = b_0y_0 \left(\Rightarrow \text{ Existenz einer Lösung}\right)$\\
Für $x \in \Z$ gilt: \[ax \equiv b\pmod m \Leftrightarrow ax = ax_0\pmod m \stackrel{Satz ~ \ref{satz:2.2}}{\Leftrightarrow} x \equiv x_0\pmod m \Leftrightarrow x \in x_0 + m_0 \Z\] Sei $z \in \Z$. Division mit Rest liefert $q, r \in \Z$ mit $z = qd + r$ und $0 \leq r < d$.\\
Also: $zm_0 = q\underbrace{dm_0}_{= m} rm_0 \in rm_0 + m\Z$.\\
Daher: $m_0\Z = \bigcup\limits_{r = 0}^{d - 1}(rm_0 + m\Z)$ und $x_0 + m_0\Z = \bigcup\limits_{r = 0}^{d - 1}(x_0 + rm_0 + m\Z)$
\end{beweis}

% Beispiel
%%%%%%%%%%%%%%%%%%%
\begin{bsp}
Zu lösen: $23x \equiv 17\pmod{100}$\\
Erweiterter euklidischer Algorithmus: 
$\begin{array}{llll}
1 & 0 & 100 & \\ 
0 & 1 & 23 & 4\\ 
1 & -4 & 8 & 2\\
-2 & 9 & 7 & 1\\
3 & -13 & 1 & \\
\end{array}$\\
Also: $1 = 3 \cdot 100 - 13 \cdot 23 \equiv 23 \cdot (-13)\pmod{100}$\\
$17 \equiv 23 \cdot (-13) \cdot 17 \equiv 23 \cdot (-221) \equiv 23 \cdot 79 \pmod{100}$
\end{bsp}

% Definition
%%%%%%%%%%%%%%%%%%%
\begin{defi}
Seien $a_1, \ldots, a_n \in \Z$.\\
Ein $k \in \N_0$ mit $a_1|k, \ldots, a_n|k$ heißt \addindex{Kleinstes gemeinsames Vielfaches}(kgV) von $a_1, \ldots, a_n$, wenn $k|m$ für alle $m \in \Z$ mit $a_1|m, \ldots, a_n|m$ gilt.
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
Im Fall $a_i = 0$ für ein $i \in \left\{1, \ldots, \right\}$ ist $0$ das einzige gemeinsame Vielfache, also auch das einzige kgV von $a_1, \ldots, a_n$.\\
Sei also $a_1 \cdot \ldots \cdot a_n \neq 0$, und sei $a_i = \prod\limits_{p \in \PZ} p^{b_{ip}}\ (i = 1, \ldots, n)$.\\
Die gemeinsamen Vielfachen von $a_1, \ldots, a_n$ haben dann die Form:
\[\prod_{p \in \PZ}p^{c_p} \text{ mit } c_p \geq \max{\left\{b_{1p}, \ldots, b_{np}\right\}} \text{ für alle } p \in \PZ.\]
Daher ist $k := \prod\limits_{p \in \PZ}p^{\max{\left\{b_{1p}, \ldots, b_{np}\right\}}}$ das einzige kgV von $a_1, \ldots, a_n$. Dies zeigt, dass beliebige $a_1, \ldots, a_n \in \Z$ stets genau ein kgV haben. Man schreibt: $k = kgV(a_1, \ldots, a_n)$.
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Satz 2.4
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:2.4}
Für $a_1, \ldots, a_n \in \Z$ gilt stets:
\begin{enumerate}[(i)]
	\item $kgV(a_1, \ldots, a_n) = kgV(a_1, kgV(a_2, \ldots, a_n))$ \footnotesize{(Assoziativgesetz)}\normalsize
	\item $kgV(a_1, \ggt(a_2, \ldots, a_n)) = \ggt(kgV(a_1, a_2), \ldots, kgV(a_1, a_n))$ \footnotesize{(Distributivgesetz)}\normalsize
	\item $\ggt(a_1, kgV(a_2, \ldots, a_n)) = kgV(\ggt(a_1, a_2), \ldots, \ggt(a_1, a_n))$
\end{enumerate}
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Wir beweisen nur (iii) [Rest analog].\\
Im Fall $a_1 = 0$ steht auf beiden Seiten $kgV(a_2, \ldots, a_n)$.\\
Sei also $a_1 \neq 0$. Im Fall $a_i = 0$ für ein $i \in \left\{2, \ldots, n\right\}$ steht links $\ggt(a_1, 0) = a_1$.\\
Wegen $\ggt(a_1, a_2), \ldots, \ggt(a_1, a_n)|a_1 = \ggt(a_1, a_i)$ steht auch rechts $a_1$.\
Sei also $a_1 \cdot \ldots \cdot \neq 0$. Schreibe $a_i = \prod\limits_{p \in \PZ}p^{b_{ip}} \ (i = 1, \ldots, n)$. Dann:
\[\text{\underline{links:}} \prod_{p \in \PZ}p^{\min{\left\{b_{1p}, \max{\left\{b_{2p}, \ldots, b_{np}\right\}}\right\}}} \ \ \ \text{ \underline{rechts:}} \prod_{p \in \PZ}p^{\max{\left\{\min{\left\{b_{1p}, b_{2p}\right\}}, \ldots, \min{\left\{b_{1p}, b_{np}\right\}}\right\}}}\]
Beide Seiten sind gleich, da für alle $p \in \PZ$ gilt:
\[\min{\left\{b_{1p}, \max{\left\{b_{2p}, \ldots, b_{np}\right\}}\right\}} = \max{\left\{\min{\left\{b_{1p}, b_{2p}\right\}}, \ldots, \min{\left\{b_{1p}, b_{np}\right\}}\right\}}\].
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Satz 2.5
%%%%%%%%%%%%%%%%%%%
\begin{satz}[Chinesischer Restsatz \footnotesize{(stand in einem chinesischen Buch über Militärführung)}\normalsize]\label{satz:2.5}
Seien $a_1, \ldots, a_k \in \Z$ und $m_1, \ldots, m_k$. Genau dann ist das System von Kongruenzen:
\[(\ast) \ x \equiv a_i \pmod{m_i} \ \ (i = 1, \ldots, k)\]
Simultan lösbar, wenn gilt:
\[(\ast\ast) \ a_i \equiv a_j \pmod{\ggt(m_i, m_j)} \ \ (i, j = 1, \ldots, k)\]
Gegebenenfalls bilden die Lösungen eine Restklasse $modulo \ kgV(m_1, \ldots, m_k)$
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
\underline{\glqq$\Rightarrow$\grqq}: Sei ($\ast$) erfüllt. Für $i, j = 1, \ldots, k$ gilt dann: $a_i \equiv x \equiv a_j \pmod{\ggt(m_i, m_j)}$.\\
\underline{\glqq$\Leftarrow$\grqq}: Sei ($\ast\ast$) erfüllt. Wie argumentieren mit Induktion.\\
Im Fall $k = 1$ ist nichts zu tun. \footnotesize(können einfach $x = a_1$ wählen)\normalsize\\
Sei also $k \geq 2$, $d := \ggt(m_1, m_2)$, $e := kgV(m_1, m_2)$.\\
Schreibe $d = ym_1 + zm_2 (y, z \in \Z)$.\\
Setze: $b := a_1 + (a_2 - a_1)\frac{ym_1}{d} = a_2 + (a_1 - a_2)\frac{zm_2}{d}$.\\
Dann: $b \equiv a_1 \pmod{m_1}$ \footnotesize ($a_1 \equiv a_2 \pmod{d} \Rightarrow d|a_2 - a_1$) \normalsize und $b \equiv a_2 \pmod{m_2}$.\\
Im Fall $k = 2$ ist also ($\ast$) lösbar.\\
Sei also $k \geq 3$. Für $i = 3, \ldots, k$ ist $a_i \equiv a_1 \equiv b \pmod{\ggt(m_i, m_1)}$ und $a_i \equiv a_2 \equiv b \pmod{\ggt(m_i, m_2)}$.\\
Daraus folgt: $a_i \equiv b \pmod{kgV(\ggt(m_i, m_1), \ggt(m_i, m_2))} \stackrel{Satz \ref{satz:2.3}(iii)}{=} \ggt(m_i, kgV(m_1, m_2)) = \ggt(m_i, e)$.\\
Nach Induktion existiert ein $x \in \Z$ mit $x \equiv b \pmod{e}$ und $x \equiv a_i \pmod{m_i}$ für $i = 3, \ldots, k$ \footnotesize (Man muss hierfür ($\ast\ast$) für dieses System von Kongruenzen überprüfen) \normalsize\\
Für $i = 1,2$ ist $x \stackrel{x \equiv b \pmod{e}}{\equiv} b \equiv a_i \pmod{m_i}$.\\
\emph{Eindeutigkeit:} Sei $x' \in \Z$ weitere Lösung von ($\ast$). Dann: $x' \equiv a_i \equiv x \pmod{m_i}$ für $i = 1, \ldots, k$, d.h. $x' \equiv x \pmod{kgV(m_1, \ldots, m_k)}$. Dann ist $x' \equiv x \equiv a_i \pmod{m_i} \ \ (i = 1, \ldots, k)$, d.h. $x'$ ist auch Lösung von ($\ast$).
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Beispiel
%%%%%%%%%%%%%%%%%%%
\begin{bsp}\hfill\\
Zu Lösen: $(\ast) \left\{\begin{array}{lll}x & \equiv & 3 \pmod{14}\\ x & \equiv & 5 \pmod{12}\\ x & \equiv & 1 \pmod{10}\end{array}\right.$ \footnotesize (Um Lösbarkeit erstmal keine Gedanken machen, wird man schon im Laufe der Rechnung feststellen, anderenfalls $\Rightarrow$ ($\ast\ast$) muss man nicht zuerst überprüfen)\normalsize\\
Allgemeine Lösung von $x \equiv 3 \pmod{14}$: $3 + 14\Z$\\
$3 + 14y \equiv 5 \pmod{12} \Leftrightarrow 2y \equiv 2 \pmod(12) \Leftarrow y = 1$ \footnotesize (man braucht nicht allg. Lösung $\Rightarrow$ es reicht, wenn man eine sieht)\normalsize\\
$kgV(14, 12) = 14 \cdot 6 = 84$\\
Allgemeine Lösung von $x \equiv 3 \pmod{14}$ und $x \equiv 5 \pmod{12}$: $17 + 84\Z$\\
$17 + 84z \equiv 1 \pmod{10} \Leftrightarrow 4z \equiv 4 \pmod{10} \Leftarrow z = 1$\\
$kgV(84, 10) = 84 \cdot 5 = 420$\\
$\Rightarrow$ Allgemeine Lösung von ($\ast$): $101 + 420\Z$
\end{bsp}

\chapter{Der Restklassenring $modulo \ m$}
%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
Bekanntlich besteht ein \addindex{Ring} aus einer Menge $R$ und zwei Verknüpfungen $+, \cdot \colon R \times R \mapsto R$ mit folgenden Eigenschaften:
\begin{enumerate}[(i)]
	\item $(R, +)$ ist abelsche Gruppe
	\item $(ab)c = a(bc)$ \hfill $(a, b, c \in R)$
	\item $a(b + c) = ab + ac$ \hfill $(a, b, c \in R)$
	\item $(a + b)c = ac + bc$ \hfill $(a, b, c \in R)$
	\item Es exisitiert ein $1 \in R$ mit $1a = a = a1$ für $a \in R$
\end{enumerate}
\footnotesize (Unterschied zwischen Ring und Körper: bzgl. Multiplikation keine Gruppe, d.h. man muss nicht dividieren können) \normalsize \hfill \\
Bekanntlich ist das Element $1$ eindeutig bestimmt. Es heißt \addindex{Einselement}, und man schreibt $1 = 1_R$.\\
Analog enthält $R$ genau ein Element $0$ (\addindex{Nullelement}) mit $a + 0 = a = 0 + a$ für $a \in R$. Ferner existiert zu jedem $a \in R$ genau ein $-a \in R$ (\addindex{Negatives Element}) mit $a + (-a) = 0 = (-a) + a$. Dabei gilt: \[0a = 0 = a0, -(-a) = a, (-a)b = -(ab)\] Statt $a + (-b)$ schreibt man $a - b$.\\
Gilt $ab = ba$ für alle $a, b \in R$, so heißt R \addindex{Kommutativer Ring}.\\
z.B.: sind $\Z$ und jeder Körper $K$ kommutative Ringe, $\N, \N_0$ sind keine Ringe.\\
Für $n \in \N$ ist $K^{n \times n}$ mit der üblichen Addition und Multiplikation von Matrizen ein Ring (\addindex{Matrixring}). Für $n \geq 2$ ist $K^{n \times n}$ nicht kommutativ. Schließlich ist ${0}$ ein Ring. \footnotesize(hier ist das Nullelement das Einselement; aber kein Körper, denn da müsste $1_K \neq 0_K$)\normalsize
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Satz 3.1
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:3.1}
Für $m \in \N$ wird $\Z / m\Z = \left\{a + m\Z \colon a \in \Z\right\} = \left\{0 + m\Z, 1 + m\Z, \ldots, m - 1 + m\Z\right\}$ zu einem kommutativen Ring mit: \[(a + m\Z) + (b + m\Z) := a \stackrel{+}{\cdot} b + m\Z \ \ \ (a, b \in \Z)\]
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Definition
%%%%%%%%%%%%%%%%%%%
\begin{defi}
$\ZmZ{m}$: Restklassenring $modulo \ m$.
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Wir müssen zuerst zeigen, dass die Verknüpfungen "`wohldefiniert"' sind, d.h. nicht von der Schreibweise der Restklasse abhängen. Das bedeutet: aus $a + m\Z = a' + m\Z$ und $b + m\Z = b' + m\Z$ folgt: \[a \stackrel{+}{\cdot} b + m\Z = a' + b' + m\Z.\] Sei also $a + m\Z = a' + m\Z$ und $b + m\Z = b' + m\Z$. Dann ist $a \equiv a' \pmod{m}$ und $b' \equiv b \pmod{m}$, also auch $a \stackrel{+}{\cdot} b \equiv a' \stackrel{+}{\cdot} b' \pmod{m}$ nach Satz \ref{satz:2.1}. Dann $a + b + m\Z = a' + b' + m\Z$.\\
Damit ist die Wohldefiniertheit gezeigt. Die Rechenregeln folgen leicht:\\
z.B.: ist $(a + m\Z)(b + m\Z) = ab + m\Z = ba + m\Z = (b + m\Z)(a + m\Z) \ (a, b \in \Z)$\\
Nullelement ist $0 + m\Z$. Einselement ist $1 + m\Z$, negativ zu $a + m\Z$ ist $-a + m\Z$.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Beispiel
%%%%%%%%%%%%%%%%%%%
\begin{bsp}
$\ZmZ{6}$: \ \
\begin{tabular}[t]{l|llllll}
+ & 0 & 1 & 2 & 3 & 4 & 5\\
\hline
0 & 0 & 1 & 2 & 3 & 4 & 5\\
1 & 1 & 2 & 3 & 4 & 5 & 0\\
2 & 2 & 3 & 4 & 5 & 0 & 1\\
3 & 3 & 4 & 5 & 0 & 1 & 2\\
4 & 4 & 5 & 0 & 1 & 2 & 3\\
5 & 5 & 0 & 1 & 2 & 3 & 4\\
\end{tabular} \ \ \
\begin{tabular}[t]{l|llllll}
$\cdot$ & 0 & 1 & 2 & 3 & 4 & 5\\
\hline
0 & 0 & 0 & 0 & 0 & 0 & 0\\
1 & 0 & 1 & 2 & 3 & 4 & 5\\
2 & 0 & 2 & 4 & 0 & 2 & 4\\
3 & 0 & 3 & 0 & 3 & 0 & 3\\
4 & 0 & 4 & 2 & 0 & 4 & 2\\
5 & 0 & 5 & 4 & 3 & 2 & 1\\
\end{tabular}
\hfill\\\\\\
$\ZmZ{6}$ ist kein Körper. z.B.: ist $\underbrace{(2 + 6\Z)}_{\neq 0 + 6\Z}\underbrace{(3 + 6\Z)}_{\neq 0 + 6\Z} = 0 + 6\Z$
\end{bsp}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
\begin{enumerate}[(i)]
	\item Ein Element $a$ in einem Ring $R$ heißt \addindex{Invertierbar} (\addindex{Einheit}, \addindex{Unit}), wenn ein $b \in R$ mit $ab = 1 0 ba$ existiert.\\
	Bekanntlich ist dann $b$ durch $a$ eindeutig bestimmt. Man nennt $b$ das Inverse zu $a$ und schreibt $b = a^{-1}$. Dann ist auch $a^{-1}$ invertierbar und $(a^{-1})^{-1} = a$. Ist auch $b \in R$ invertierbar, so ist auch $ab$ invertierbar, und $(ab)^{-1} = b^{-1}a^{-1}$. Daher ist \[U(R) = \left\{a \in R \colon a \text{ invertierbar}\right\}\] ein Gruppe bezüglich $\cdot$, die \addindex{Einheitsgruppe} von $R$; $1_R$ ist das Einselement von $U(R)$.
	\item Im Fall $R = \ZmZ{m} \ \ (m \in \N)$ heißt: 
	\[
	\begin{array}{lcl}
	U(R) & = & \left\{a + m\Z \colon \exists \ b \in \Z \colon (a + m\Z)(b + m\Z) = 1 + m\Z\right\}\\
	     & = & \left\{a + m\Z \colon \exists \ b \in \Z \colon ab \equiv 1 \pmod{m}\right\}\\
	     & \stackrel{Satz \ref{satz:2.3}}{=} & \left\{a + m\Z \colon \ggt(a, m) = 1\right\}
	\end{array}
	\] \addindex{Prime Restklassengruppe $modulo \ m$} \footnotesize(Gruppe ist nicht prim, sondern nur die Restklassen)\\\normalsize
	z.B.: ist $U(\ZmZ{6}) = \left\{1 + 6\Z, 5 + 6\Z\right\}$ \footnotesize(bei einem Körper müssen alle Elemente inverterierbar sein!)\\ \normalsize
	\item Für $a + m\Z \in U(\ZmZ{m})$ berechnet man $(a + m\Z)^{-1}$ mit dem erweiterten euklidischen Algorithmus.\\
	Im Fall $a + m\Z = 39 + 100\Z$ erhalten wir z.B.: \[1 = \ggt(39, 100) = 16 \cdot 100 - 41 \cdot 39 \equiv 59 \cdot 39 \pmod{100}\] Also: $(39 + 100\Z)^{-1} = 59 + 100\Z$.
	\item Die \addindex{Eulerische $\varphi$-Funktion} $\varphi \colon \N \mapsto \N$ (\textsc{Euler} 1707--1783) wird definiert durch:
	\[ \begin{array}{lcl}
	\varphi(m) & := & \left|U(\ZmZ{m})\right| = \left|\left\{a + m\Z \colon \ggt(a, m) = 1\right\}\right|\\
	           & = & \left|\left\{a \in \N \colon 1 \leq a \leq m, \ggt(a, m) = 1\right\}\right| \ \ \ (m \in \N)
	\end{array}\]
	z.B. ist $\varphi(6) = 2$
\end{enumerate}
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Satz 3.2
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:3.2}
Für $m \in \N$ ist $\sum\limits_{d|m} \varphi(d) = m$; dabei durchläuft die Summe alle positiven Teiler $d$ von $m$.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
\[\begin{array}[t]{lcl}
m & = & \left|\left\{a \in \N \colon 1 \leq a \leq m\right\}\right|\\
  & = & \sum\limits_{d|m}\left|\left\{a \in \N \colon 1 \leq a \leq m, \ggt(a, m) = d\right\}\right|\\
  & = & \left|\left\{b \in \N \colon 1 \leq b \leq \frac{m}{d}, \ggt(b, \frac{m}{d}) = 1\right\}\right|\\
  & = & \sum\limits_{d|m} \varphi(\frac{m}{d}) = \sum\limits_{d|m} \varphi(d)
\end{array}\].
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
Damit kann man $\varphi(m)$ induktiv berechnen:\\
\[\begin{array}[t]{ll}
\varphi(1) = 1, & \varphi(1) + \varphi(2) = 2 \Rightarrow \varphi(2) = 1\\
                & \varphi(1) + \varphi(5) = 5 \Rightarrow \varphi(5) = 4\\
                & \varphi(1) + \varphi(2) + \varphi(5) + \varphi(10) = 10 \Rightarrow \varphi(10) = 4
\end{array}\]
Wir suchen nach einer expliziten Formel.
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Definition
%%%%%%%%%%%%%%%%%%%
\begin{defi}
Die \addindex{Möbius-Funktion} $\mu \colon \N \mapsto \Z$ (\textsc{Möbius} 1790--1868) wird definiert durch:
\[\mu(m) := \begin{cases}(-1)^r & \text{falls } m \text{ Produkt von } r \text{ verschiedenen Primzahlen ist}\\ 0 & \text{sonst}\end{cases}\]
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Beispiel
%%%%%%%%%%%%%%%%%%%
\begin{bsp}
$\mu(4) = \mu(2^2) = 0$\\
$\mu(6) = \mu(2 \cdot 3) = (-1)^2 = 2$\\
$\mu(1) := 1$
\end{bsp}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
$\ggt(m, n) = 1 \Rightarrow \mu(mn) = \mu(m) \cdot \mu(n)$
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Satz 3.3
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:3.3}
$m \in \N \Rightarrow \sum\limits_{d|m} \mu(d) \begin{cases}1 & m = 1\\ 0 & \text{sonst}\end{cases}$
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
o.B.d.A. $m \neq 1$. Betrachte Primfaktorzerlegung $m = p_1^{a_1} \cdot \ldots \cdot p_r^{a_r}$.\\
Dann: \[\begin{array}{lcl}
\sum\limits_{d|m}\mu(d) & = & \sum\limits_{i_1 = 0}^{a_1} \ldots \sum\limits_{i_r = 0}^{a_r} \underbrace{\mu(p_1^{i_1} \cdot \ldots \cdot p_r^{i_r})}_{\mu(p_1^{i_1}) \cdot \ldots \cdot \mu(p_r^{i_r})} \\
 & = & \prod\limits_{j = 0}^{r}\left[\underbrace{\mu(p_j^0)}_{= 1} + \underbrace{\mu(p_j^1)}_{= -1} + \underbrace{\mu(p_j^2)}_{= 0} + \ldots + \underbrace{\mu()p_j^{a_i}}_{= 0}\right] = 0
\end{array}\].
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Satz 3.4
%%%%%%%%%%%%%%%%%%%
\begin{satz}[Möbius Inversion]\label{satz:3.4}
Für Funktionen $f, F \colon \N \mapsto \C$ sind äquivalent:
\begin{enumerate}[(1)]
	\item $F(n) = \sum\limits_{d|n} f(d)$ für $n \in \N$
	\item $f(n) = \sum\limits_{d|n} \mu(d)F(\frac{n}{d})$ für $n \in \N$
\end{enumerate}
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
(1) $\Rightarrow$ (2): Sei (1) erfüllt. Für $m \in \N$ gilt dann:\\
\[\begin{array}{lcl}
\sum\limits_{d|m}\mu(d)F(\frac{m}{d}) & = & \sum\limits_{d|m}\mu(\frac{m}{d})F(d) = \sum\limits_{e|d|m}\mu(\frac{m}{d})f(e)\\
& = & \sum\limits_{e|m}f(e)\sum\limits_{d, e|d|m}\mu(\frac{m}{d}) = \sum\limits_{e|m}f(e)\underbrace{\sum\limits_{c|\frac{m}{e}}\mu(c)}_{= 0 \text{ außer für }\frac{m}{e} = 1}\\
& = & f(m)
\end{array}\]
(2) $\Rightarrow$ (1): Sei (2) erfüllt. Für $m \in \N$ gilt dann:\\
\[\begin{array}{lcl}
\sum\limits_{d|m}f(d) & = & \sum\limits_{d|m}\sum\limits_{e|d}\mu(e)F(\frac{d}{e}) = \sum\limits_{e|d|m}\mu(\frac{d}{e})F(e)\\
& = & \sum\limits_{e|m}F(e)\sum\limits_{d, e|d|m}\mu(\frac{d}{e}) = \sum\limits_{e|m}F(e)\underbrace{\sum\limits_{c|\frac{m}{e}}\mu(c)}_{= 0 \text{ außer für }\frac{m}{e} = 1}\\
& = & F(m)
\end{array}\]
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Satz 3.5
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:3.5}
Für $m \in \N$ mit Primfaktorzerlegung $m = p_1^{a_1} \cdot \ldots \cdot p_r^{a_r}$ gilt:\\
\[\varphi(m) = \left(p_1^{a_1} - p_1^{a_1-1}\right) \ldots \left(p_r^{a_r} - p_r^{a_r-1}\right) = m\left(1 - \frac{1}{p_1}\right) \ldots \left(1 - \frac{1}{p_r}\right)\]
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Möbius-Inversion mit $f := \varphi, F := id_{\N}$ ergibt:\\
\[\varphi(m) = \sum_{d|m} \mu(d) \underbrace{F(\frac{m}{d})}_{= \frac{m}{d}} = m - \frac{m}{p_1}- \frac{m}{p_2} - \ldots - \frac{m}{p_r} + \frac{m}{p_1p_2} + \ldots + \frac{m}{p_{r-1}p_r} + \ldots + (-1)^r\frac{m}{p_1p_2\ldots p_r} = m(1 - \frac{1}{p_1}) \ldots (1 - \frac{1}{p_r})\]
\end{beweis}

\end{bem}

\end{bsp}

% Satz 3.6
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:3.6}
Für $m \in \N$ gilt: $\ZmZ{m}$ Körper $\Leftrightarrow$ $m$ Primzahl
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
o.B.d.A. $m \neq 1$ \footnotesize (Körper benötigt mind. 2 Elemente)\\ \normalsize
\underline{\glqq$\Rightarrow$\grqq}: Sei $m \notin \PZ$, dann gilt $m = a \cdot b$ mit $a, b \in \left\{1, \ldots, m - 1\right\}$. Dann ist $a + m\Z \neq 0 \neq b + m\Z$, aber \[(a + m\Z)(b + m\Z) = m + m\Z = 0 + m\Z\]
Also: $\ZmZ{m}$ kein Körper.\\
\underline{\glqq$\Leftarrow$\grqq}: Sei $m \in \PZ$. Dann ist $\varphi(m) = m - 1$, d.h. $\ZmZ{m}$ enthält $m - 1$ invertierbare Elemente, d.h. alle Elemente in $\ZmZ{m}$ außer der Null sind invertierbar. Also ist $\ZmZ{m}$ ein Körper.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Satz 3.7
%%%%%%%%%%%%%%%%%%%
\begin{satz}[Wilson 1741--1793]\label{satz:3.7}
Für jeden endlichen Körper $K$ gilt:
\[\prod_{0 \neq x \in K}x = -1.\]
Insbesondere gilt für $p \in \PZ$:
\[(p - 1)! \equiv -1 \pmod{p}.\]
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
Setze $\F_p := \ZmZ{p}$ für $p \in \PZ$
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
In $\prod\limits_{0 \neq x \in K}x$ kürze man jeweils $x$ gegen $x^{-1}$, dies geht, falls $x \neq x^{-1}$ ist.
\[x = x^{-1} \Leftrightarrow x^2 = 1 \Leftrightarrow \underbrace{x^2 - 1}_{(x + 1)(x - 1)} = 0 \Leftrightarrow x = \pm 1.\]
Also $\prod\limits_{0 \neq x \in K}x = 1 \cdot (-1) = -1$.\\
Für $K := \ZmZ{p}$ erhält man: $\underbrace{\prod\limits_{i = 1}^{p - 1}(i + p\Z)}_{= -1 + p\Z} = \prod\limits_{i = 1}^{p}i + p\Z = (p - 1)! + p\Z$, d.h. $(p - 1)! = -1 \pmod{p}$
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Satz 3.8
%%%%%%%%%%%%%%%%%%%
\begin{satz}[\textsc{Euler-Fermat}]\label{satz:3.8}
Für jede endliche abelsche Gruppe $G$ und jedes $g \in G$ gilt: \[g^{\left|G\right|}.\] Insbesondere ist $a^{\varphi(m)} \equiv 1 \pmod{m}$ für alle $m \in \N$, $a \in \Z$ mit $\ggt(a, m) = 1$.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Sei $G = \left\{x_1, \ldots, x_n\right\}$, $n = \left|G\right|$. Für $i, j = 1, \ldots, n$ gilt dann \[gx_i = gx_j \Leftrightarrow g^{-1}gx_i = g^{-1}gx_j \Leftrightarrow x_i = x_j\] Daher ist auch $G = \left\{gx_1, \ldots, gx_n\right\}$. Also gilt: \[x := \prod_{i = 1}^{n}x_i = \prod_{i = 1}^{n}\left(gx_i\right) = g^n\prod_{i = 1}^{n} = g^nx,\] d.h. $g^n = 1$.\\
Für $G = U\left(\ZmZ{m}\right)$, d.h. $\left|G\right| = \varphi(m)$ und $g := a + m\Z \in G$ gilt also: \[1 + m\Z = g^{\left|G\right|} = \left(a + m\Z\right)^{\varphi(m)} = a^{\varphi(m)} + m\Z,\] d.h. $a^{\varphi(m)} \equiv 1 \pmod{m}$.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
\begin{enumerate}[(i)]
	\item Für $p \in \PZ$ gilt also:
	\begin{itemize}
		\item $a^{p-1} \equiv 1 \pmod{p}$ für alle $a \in \Z$ mit $p\not| a$
		\item $a^p \equiv a \pmod{p}$ für alle $a \in \Z$
	\end{itemize}
	\item Wir werden den Satz später auf nicht abelsche, endliche Gruppen verallgemeinern.
\end{enumerate}
\end{bem}

\chapter{Das RSA-Verfahren in der Kryptographie}
\addindex{RSA}\emph{(Rivest, Shamir, Adlemon 1982)} Public Key Cryptography\\\\
Ausgangspunkt: Personen $A$, $B$, $C$,\ldots wollen miteinander über einen unsicheren Kanal kommunizieren.\\
Dazu soll jede Nachricht so verschlüsselt werden, dass nur der richtige Empfänger sie entschlüsseln kann. Wie geht das?\\
Am Anfang wählt jeder Teilnehmer $T$ zwei große Primzahlen $p_T, q_T$ (ca. 100 Stellen) und eine große Zahl $r_T \in \N$, die zu $\varphi\left(p_Tq_T\right) = \left(p_T - 1\right)\left(q_T - 1\right)$ Teilerfremd ist. Mit dem erweitertem euklidischen Algorithmus berechnet er dann ein $s_T \in \N$ mit $r_Ts_T \equiv 1 \pmod{\varphi\left(p_Tq_T\right)}$.\\
Die Zahlen $p_T$, $q_T$, $s_T$ hält er geheim, die Zahlen $n_T := p_Tq_T$ und $r_T$ kommen in ein allgemein zugängliches Telefonbuch.\\
\[\text{Schlüssel von } T \colon p_T, q_T, s_T \text{ geheim; } n_T, r_T \text{ offen.}\]
Will Teilnehmer $A$ an Teilnehmer $B$ eine Nachricht $M$ schicken, so geht er folgendermaßen vor: \\
(Man kann annehmen: $M \in \N$ und $M < n_B$)\\
$A$ berechnet $M' \in \N$ mit $M' < n_B$ und $M' \equiv M^{r_B} \pmod{n_B}$. Dann schreibt er $M'$ an $B$\\
\[\begin{array}{ccccccccc}
\text{Absender} & \rightarrow & \text{Encoder} & \rightarrow & \text{Kanal} & \rightarrow & \text{Decoder} & \rightarrow & \text{Empfänger}\\
& & & & \uparrow & & & & \\
& & & & \text{Feind} & & & & \\
\end{array}\]
Der Empfänger $B$ kann $M'$ entschlüsseln, in dem er $(M')^{s_B}$ berechnet, denn es gilt: \[(M')^{s_B} \equiv M \pmod{n_B}\] Das sieht man folgendermaßen:\\
Wegen $r_Bs_B \equiv 1 \pmod{\left(p_B - 1\right)\left(q_B - 1\right)}$ existiert ein $t_B \in \N$ mit $r_Bs_B = 1 + t_B\left(p_B - 1\right)\left(q_B - 1\right)$. Also gilt: \[\left(M'^{s_B}\right) = \left(M^{r_B}\right)^{s_B} = M^{1 + t_B\left(p_B - 1\right)\left(q_b - 1\right)} = M\left(M^{r_B - 1}\right)^{t_B\left(q_B - 1\right)} \equiv M \pmod{p_B}\]
Im Fall $p_B\not| M$ folgt aus dem Satz \ref{satz:3.8} (\textsc{Euler-Fermat}), und im Fall $p_B|M$ ist dies trivial.\\
Analog ist $\left(M'\right)^{s_B} \equiv M \pmod{q_B}$, also auch $\left(M'\right)^{s_B} \equiv M \pmod{p_Bq_B}$.\\
Nehmen wir an, ein "`Feind"' hört die verschlüsselte Nachricht $M'$ ab. Kann er sie entschlüsseln?\\
\emph{Theoretisch ja}: schließlich muss er "`nur"' $n_B$ in Primfaktoren zerlegen. Dann kennt er $p_B$ und $q_B$, kann also $s_B$ mit dem erweiterten euklidischen Algorithmus berechnen und $M'$ genauso wie $B$ entschlüsseln.\\
\emph{Praktisch nein}: denn es ist kein Verfahren bekannt, das "`große"' Zahlen "`schnell"' in Primfaktoren zerlegt. Auch sind keine anderen Verfahren zur Entschlüsselung von $M'$ bekannt. Daher ist das RSA Verfahren in der Praxis gut etabliert. (\emph{http://www.rsasecurity.com})

\chapter{Polynome}
$K$ Körper

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}[vergleiche lineare Algebra]
Ein \addindex{Polynom} mit Koeffizienten in $K$ ist eine Folge $\varphi = \left(a_0, a_1, a_2, \ldots\right)$ von Elementen $a_i \in K$ mit $\left|\left\{i \in \N_0 \colon a_i \neq 0\right\}\right| < \infty$.\\
Diese Polynome bilden einen $K$-Vektorraum $P$ mit $\varphi + \psi := \left(a_0 + b_0, a_1 + b_1, \ldots\right)$, $r\varphi := \left(ra_0, ra_1, \ldots\right)$ für $\varphi = \left(a_0, a_1, \ldots\right)$, $\psi := \left(b_0, b_1, \ldots\right) \in P, r \in K$.\\
Setzt man $\varphi\psi = \left(c_0, c_1, c_2, \ldots\right)$ mit $c_0 = a_0b_0$, $c_1 = a_0b_1 + a_1b_0, \ldots, c_i := \sum\limits_{j + k = i}a_jb_k, \ldots$, so wird $P$ zu einem kommutativen Ring (\addindex{Polynomring}) mit $r\left(\varphi\psi\right) = \left(r\varphi\right)\psi = \varphi\left(r\psi\right)$ für $r \in K, \ \varphi, \psi \in P$.\\
Nullelement in $P$ ist das Nullpolynom $(0, 0, \ldots)$, Einselement das Einspolynom $(1, 0, 0, \ldots)$.\\
Ferner heißt $x := (0, 1, 0, \ldots)$ Unbestimmte oder Variable. Für $\varphi = \left(a_0, a_1, a_2, \ldots\right) \in P$ ist $\varphi x = \left(0, a_0, a_1, \ldots\right)$, insbesondere ist $x^2 = (0, 0, 1, 0, \ldots)$, $x^3 = (0, 0, 0, 1, 0, \ldots)$ u.s.w.\\
Wegen $\left|\left\{i \in \N_0 \colon a_i \neq 0\right\}\right| < \infty$ ist also $\varphi = \sum\limits_{i = 0}^{\infty}a_ix^i$, dabei ist wie üblich $x^0$ das Einspolynom. Daher kann man jedes Element in $P$ in der Form $\varphi = \sum\limits_{i = 0}^{\infty}a_ix^i$ mit eindeutig bestimmten Koeffizienten $a_i \in K$ schreiben, von denen nur endlich viele von $0$ verschieden sind. Dies werden wir in Zukunft stets tun. Für $\varphi = \sum\limits_{i = 0}^{\infty}a_ix^i$, $\psi = \sum\limits_{i = 0}^{\infty}b_ix^i \in P$ und $r \in K$ gilt dann:\\
\begin{itemize}
	\item $\varphi = \psi \Leftrightarrow a_i = b_i \text{ für alle } i$
	\item $\varphi + \psi = \sum\limits_{i = 0}^{\infty}\left(a_i + b_i\right)x^i$
	\item $r\varphi = \sum\limits_{i = 0}^{\infty}\left(ra_i\right)x^i$
	\item $\varphi\psi = \sum\limits_{i = 0}^{\infty}\left(\sum\limits_{j + k = i}a_jb_k\right)x^i$
\end{itemize}
Statt $P$ schreiben wir von jetzt ab $K\left[x\right]$.\\
Ist $0 \neq \varphi = \sum\limits_{i = }^{\infty}a_ix^i \in K\left[x\right]$, so heißt $d = \deg\varphi = \max{\left\{i \in \N_0 \colon a_i \neq 0\right\}}$ Grad von $\varphi$. Dann ist $\varphi = \sum\limits_{i = 0}^da_ix^i$.\\
Das Nullpolynom erhält den Grad $-\infty$.\\
Für $\varphi, \psi \in K\left[x\right]$ und $0 \neq r \in K$ gilt dann:
\begin{enumerate}[(i)]
	\item $\deg{\left(r\varphi\right)} = \deg{\left(\varphi\right)}$
	\item $\deg{\left(\varphi + \psi\right)} \leq \max{\left\{\deg\varphi, \deg\psi\right\}}$
	\item $\deg{\left(\varphi\right)} \neq \deg{\left(\psi\right)} \Rightarrow \deg{\left(\varphi + \psi\right)} = \max{\left\{\deg{\left(\varphi\right)}, \deg{\left(\psi\right)}\right\}}$
	\item $\deg{\left(\varphi\psi\right)} = \deg\varphi + \deg\psi$
\end{enumerate}
Für $r, s \in K$ ist $rx^0 \pm sx^0 = \left(r + s\right)x^0$.\\
Daher können wir jeweils $r \in K$ mit $rx^0 \in K\left[x\right]$ identifizieren und so $K$ als Teilmenge von $K\left[x\right]$ auffassen. Dann nennt man die Elemente in $K$ auch die konstanten Polynome von $K\left[x\right]$. Ist $0 \neq \varphi = \sum\limits_{i = 0}^{\infty}a_ix^i \in K\left[x\right]$, $d = \deg{\left(\varphi\right)}$ und $a_d = 1$, so heißt $\varphi$ normiert. Außerdem betrachtet man das Nullpolynom als normiert. Wir für $\Z$ hat man:
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Satz 5.1
%%%%%%%%%%%%%%%%%%%
\begin{satz}[Division mit Rest]\label{satz:5.1}
Seien $\varphi, \psi \in K\left[x\right]$ mit $\psi \neq 0$. Dann existiert eindeutig bestimmtes $\kappa, \rho \in K\left[x\right]$ mit $\varphi = \kappa\psi + \rho$ und $\deg\rho < \deg\varphi$.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Definition
%%%%%%%%%%%%%%%%%%%
\begin{defi}
$\kappa$ Quotient, $\rho$ Rest. Im Fall $\rho = 0$ schreibt man $\kappa = \frac{\varphi}{\psi}$
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Beispiel
%%%%%%%%%%%%%%%%%%%
\begin{bsp}
$x^5 + x^4 + x^3 + x^2 + x + 1 = \left(x^3 - x^2 - 1\right)\underbrace{\left(x^2 + 2x + 3\right)}_{\kappa} + \underbrace{\left(4x^2 + 3x + 4\right)}_{\rho}$
\end{bsp}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
Seien $\varphi, \psi \in K\left[x\right]$. Man nennt $\varphi$ Teiler von $\psi$ und schreibt $\varphi|\psi$, falls $\psi \equiv \varphi\omega$ für ein $\omega \in K\left[x\right]$ ist.\\
Dabei gilt wie in $\Z$:
\begin{enumerate}[(i)]
	\item $\varphi|0$, $1|\varphi$, $\varphi|\varphi$
	\item $\varphi|\psi \wedge \psi|\omega \Rightarrow \varphi|\omega$
	\item $\varphi|\psi \Rightarrow c\varphi|d\psi \ \ (c, d \in \kappa\backslash\left\{0\right\})$
	\item $\varphi|\psi \wedge \psi|\varphi \Rightarrow \exists \ c \in \kappa \backslash \left\{0\right\} = \varphi = c\psi$
	\item $0|\varphi \Leftrightarrow \varphi = 0$
	\item $\varphi|\psi \wedge \varphi|\omega \Rightarrow \varphi|\alpha\psi + \beta\omega \ \ \left(\alpha, \beta \in K \left[x\right]\right)$
\end{enumerate}
Seien $\varphi_1, \ldots, \varphi_n \in K\left[x\right]$. Ein $\tau \in K\left[x\right]$ mit $\tau|\varphi_1, \ldots, \tau|\varphi_n$ heißt \addindex{Gemeinsamer Teiler} von $\varphi_1, \ldots, \varphi_n$.\\
Mit $\gt(\varphi_1, \ldots, \varphi_n)$ bezeichnen wir die Menge aller gemeinsamer Teiler von $\varphi_1, \ldots, \varphi_n$. Ein normiertes Polynom $\delta \in \gt(\varphi_1, \ldots, \varphi_n)$ heißt \addindex{Größter gemeinsamer Teiler} (größten gemeinsamen Teiler), von $\varphi_1, \ldots, \varphi_n$, falls $\tau|\delta$ für alle $\tau \in \gt(\varphi_1, \ldots, \varphi_n)$ gilt.\\
Wie in $\Z$ kann man zeigen, dass $\varphi_1, \ldots, \varphi_n$ stets genau einen $\ggt \delta$ haben. Wir schreiben $\delta = \ggt(\varphi_1, \ldots, \varphi_n)$. Wie in $\Z$ ist $\delta = \ggt(\varphi_1, \ggt(\varphi_2, \ldots, \varphi_n))$.\\
Daher genügt es meist, den größten gemeinsamen Teiler von zwei Polynomen zu berechnen. Dies geschieht win in $\Z$ mit dem erweiterten euklidischen Algorithmus:
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Beispiel
%%%%%%%%%%%%%%%%%%%
\begin{bsp}
$\varphi = x^4 + x^3 + 2x^2 + x + 1, \psi = x^4 - x^3 + 2x^2 - x + 1$,\\
Erweiterter euklidischer Algorithmus:\\
\[ \begin{array}{cclc}
1 & 0 & x^4 - x^3 + 2x^2 + x + 1 & \\
0 & 1 & x^4 - x^3 + 2x^2 - x + 1 & 1\\
1 & -1 & 2x^3 + 2x & \frac{1}{2}x - \frac{1}{2}\\
-\frac{1}{2}x + \frac{1}{2} & \frac{1}{2}x + \frac{1}{2} & x^2 + 1 & \\
 & & 0 & 
\end{array}\]
Also $x^2 + 1 = \ggt(\varphi, \psi) = \left(-\frac{1}{2}x + \frac{1}{2}\right)\varphi + \left(\frac{1}{2}x + \frac{1}{2}\right)\psi$
\end{bsp}

%%%%%%%%%%%%%%%%%%%
% Satz 5.2
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:5.2}
Für $\varphi_1, \ldots, \varphi_n, \psi \in K\left[x\right]$ gilt: \[\ggt(\varphi_1, \ldots, \varphi_n)|\psi \Leftrightarrow \exists \ \xi_1, \ldots, \xi_n \in K\left[x\right] \colon \xi_1\varphi_1 + \ldots + \xi_n\varphi_n = \psi\]
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Analog wie in $\Z$.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
Für $\varphi_1, \ldots, \varphi_n \in K\left[x\right]$ ist also genau dann $\ggt(\varphi_1, \ldots, \varphi_n) = 1$, wenn $\xi_1, \ldots, \xi_n \in K\left[x\right]$ existiert mit $\xi_1\varphi_1 + \ldots + \xi_n\varphi_n = 1$ Gegebenfalls heißen $\varphi_1, \ldots, \varphi_n$ \addindex{Teilerfremd}.
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Definition
%%%%%%%%%%%%%%%%%%%
\begin{defi}
Ein normiertes Polynom $\pi \in K\left[x\right] \backslash K$ heißt \addindex{Irreduzibel}, falls $\pi$ keine Teiler $\tau$ mit $0 < \deg\tau < \deg\pi$ hat.
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Beispiel
%%%%%%%%%%%%%%%%%%%
\begin{bsp}
\begin{enumerate}[(i)]
	\item Normierte Polynome vom Grad $1$ sind stets irreduzibel
	\item $x^2 + 1 \left(x - i\right)\left(x + i\right)$ ist irreduzibel in $\C\left[x\right]$, aber irreduzibel in $\R\left[x\right]$.
\end{enumerate}
\end{bsp}

%%%%%%%%%%%%%%%%%%%
% Satz 5.3
%%%%%%%%%%%%%%%%%%%
\begin{satz}[Eindeutige Primfaktorzerlegung]\label{satz:5.3}\index{Eindeutige Primfaktorzerlegung}
Zu jedem $\varphi \in K\left[x\right] \backslash K$ existiert bis auf die Reihenfolge eindeutig bestimmte irreduzible Polynome $\pi_1, \ldots, \pi_r \in K\left[x\right]$ und eine eindeutig bestimmte Konstante $c \in K$ mit $\varphi = c\pi_1 \cdot \ldots \cdot \pi_r$. Die Teiler von $\varphi$ sind dann die Polynome $d\pi_{i_1} \cdot \ldots \cdot \pi_{i_s}$ mit $d \in K \backslash \left\{0\right\}$ und $1 \leq i_1 < \ldots < i_s \leq r$.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
Für $\varphi = \sum\limits_{i = 0}^{n}a_ix^i \in K\left[x\right]$ und $b \in K$ setzt man $\varphi(b) := \sum\limits_{i = 0}^{n}a_ib^i$ und sagt, dass $\varphi(b)$ durch Einsetzen von $b$ in $\varphi$ entsteht. Auf diese Weise definiert $\varphi$ eine \addindex{Funktion} $K \mapsto K, b \mapsto \varphi(b)$.\\
Dabei können verschiedene $\varphi_1, \varphi_2 \in K\left[x\right]$ die gleiche Funktion definieren, z.B. gilt im Fall $K = \F_2, \varphi_1 = x^2 + x + 1, \varphi_2 = 1 \colon \varphi_1(0) = 1 = \varphi_2(0), \varphi_1(1) = 1 = \varphi_2(1)$, aber $\varphi_1 \neq \varphi_2$.\\
Daher ist wichtig zwischen Polynomen und den durch sie definierten Funktionen zu unterscheiden.\\
Für $a, b \in K$ und $\varphi, \psi \in K\left[x\right]$ gilt offenbar:
\begin{enumerate}[(i)]
	\item $\left(\varphi \stackrel{\pm}{\cdot} \psi\right)(b) = \varphi(b) \stackrel{\pm}{\cdot} \psi(b)$
	\item $\left(a\varphi\right)(b) = a\varphi(b)$
\end{enumerate}
Ist $\varphi(b) = 0$, so heißt $b$ \addindex{Nullstelle} von $\varphi$.
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Satz 5.4
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:5.4}
Für $\varphi \in K\left[x\right]$ und $b \in K$ ist $\varphi(b)$ der Rest bei der Division und $\varphi$ durch $x - b$. Insbesondere gilt $\varphi(b) = 0 \Leftrightarrow x - b|\varphi$.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Division mit Rest liefert $\kappa, \rho \in K\left[x\right]$ mit $\varphi = \left(x - b\right)\kappa + \rho$ und $\rho \in K$.\\
Einsetzen ergibt:\\
$\varphi(b) = \underbrace{(b - b)}_{= 0}\kappa(b) + \rho = \rho$. Daher gilt die erste Behauptung. Im Fall $\varphi(b) = 0$ ist also $\varphi = (x - b)\kappa$, d.h. $x - b|\varphi$. Umgekehrt gilt im Fall $\varphi = (x - b)\kappa \colon \varphi(b) = \underbrace{(b - b)}_{= 0}\kappa(b) = 0$.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Satz 5.5
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:5.5}
Jedes $\varphi \in K[x]$ mit $n := \deg\varphi \neq -\infty$ hat in $K$ höchstens $n$ Nullstellen.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Seien $a_1, \ldots, a_m$ paarweise verschiedene Nullstellen von $\varphi$ in $K$. Dann sind $x - a_1, \ldots, x - a_m$ paarweise verschiedene irreduzible Polynome in $K[x]$. Nach Satz \ref{satz:5.4} ist $\varphi$ durch $x - a_1, \ldots, x - a_m$ teilbar, also auch durch $\underbrace{(x - a_1) \cdot \ldots \cdot (x - a_m)}_{\deg \ = m}$. Insbesondere ist $n = \deg\varphi \geq m$.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Definition
%%%%%%%%%%%%%%%%%%%
\begin{defi}
Für $\varphi = \sum\limits_{i = 0}^{\infty}a_ix^i \in K[x]$ definiert man die (\emph{formale})\addindex{Ableitung} durch $\varphi' := \sum\limits_{i = 1}^{\infty}ia_ix^{i - 1} \in K[x]$.
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}
Dann gelten die folgende Rechenregeln (Übungs-Aufgabe):
\begin{enumerate}[(i)]
	\item $\left(\varphi + \psi\right)' = \varphi' + \psi'$
	\item $\left(\varphi\psi\right)' = \varphi'\psi + \varphi\psi'$
	\item $\left(\varphi \circ \psi\right)' = \left(\varphi' \circ \psi\right)\psi'$; dabei ist $\varphi \circ \psi$ für $\varphi = \sum\limits_{i = 0}^{\infty}a_ix^i$ definiert durch $\varphi \circ \psi := \sum\limits_{i = 0}^{\infty}a_i\psi^i$.
\end{enumerate}
\end{bem}

\chapter{Restklassenringe von Polynomen}
$K$ Körper!

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}\label{bem:6.1}
Seien $\alpha, \beta, \varphi \in K[x]$ mit $0 \neq \varphi|\alpha - \beta$. Dann nennt man $\alpha$ kongruent zu $\beta \ modulo \ \varphi$ und schreibt $\alpha \equiv \beta \pmod{\varphi}$. Man zeigt leicht, dass dies eine Äquivalenzrelation ist. Die Äquivalenzklasse von $\alpha$ heißt Restklasse $modulo \ \varphi$.\\
Sie hat die Form $\alpha + \varphi K[x] := \left\{\alpha + \varphi\tau \colon \tau \in K[x]\right\}$.\\
Man setzt $(\varphi) := \varphi K[x]$.\\
Divison mit Rest liefert $\kappa, \rho \in K[x]$ mit $\alpha = \kappa\varphi + \rho$ und $\deg\rho < \deg\varphi$. Dann ist $\alpha \equiv \rho \pmod{\varphi}$, d.h. $\alpha + (\varphi) = \rho + (\varphi)$.\\
Daher enthält jede Restklasse $modulo \ \varphi$ ein Polynom $\rho$ mit $\deg\rho < \deg\varphi$.\\
Die Menge aller Restklassen $modulo \ \varphi$ ist also 
\[\begin{array}{lcl}
K[x] / (\varphi) & := & \left\{\alpha + (\varphi) \colon \alpha \in K[x]\right\}\\
& = & \left\{\rho + (\varphi) \colon \rho \in K[x], \deg\rho < n\right\}\\
& = & \left\{a_0 + a_1x + \ldots + a_{n - 1}x^{n - 1} + (\varphi) \colon a_0, \ldots, a_{n - 1} \in K\right\}
\end{array}\] 
mit $n := \deg\varphi$. Seien $\rho, \sigma \in K[x]$ mit $\deg\rho < n, \deg\sigma < n$ und $\rho + (\varphi) = \sigma + (\varphi)$, d.h. $\varphi|\rho - \sigma$.\\
Gradvergleich liefert $\rho = \sigma$. Dies zeigt, dass die Restklassen $\rho + (\varphi)$ mit $\deg\rho < n$ paarweise verschieden sind. Im Fall $\left|K\right| < \infty$ ist $\left|K[x] / (\varphi)\right| = \left|K\right|^{\deg\varphi}$.
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}\label{bem:6.2}
Sei $0 \neq \varphi \in K[x]$. Man zeigt leicht, dass die Kongruenz $modulo \ \varphi$ mit Addition und Multiplikation in $K[x]$ verträglich ist, d.h. für $\alpha_1, \alpha_2, \beta_1, \beta_2 \in K[x]$ gilt: \[\alpha_1 \equiv \alpha_2 \pmod{\varphi} \wedge \beta_1 \equiv \beta_2 \pmod{\varphi} \Rightarrow \alpha_1 \stackrel{+}{\cdot} \beta_1 \equiv \alpha_2 \stackrel{+}{\cdot} \beta_2 \pmod{\varphi}.\] Dies erlaubt es, auf $K[x] / (\varphi)$ Addition und Multiplikation zu definieren: \[\left[\alpha + (\varphi)\right] \stackrel{+}{\cdot} \left[\beta + (\varphi)\right] := \alpha \stackrel{+}{\cdot} \beta + (\varphi) \ \ \ \left(\alpha, \beta \in K[x]\right)\] Man rechnet nach, dass $K[x] / (\varphi)$ mit diesen Verknüpfungen zu einem kommutativen Ring wird, dieser heißt \addindex{Restklassenring $modulo \ \varphi$}.\\
Nullelement ist $0 + (\varphi)$, Einselement $1 + (\varphi)$.\\
Außerdem kann man Elemente in $K[x] / (\varphi)$ mit Elementen aus $K$ multiplizieren: \[a\left(\beta + (\varphi)\right) := a\beta + (\varphi) \ \ \ \left(a \in K, \beta \in K[x]\right).\] So wird $K[x] / (\varphi)$ zu einem $K$-Vektorraum. Bemerkung \ref{bem:6.1} zeigt, dass die Elemente $1 + (\varphi), x + (\varphi), x^2 + (\varphi), \ldots, x^{n - 1} + (\varphi) \ \ (n := \deg\varphi)$ eine $K$-Basis von $K[x] / (\varphi)$ bilden. Insbesondere ist $\dim_K K[x] / (\varphi) = \deg\varphi$.
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Satz 6.1
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:6.1}
Für $\alpha \in K[x]$ und $0 \neq \in K[x]$ sind äquivalent:
\begin{enumerate}[(1)]
	\item $\alpha + (\varphi)$ ist in $K[x] / (\varphi)$ invertierbar
	\item Es existiert ein $\beta \in K[x]$ mit $\alpha\beta \equiv 1 \pmod{p}$
	\item $\ggt(\alpha, \varphi) = 1$
\end{enumerate}
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Analog zu $\ZmZ{m}$.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}\label{bem:6.3}
Für $0 \neq \varphi \in K[x]$ gilt nach Satz \ref{satz:6.1}: \[U\left(K[x] / (\varphi)\right) = \left\{\alpha + (\varphi) \colon \alpha \in K[x], \ggt(\alpha, \varphi) = 1\right\}.\] Der erweiterte euklidische Algorithmus liefert zu $\alpha \in K[x]$ mit $\ggt(\alpha, \varphi) = 1$, Polynome $\beta, \gamma \in K[x]$ mit $\alpha\beta + \varphi\gamma = 1$. Dann: $\alpha\beta \equiv 1 \pmod{\varphi}$, d.h. $\left[\alpha + (\varphi)\right]\left[\beta + (\varphi)\right] = 1 + (\varphi).$ Also ist $\left[\alpha + (\varphi)\right]^{-1} = \beta + (\varphi)$.
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Satz 6.2
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:6.2}
Sei $0 \neq \varphi \in K[x]$ normiert, dann gilt: \[K[x] / (\varphi) \text{ Körper } \Leftrightarrow \varphi \text{ irreduzibel.}\]
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
\underline{\glqq$\Rightarrow$\grqq}: Sei $K[x] / (\varphi)$ Körper. Wegen $K[x] / (\varphi) \neq \left\{0\right\} ist \deg\varphi > 0$.\\
Wäre $\varphi$ reduzibel, so wäre $\varphi = \alpha\beta$ mit $\alpha, \beta \in K[x]$, $\deg\alpha < \deg\varphi$, $\deg\beta < \deg\varphi$. Dann wäre $\alpha + (\varphi) \neq 0 \neq \beta + (\varphi)$, aber $\left[\alpha + (\varphi)\right]\left[\beta + (\varphi)\right] = \alpha\beta + (\varphi) = \varphi + (\varphi) = 0 + (\varphi) \Rightarrow$ Widersprich!\\
\underline{\glqq$\Leftarrow$\grqq}: Sei $\varphi$ irreduzibel. Dann $\deg\varphi > 0$ und $K[x] / (\varphi) \neq \left\{0\right\}$. Sei $0 \neq \alpha \in K[x]$ mit $\deg\alpha < \deg\varphi$.\\
Dann: $\varphi\not|\alpha$, also $\ggt(\alpha, \varphi) = 1$\\
Folglich: $\alpha + (\varphi) \in U\left(K[x]/(\varphi)\right)$. Also $K[x] / (\varphi)$ Körper.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Beispiel
%%%%%%%%%%%%%%%%%%%
\begin{bsp}\label{bsp:6.1}
Sei $p \in \PZ$ und $n \in \N$. Um einen Körper mit $p^n$ Elementen zu konstruieren, genügt es also, ein irreduzibles Polynom vom Grad $n$ in $\F_p[x]$ zu finden. Wir werden später sehen, dass dies für alle $p$, $n$ geht. Im Fall $p = 2$, $n = 3$ ist z.B. $\varphi = x^3 + x + 1$ irreduzibel in $\F_2[x]$; denn sonst wäre $\varphi = \left(x - a\right)\left(x^2 + bx + c\right)$ mit $a, b, c \in \F_2$, also $\varphi(a) = 0$. Aber: $\varphi(0) = 1 = \varphi(1)$.\\
Daher ist $\F_2[x] / \left(x^3 + x + 1\right)$ Körper mit $2^3 = 8$ Elementen. (ÜA: Addition- und Multiplikationstabelle aufstellen!)
\end{bsp}

%%%%%%%%%%%%%%%%%%%
% Definition
%%%%%%%%%%%%%%%%%%%
\begin{defi}
Ein \addindex{Teilkörper} von $K$ ist eine Teilmenge $k$ von $K$ mit folgenden Eigenschaften:
\begin{enumerate}[(i)]
	\item $0, 1 \in k$
	\item $a, b \in k \Rightarrow a \stackrel{\pm}{\cdot} b \in k$
	\item $a \in k, 0 \neq b \in k \Rightarrow \frac{a}{b} \in k$
\end{enumerate}
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}\label{bem:6.4}
Gegebenenfalls ist $k$ mit den entsprechenden eingeschränkten Verknüpfungen ein Körper. Man nennt $K$ \addindex{Erweiterungskörper} von $k$, dass Paar $(K, k)$ heißt \addindex{Körpererweiterung}. Man schreibt: $K|k$.
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Beispiel
%%%%%%%%%%%%%%%%%%%
\begin{bsp}\label{bsp:6.2}
\begin{enumerate}[(a)]
	\item $\C|\R$, $\R|\Q$ sind Körpererweiterungen.
	\item Für jede Familie $(k_i)_{i \in I}$ von Teilkörpern von $K$ ist auch $\bigcap\limits_{i \in I}k_i$ ein Teilkörper von $K$. Insbesondere ist der Durchschnitt aller Teilkörper von $K$ ein Teilkörper $P$ von $K$; $P$ heißt \addindex{Primkörper} von $K$.
\end{enumerate}
\end{bsp}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}\label{bem:6.5}
Sei $L|K$ Körpererweiterung. Dann kann man $L$ als $K$-Vektorraum auffassen. Man nennt $\left[L:K\right] := \dim_KL$ den Grad der Körpererweiterung.
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Beispiel
%%%%%%%%%%%%%%%%%%%
\begin{bsp}\label{bsp:6.3}
$\left[\C|\R\right] = 2$, $\left[\R|\Q\right] = \infty$
\end{bsp}

%%%%%%%%%%%%%%%%%%%
% Satz 6.3
%%%%%%%%%%%%%%%%%%%
\begin{satz}[Gradsatz]\label{satz:6.3}
Für jede doppelte Körpererweiterung $M|L|K$ gilt: \[\left[M:K\right] = \left[M:L\right]\left[L:K\right]\]
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Aus $\left[M:L\right] = \infty$ oder $\left[L:K\right] = \infty$ folgt natürlich $\left[M:K\right] = \infty$. Daher seien $\left[M:L\right]$ und $\left[L:K\right]$ endlich. Wähle $K$-Basis $a_1, \ldots, a_m$ von $L$ und $L$-Basis $b_1, \ldots, b_n$ von $M$. Wir zeigen, dass die Elemente $a_ib_j \ \left(i = 1, \ldots, m; j = 1, \ldots, n\right)$ eine $K$-Basis von $M$ bilden.\\
Zum Beweis sei $z \in M$. Dann existieren $y_1, \ldots, y_n \in L$ mit $z = \sum\limits_{j = 1}^{n}y_jb_j$. Für $j = 1, \ldots, n$ existieren Elemente $x_{ij} \in K$ mit $y_j = \sum\limits_{i = 1}^{m}x_{ij}a_i$.\\
Also: $z = \sum\limits_{i = 1}^{m}\sum\limits_{j = 1}^{n}x_{ij}a_ib_j$\\
Seien jetzt Elemente $t_{ij} \in K$ gegeben mit $\underbrace{\sum\limits_{i = 1}^{m}\sum\limits_{j = 1}^{n}t_{ij}a_ib_j}_{= \sum\limits_{j = 1}^{n}\underbrace{\left(\sum\limits_{i = 1}^{m}t_{ij}a_i\right)}_{\in L}b_j} = 0$ Da $b_1,\ldots, b_n$ linear unabhängig über $L$ sind, folgt $\sum\limits_{i = 1}^{m}t_{ij}a_i = 0$ für $j = 1, \ldots, n$. Da $a_1, \ldots, a_m$ linear unabhängig über $K$ sind, folgt $t_{ij} = 0$ für alle $i, j$.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Definition
%%%%%%%%%%%%%%%%%%%
\begin{defi}
Seien $K, L$ Körper. Eine Abbildung $f \colon K \mapsto L$ mit $f(a \stackrel{+}{\cdot} b) = f(a) \stackrel{+}{\cdot} f(b)$ für $a, b \in K$ und $f(1) = 1$ nennt man \addindex{Homomorphismus} (von Körpern). Wie üblich definiert man Mono-, Epi-, Iso-, Endo- und Automorphismus (von Körpern).
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}\label{bem:6.6}
\begin{enumerate}[(i)]
	\item Für Körper $K, L, M$ und Homomorphismus $f \colon K \mapsto L, g \colon L \mapsto M$ ist auch $g \circ f \colon K \mapsto M$ ein Homomorphismus.
	\item Ist $f \colon K \mapsto L$ ein Isomorphismus, so auch $f^{-1} \colon L \mapsto K$.
	\item Für jeden Homomorphismus $f \colon K \mapsto L$ und $x \in K$ gilt: $f(x) = 0 \Leftrightarrow x = 0$, denn aus $f(0) = f(0 + 0) = f(0) + f(0)$ folgt $f(0) = 0$. Ist andererseits $0 \neq c \in K$, so ist $1 = f(1) = f(xx^{-1}) = f(x)f(x^{-1})$, also $f(x) \neq 0$.
	\item Jeder Homomorphismus von Körpern $f \colon K \mapsto L$ ist injektiv, denn sind $a, b \in K$ mit $f(a) = f(b)$, so ist $0 = f(b) - f(a) = f(b - a)$, d.h. $b - a = 0$ nach (iii). Also: $b = a$
	\item Die komplexe Konjugation $\C \mapsto \C, z \mapsto \overline{z}$, ist ein Automorphismus.
	\item Man nennt Körper $K, L$ isomorph und schreibt $K \cong L$, falls ein Isomorphismus $f \colon K \mapsto L$ existiert. Wie üblich zeigt man: $\cong$ ist Äquivalenzrelation
	\item Für jeden Homomorphismus von Körpern $f \colon K \mapsto L$ ist $Bld(f) = f(K)$ ein zu $K$ isomorpher Teilkörper von $L$.
\end{enumerate}
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Beispiel
%%%%%%%%%%%%%%%%%%%
\begin{bsp}\label{bsp:6.4}
In $\R[x]$ ist $x^2 + 1$ irreduzibel. Daher ist $\R[x]/\left(x^2 + 1\right)$ ein Körper. Die Elemente in $\R[x]/\left(x^2 + 1\right)$ haben die Form $a + bx + \left(x^2 + 1\right)$ mit eindeutig bestimmten $a, b \in \R$. Daher gilt: 
\[\begin{array}{lcl}
\left[a + bx + \left(x^2 + 1\right)\right]\left[c + dx + \left(x^2 + 1\right)\right] & = & ac + \left(ad + bc\right)x + bd\underbrace{x^2}_{x^2 + 1 - 1} + \left(x^2 + 1\right)\\
& = & ac + \left(ad + bc\right)x - bd + \left(x^2 + 1\right)\\
& = & \left(ac - bd\right) + \left(ad + bc)\right)x + \left(x^2 + 1\right)
\end{array}\]
Daher ist die folgende Abbildung ein Isomorphismus von Körpern: \[f \colon \C \mapsto \R[x]/\left(x^2 + 1\right), a + b\operatorname{i} \mapsto a + bx + \left(x^2 + 1\right)\]\footnotesize (Dies ist eine bessere Methode, um die Komplexen Zahlen einzuführen.)\normalsize
\end{bsp}

% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}\label{bem:6.7}
Die \addindex{Charakteristik} $p = char(K)$ von $K$ ist definiert durch $p := \min{\left\{n \in \N \colon \underbrace{1 + 1 + 1 + \ldots + 1}_{Summanden} = 0\right\}}$, falls das Minimum existiert. Anderenfalls sagt man, dass $K$ die Charakteristik $0$ hat, und schreibt: $char(K) = 0$ z.B. ist $char(\Q) = char(\R) = char(\C) = 0$, und $char(\F_p) = p$ für $p \in \PZ$.\\
Für jeden Körper mit $char(K) \neq 0$ ist $p := char(K) \in \PZ$, denn aus $p = ab$ mit $a, b \in \N$ folgt: \[0 = \underbrace{1 + \ldots + 1}_{p Summanden} = \underbrace{\left(1 + \ldots + 1\right)}_{a}\underbrace{\left(1 + \ldots + 1\right)}_{b},\] d.h. $\underbrace{1 + \ldots + 1}_{a} = 0$ oder $\underbrace{1 + \ldots + 1}_{b} = 0$
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Satz 6.4
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:6.4}
Für den Primkörper $P$ von $K$ gilt:
\begin{enumerate}[(i)]
	\item $char(K) = 0 \Rightarrow P \cong \Q$
	\item $char(K) = p \in \PZ \Rightarrow P \cong \F_p$
\end{enumerate}
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
\begin{enumerate}[(i)]
	\item Für $n \in \N$ sei $n1_K := \underbrace{1 + \ldots + 1}_{n \text{ Summanden}}$.\\
	Wegen $char(K) = 0$ ist $n1_K \neq 0$ für alle $n \in \N$. Wir setzen $0 \cdot 1_K := 0$ und $(-n)1_K := -(n1_K).$ Dann gilt für $m, n \in \Z \colon \left(m \stackrel{\pm}{\cdot} n\right)1_K = \left(m1_K\right) \stackrel{\pm}{\cdot} \left(n1_K\right)$.\\
	Wir definieren eine Abbildung $f \colon \Q \mapsto K$ durch $f(\frac{m}{n}) := \frac{m1_K}{n1_K} \ \left(m, n \in \Z, n \neq 0\right)$. Dann ist $f$ wohldefiniert, aus $\frac{m}{n} = \frac{m'}{n'}$ folgt nämlich $mn' = m'n$, d.h. $\underbrace{\left(mn'\right)1_K}_{\left(m1_K\right)\left(n'1_K\right)} = \underbrace{\left(m'n\right)1_K}_{\left(m'1_K\right)\left(n1_K\right)}$ und damit $\frac{m1_K}{n1_K} = \frac{m'1_K}{n'1_K}$\\
	Man rechnet leicht nach, dass $f$ ein Homomorphismus ist. Daher ist $Bld(f)$ ein zu $\Q$ isomorpher Teilkörper von $K$. Nach Definition von $P$ ist $P \subseteq Bld(f)$. Anderenseits ist $1_K \in P$, $1_K + 1_K \in P, \ldots$. Daher: $n1_K \in P$ für $n \in \N$. Also $\frac{m1_K}{n1_K} \in P$ für alle $m, n \in \Z$ mit $n \neq 0$. Daher: $Bld(f) \subseteq P$, d.h. $P = Bld(f) \cong \Q$
	\item Sei $char(K) = p > 0$, also $p \in \PZ$. Wir definieren $n1_K$ für $n \in \Z$ wie in (i). Nach Voraussetzung ist $p1_K = 0$. Wir definieren eine Abbildung $f \colon \ZmZ{p} \mapsto K$ durch $f(a + p\Z) := a1_K \ (a \in \Z)$. Das ist wohldefiniert, ist nämlich $a' \in \Z$ mit $a + p\Z = a' + p\Z$, so existiert ein $z \in \Z$ mit $a' = a +pz$.\\
	Folglich: $a'1_K = a1_K + \underbrace{\left(p1_K\right)}_{= 0}\left(z1_K\right) = a1_K$. Man rechnet leicht nach, dass $f$ ein Homomorphismus ist. Wie in (i) folgt: \[\ZmZ{p} \cong Bld(f) = P.\]
\end{enumerate}.
\end{beweis}

\footnotesize (Körper mit Charakteristik $0$ ist immer isomorph zu $\Q$, Körper mit Charakteristik $p \in \PZ$ ist immer isomorph zu $\ZmZ{p}$) \normalsize

%%%%%%%%%%%%%%%%%%%
% Satz 6.5
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:6.5}
Sei $K$ endlich. Dann $p := char(K) > 0$ und $\left|K\right| = p^n$ für ein $n \in \N$.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Im Fall $char(K) = 0$ wäre der Primkörper $P$ von $K$ zu $\Q$ isomorph, also unendlich. Also: $p := char(K) > 0$. Dann $K|P$ Körpererweiterung. Da $K$ endlich ist, ist $n := \left[K:P\right] < \infty$.\\
Folglich: $\left|K\right| = \left|P\right|^n = p^n$.
\end{beweis}

\begin{bem}\label{bem:6.8}
Es gibt also keine Körper mit $6$ oder $10$ Elementen. Anzahl kann nur $p^n \ \left(p \in \PZ\right)$ sein.
\end{bem}

% Bemerkung

\begin{bem}\label{bem:6.9}
Für ein Element $a$ einer Gruppe $G$ nennt man $ord(a) := \min{\left\{n \in \N \colon a^n = 1\right\}}$ die Ordnung von $a$. Existiert das Minimum nicht, so sagt man, dass $a$ unendliche Ordnung hat, und schreibt: $ord(a) = \infty$. Ist $G$ endlich, so hat jedes Element in $G$ endliche Ordnung (ÜA 10).\\
Im Folgenden sei $ord(a) = n < \infty$.\\
Nach ÜA 10 enthält dann \[\left\{a^m \colon m \in \Z\right\} = \left\{1, a, a^2, \ldots, a^{n-1}\right\}\] genau $n$ verschiedene Elemente.
Genauer gilt: $a^k = a^l \Leftrightarrow k \equiv l \pmod{n}$. Insbesondere gilt für $m \in \Z \colon a^m = 1 \Leftrightarrow n|m$. Für $i \in \Z$ folgt daraus: $ord(a^i) = \nicefrac{n}{\ggt(i, n)}$.
\end{bem}

\end{satz}

% Beispiel

\begin{bsp}\label{bsp:6.5}
Sei $\left|K\right| = p^n \ \left(p \in \PZ, n \in \N\right)$. Dann existiert ein $a \in K \backslash \left\{0\right\}$ mit $K \backslash \left\{0\right\} = \left\{1, a, a^2, \ldots, a^{p^n-2}\right\}$ und $a^{p^n - 1} = 1$. Man nennt $a$ eine \addindex{Primitivwurzel} in $K$.\\
\footnotesize (2 eignet sich oft als Primitivwurzel.) \\ \normalsize
In $\underbrace{\F_{17}}_{\text{Multiplikative Gruppe hat 16 Elemente!}}$ gilt: 
\[2^0 = 1,\ 2^1 = 2,\ 2^2 = 4,\ 2^3 = 8,\ 2^4 = 16 = -1,\ 2^8 = \left(-1\right)^2 = 1\]
\[3^0 = 1,\ 3^1 = 3,\ 3^2 = 9,\ 3^4 = -4,\ 3^8 = \left(-4\right)^2 = 16 = -1,\ 3^{16} = \left(-1\right)^2 = 1\]
\footnotesize ($\Rightarrow$ Alle Potenzen von $3^0$ - $3^{15}$ sind verschieden im Gegensatz zu $2^0$ - $2^{15}$.)\\ \normalsize
Daher ist $2$ keine Primitivwurzel, aber $3$.
\end{bsp}

%%%%%%%%%%%%%%%%%%%
% Satz 6.7
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:6.7}
Sei $char(K) = p > 0$. Dann ist die Abbildung $f \colon K \mapsto K,\ x \mapsto x^p$ ein Körperendomorphismus.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Offenbar ist $f(1) = 1$, und für $x, y \in K$ ist $f(xy) = \left(xy\right)^p = x^py^p = f(x)f(y)$, und $f(x + y) = (x + y)^p = x^p + \binom{p}{1}x^{p-1}y + \ldots + \binom{p}{p-1}xy^{p-1} + y^p$. Für $i = 1, \ldots, p-1$ ist $\binom{p}{i} \in \Z$ und $\binom{p}{i} = \frac{p!}{i!\left(p - i\right)!}$, d.h. $p|\binom{p}{i}$.\\
Daher: $\binom{p}{i}1_K = \underbrace{1 + \ldots + 1}_{p_i \text{ Summanden}} = 0$ und $\binom{p}{i}x^{p-1}y^i = 0$.\\
Folglich: $(x + y)^p = x^p + y^p = f(x) + f(y)$.
\end{beweis}

% Bemerkung
%%%%%%%%%%%%%%%%%%%
\begin{bem}\label{bem:6.10}
Ist $K$ endlich, so ist $f$ bijektiv, denn nach Bemerkung \ref{bem:6.6}(iv) ist $f$ injektiv. Man spricht dann auch von dem \addindex{Frobenius-Automorphismus} von $K$.
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Satz 6.8
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:6.8}
Sei $E$ ein endlicher Körper und $\left|E\right| = p^e \ \left(p \in \PZ, e \in \N\right)$.
\begin{enumerate}[(i)]
	\item Ist $F \subseteq E$ Teilkörper, so ist $\left|F\right| = p^f$ für ein $f \in \N$ mit $f|e$.
	\item Umgekehrt existiert zu jedem positiven Teiler $f$ von $e$ genau ein Teilkörper $F \subseteq E$ mit $\left|F\right| = p^f$, nämlich $F := \left\{x \in E \colon x^{p^f} = x\right\}$.
\end{enumerate}
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beispiel
%%%%%%%%%%%%%%%%%%%
\begin{bsp}\label{bsp:6.6}
Ein Körper $K$ mit $2^{12}$ Elementen hat also Teilkörper mit $2,\ 2^2,\ 2^3,\ 2^4,\ 2^6,\ 2^{12}$ Elementen.
\end{bsp}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
\begin{enumerate}[(i)]
	\item Sei $F \subseteq E$ Teilkörper. Dann: $p = char(E) = char(F)$, also $\left|F\right| = p^f$ für ein $f \in \N$. Sei $n := \left[E:F\right]$.\\
	Dann: $E \cong F^n$ als $F$-Vektorraum.\\
	Insbesondere $p^e = \left|E\right| = \left|F\right|^n = p^{f^n}$, d.h. $f|e$.
	\item Sei $f \in \N$ mit $f|e$. Nach ÜA 5 ist dann: $p^f-1|p^e-1$. Nach Beispiel \ref{bsp:6.5} existiert ein $a \in E$ mit $a^{p^e-1} = 1$ und $E = \left\{0, 1, a, a^2, \ldots, a^{p^e-2}\right\}$. Die Elemente \[b_i := a^{i\left(p^e-1\right)/\left(p^f-1\right)} \ \left(i = 0, 1, \ldots, p^f-2\right)\] sind paarweise verschieden mit $b_i^{p^f-1} = 1$, d.h. $b_i^{p^f} = b_i$. Daher sind $0, b_0, b_1, \ldots, b_{p^f-2}$ genau die Nullstellen von $x^{p^f} - x$ in $E$. Folglich enthält $F := \left\{b \in E \colon b^{p^f} = b\right\}$ genau $p^f$ Elemente. Diese bilden einen Teilkörper von $E$: denn $0, 1 \in F$, und für $b, c \in F$ gilt: $\left(b \stackrel{\pm}{\cdot} c\right)^{p^f} \overbrace{=}^{\text{Satz } \ref{satz:6.7}} b^{p^f} \pm c^{p^f} = b \stackrel{\pm}{\cdot} c$, d.h. $b \stackrel{\pm}{\cdot} c \in F$. Analog $\frac{b}{c} \in F$ für $c \neq 0$.\\
	\textsc{Eindeutigkeit:} Ist $F' \subseteq E$ ein weiterer Teilkörper mit $\left|F'\right| = p^f$, so ist $b^{p^f} = b$ für alle $b \in F'$. Nach Beispiel \ref{bsp:6.5}. Daher: $F' \subseteq F$, d.h. $F' = F$.
\end{enumerate}.
\end{beweis}

\chapter{Algebraische Körpererweiterungen}

%%%%%%%%%%%%%%%%%%%
% Definition 7.1
%%%%%%%%%%%%%%%%%%%
\begin{defi}\label{defi:7.1}
Sei $L|K$ eine Körpererweiterung. Ein $b \in L$ heißt \addindex{Algebraisch} über $K$, falls ein $\varphi \in K[x] \backslash \left\{0\right\}$ mit $\varphi(b) = 0$ existiert. Anderenfalls: $b$ \addindex{Transzendent} über $K$.\\
Sind alle $b \in L$ algebraisch über $K$, so heißt $L|K$ eine \addindex{Algebraische Körpererweiterung} (anderenfalls: \addindex{Transzendente Körpererweiterung}).
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Beispiel 7.1
%%%%%%%%%%%%%%%%%%%
\begin{bsp}\label{bsp:7.1}
\begin{enumerate}[(i)]
	\item Jedes $a \in K$ ist algebraisch über $K$ als Nullstelle von $x - a \in K[x]$.
	\item $i \in \C$ ist algebraisch über $\R$ als Nullstelle von $x^2 + 1 \in \R[x]$.
	\item $\sqrt[3]{2} \in \R$ ist algebraisch über $\Q$ als Nullstelle von $x^3 - 2 \in \Q[x]$.
	\item Man kann zeigen: $e, \pi$ sind transzendent über $\Q$.
\end{enumerate}
\end{bsp}

%%%%%%%%%%%%%%%%%%%
% Bemerkung 7.1
%%%%%%%%%%%%%%%%%%%
\begin{bem}\label{bem:7.1}
Sei $b \in L$ algebraisch über $K$.\\
Wähle Polynom $\mu(b) = 0$, o.B.d.A. $\mu$ normiert. Dann ist $\mu$ eindeutig bestimmt. Ist nämlich $\nu \in K[x] \backslash \left\{0\right\}$ normiert mit $\deg{\nu} = \deg{\mu}$ und $\nu(b) = 0$, so ist $\mu - \nu \in K[x]$ mit $\deg{\mu - \nu} < \deg{\mu}$ und $\left(\mu - \nu\right)(b) = 0$. Also: $\mu - \nu = 0$ nach Wahl von $\mu$. Also: $\mu = \nu$.\\
Bezeichnung: $\mu$ \addindex{Minimalpolynom} von $b$ über $K$.
\end{bem}

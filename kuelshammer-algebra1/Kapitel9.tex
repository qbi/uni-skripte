% Bemerkung 9.1
%%%%%%%%%%%%%%%%%%%
\begin{bem}[Klassische Konstruktionsprobleme]\label{bem:9.1}
\begin{enumerate}[(i)]
	\item Quadratur des Kreises\\
	Zu einem vorgegebenen Kreis konstruiere man (nur mit Zirkel und Lineal) ein Quadrat mit dem gleichen Flächeninhalt.
	\item Rektifikation des Kreises\\
	Zu einem vorgegebenen Kreis konstruiere man eine Strecke deren Länge genau der Umfang des Kreises ist.
	\item Delisches Problem\footnote{Name kommt von griechischer Insel Delos}\\
	Zu einem gegebenen Würfel konstruiere man einen Würfel mit doppeltem Volumen.
	\item Winkeldreiteilung\\
	Man gebe ein Verfahren an, mit dem man beliebige Winkel in 3 gleiche Teile teilen kann.
	\item Konstruktion regelmäßiger n--Ecke\\
	Für $n = 3, 4, 5, 6, \ldots$ konstruiere man nur mit Zikel und Lineal ein regelmäßiges n--Eck.
\end{enumerate}
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Definition 9.1
%%%%%%%%%%%%%%%%%%%
\begin{defi}\label{defi:9.1}
Gegeben sei eine Teilmenge $M = M_0 \subseteq \R^2 = \C$ mit $0, 1 \in M$.\\
Dann sei $\mathcal{G}(M)$ die \highl{Menge!aller Geraden}{Menge aller Geraden}, die durch mindestens zwei Punkte in $M$ gehen, und $\mathcal{K}(M)$ die \highl{Menge!aller Kreise}{Menge aller Kreise}, deren Mittelpunkt in $M$ liegt und deren Radius gleich dem Abstand von zwei Punkten in $M$\randbem{Es muss kein Punkt auf der Peripherie liegen.} ist.\\
Für $i \in \N$ sei $M_i$ die Menge, die aus den Elementen in $M_{i-1}$ und den Schnittpunkten von zwei verschiedenen Geraden in $\mathcal{G}(M_{i-1})$, den Schnittpunkten von zwei verschiedenen Kreisen in $\mathcal{K}(M_{i-1})$ und den Schnittpunkten einer Geraden in $\mathcal{G}(M_{i-1})$ mit einem Kreis in $\mathcal{K}(M_{i-1})$ besteht.\\
Die Punkte in $M_{\infty} loneqq \bigcup\limits_{i = 0}^{\infty}M_i$ heißen dann \highl{Punkte!Konstruierbar}{konstruierbar}.
\end{defi}

% Satz 9.1
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:9.1}
Für jede Teilmenge $M \subseteq C$ mit $0, 1 \in M$ ist $M_{\infty}$ ein Teilkörper von $\C$, der mit einer komplexen Zahl $z$ auch $\overline{z}$ und jede Quadratwurzel aus $z$ enthält\randbem{d.\,h. $M_{\infty}$ abgeschlossen gegenüber $+, -, \cdot, :, -, \sqrt{ \ }$}.
\end{satz}

% Satz 9.2
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:9.2}
Für jeden Teilkörper $L \subseteq C$ mit $L = \overline{L} $\randbem{$L = \overline{L} \Rightarrow$ abgeschlossen gegenüber konjungiert komplex.} und $\operatorname{i} \in L$ gilt: \randbem{Zu gewissen Teil Umgekehrung von \autoref{satz:9.1}.}
\begin{enumerate}[(i)]
	\item Ist $z \in \C$ Schnittpunkt von zwei verschiedenen Geraden $\mathcal{G}(L)$, so ist $z \in L$
	\item Ist $z \in \C$ Schnittpunkt einer Gerade in $\mathcal{G}(L)$ und eines Kreises in in $\mathcal{K}(L)$, so ist $z \in L(w)$ für ein $w \in \C$ mit $w^2 \in L$
	\item Ist $z \in \C$ Schnittpunkt von zwei verschiedenen Kreisen in $\mathcal{K}(L)$, so ist $z \in L(w)$ für ein $w \in \C$ mit $w^2 \in L$
\end{enumerate}
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
\begin{enumerate}[(i)]
	\item Seien $a, b, c, d \in L$ mit $(a + \R b) \cap (c + \R d) = \left\{z\right\}$, d.\,h. $a + \lambda b = z = c + \mu d$ für geeignete $\lambda, \mu \in \R$.\\
	Dann ist $(\lambda, \mu)$ die einzige Lösung des linearen Gleichungssystems:
	\[\begin{array}{lcl}
		\lambda\Re(b) - \mu\Re(d) & = & \Re(c - d)\\
		\mu\Im(b) - \mu\Im(d) & = & \Im(c - a)
	\end{array}\]
	und $(\lambda, \mu)$ liegt in $L \times L$; wegen $\Re(b) = \frac{b + \overline{b}}{2}$, $\Im(b)= \frac{b + \overline{b}}{2\operatorname{i}}$, $\ldots \in L$.\\
	Daher: $z = a + \lambda b \in L$
	
	\item Sei $z$ Schnittpunkt der Geraden $a + \R b$ $(a, b \in L)$ mit dem Kreis \[\left\{x + \operatorname{i}y \colon x, y \in \R, (x - c)^2 + (y - d)^2 = \rho^2\right\} \ \ (c, d \in L, \rho \in L).\]
	Schreibe $z = a + \lambda b \ (\lambda \in \R)$. Dann:
	\[(\Re(a) + \lambda\Re(b) - c)^2 + (\Im(a) + \lambda\Im(b) - d)^2 = \rho^2\]
	Dies ist eine quadratische Gleichung für $\lambda$ mit Kooeffizienten in $L \cap \R$.\\
	Ist $D$ die entsprechende Diskriminante, so ist $D \in L$ und $\lambda \in L(\sqrt{D})$, also $z = a + \lambda b \in L(\sqrt{D})$.
	
	\item Sei $z$ Schnittpunkt der Kreise mit folgenden Gleichungen:
	\[(x - a_j)^2 + (y - b_j)^2 = g_j^2 \ \ \ (a_j, b_j, \rho_j \in L, j = 1, 2).\]
	Subtraktion ergibt eine Gleichung der Form $(a_1 - a_2)x + (b_1 - b_2) = c, \ (c \in L \cap \R)$.
	Da die Mittelpunkte $a_1 + \operatorname{i}b_1, a_2 + \operatorname{i}b_2$ der gegebenen Kreise verschieden sind, beschreibt diese Gleichung eine Gerade, auf der mindestens zwei Punkte aus $L$ liegen. Also folgt die Behauptung aus (ii).
\end{enumerate}
\end{beweis}

\begin{defi}\label{defi:9.2}
Für jede Körpererweiterung $L\mid K$ und jede Teilmenge $M \subseteq L$ sei $K(M)$ der Durchschnitt aller Teilkörper von $L$, die sowohl $K$ als auch $M$ enthalten.
\end{defi}

% Satz 9.3
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:9.3}
Für jede Teilmenge $M \subseteq C$ mit $0, 1 \in M$ und jedes $z \in \C$ gilt:\\
$z$ ist aus $m$ konstruierbar $\Leftrightarrow \exists$ Folge von Teilkörpern 
\[\Q(M \cup \overline{M}) = L_0 \subseteq L_1 \subseteq \ldots \subseteq L_M\] von $\C$ mit $z \in L_m$ und $\left[L_n:L_{n-1}\right] = 2$ für $n = 1, \ldots, m$.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
	\begin{itemize}
		\item["`$\Rightarrow$"'] Die angegebene Bedingung sei erfüllt. Nach \autoref{satz:9.1} ist $\Q(M \cup \overline{M}) \subseteq M_{\infty}$. Sei also $n \in \left\{1, \ldots, m\right\}$ mit $L_{n-1} \subseteq M_{\infty}$, und sei $x \in L_n \backslash L_{n-1}$. Dann ist $L_n = L_{n-1}(x)$ und das Minimalpolynom von $x$ über $L_{n-1}$ hat die Form $\mu = X^2 + aX + b \ (a, b \in L_{n-1})$.\\
		Also: $(x + \frac{a}{2})^2 = x^2 + ax + \frac{a^2}{4} = \frac{a^2}{4} - b \in L_{n-1} \subseteq M_{\infty}$, also auch $x + \frac{a}{2} \in M_{\infty}$ nach 9.1. Damit $L_n = L_{n-1}(x + \frac{a}{2}) \subseteq M_{\infty}$.\\
		Insbesondere $z \in M_{\infty}$.
		\item["`$\Leftarrow$"'] Sei $z \in M_{\infty}$, also $z \in M$, für ein $k \in \N_{0}$, \OE{} $k \in \N$. Dann ist $z$ in einem Schritt aus endlich vielen Punkten $y_1, \ldots, y_n \in M_{k-1}$ konstruierbar. Wir können annehmen, dass für $y_1, \ldots, y_n$ die Behauptung schon gilt. Für $j = 1, \ldots, n$ existiert also eine Folge $\Q(M \cup \overline{M}) = L_{j_0} \subseteq L_{j_1} \subseteq \ldots \subseteq L_{j_n}$ von Teilkörpern von $\C$ mit $y_j \in L_{j_{e_j}}$ und $\left[L_{j_k}:L_{j_{k-1}}\right] = 2$ für $k = 1, \ldots, e_j$. Jedes $L_{j_k}$ entsteht aus $L_{j, k-1}$ durch Adjunktion einer Quadratwurzel. Adjungiert man zu $\Q(M \cup \overline{M})$ nacheinander all diese Quadratwurzeln, so erhält man eine Folge von Teilkörpern $\Q(M \cup \overline{M}) = K_0 \subseteq K_1 \subseteq \ldots \subseteq K_r$ von $\C$ mit $i, y_1, \ldots, y_n \in K_r = \overline{K_r}$. Aus \autoref{satz:9.2} folgt die Behauptung für $z$.
	\end{itemize}
\end{beweis}

% Bemerkung 9.3
%%%%%%%%%%%%%%%%%%%
\begin{bem}\label{bem:9.3}
\begin{enumerate}[(i)]
	\item Delisches Problem:\\
	Zu einem gegebenen Würfel der Kantenlänge 1 konstruiere man einen Würfel mit dem doppelten Volumen. Für die Kantenlänge $x$ des gesuchten Würfels gilt also $x^3 = 2$, aber $\left[\Q(\sqrt[3]{2}):\Q\right] = 3$. \randbem{Weil das Minimalpolynom von $\sqrt[3]{2}$ über $\Q$ Grad 3 hat $(X^3 - 2)$} Deshalb ist die Konstruktion nach \autoref{satz:9.3} unmöglich.
	
	\item Quadratur des Kreises:\\
	Zu einem gegebenen Kreis mit Radius 1 konstruiere man ein Quadrat mit dem gleichen Flächeninhalt. Für die Seitenlänge $x$ des gesuchten Qudrates gilt also $x^2 = \pi$. Wäre $\sqrt{\pi}$ aus $\left\{0, 1\right\}$ konstruierbar, so auch $\pi$. Dann wäre $\left[\Q(\pi):\Q\right]$ eine Potenz von 2, also wäre $\pi$ algebraisch über $\Q$. Man kann zeigen, dass $\pi$ transzendent über $\Q$ ist (\textsc{Lindemann} 1852--1939). Daher ist die Quadratur des Kreises unmöglich.
	
	\item Rektifikation des Kreises:\\
	Zu einem gegebenen Kreis $K$ konstruiere man eine Strecke, deren Länge $x$ gleich dem Umfang von $K$ ist. \OE{} habe $K$ Radius 1. Es gilt also: $x = 2\pi$. Da $\pi$ transzendent über $\Q$ ist, ist das nicht möglich.
	
	\item Dreiteilung von Winkeln:\randbem{beliebige Winkel!}\\
	Zu einem gegebenen Winkel $\varphi$ soll der Winkel $\frac{\varphi}{3}$ konstruiert werden. Die Frage ist also, ob $e^{\frac{i\varphi}{3}}$ aus $\left\{0, 1, e^{i\varphi}\right\}$ konstruiert werden kann. Äquivalent dazu ist die Konstruierbarkeit von $x = \cos{\frac{\varphi}{3}}$ aus $\left\{0, 1, \cos{\varphi}\right\}$. Wegen 
	\[\cos{(3\beta)} = 4\cos^3\beta - 3\cos\beta\]
	ist $x$ Nullstelle von $4X^3 - 3X - c$. Im Fall $\varphi = \frac{\pi}{3}$ ist $\cos\varphi = \frac{1}{2} = c$, also $x$ Nullstelle von $8X^3 - 6X - 1$ und $2x$ Nullstelle von $X^3 - 3X - 1$. Wir behaupten: $X^3 - 3X - 1$ irreduzibel in $\Q\left[X\right]$. Anderenfalls hätte $X^3 - 3X - 1$ eine Nullstelle $y$ in $\Q$. Schreribe $y = \frac{m}{n}$ $(m, n \in \Z, \ggt(m, n) = 1)$. Dann: $m^3 - 3mn^2 - n^3 = 0$. Also ist jeder Primteiler von $m$ auch Teiler von $n$ und umgekehrt. Daher ist $y = \pm 1$, was aber auch nicht geht.
	Also ist $X^3 - 3X - 1$ irreduzibel in $\Q\left[X\right]$ und $\left[Q(x):Q\right] = 3$. Aus \autoref{bem:9.2} folgt, dass die Winkeldreiteilung allgemein nicht geht.
	
	\item Konstruktion des regelmäßigen 7--Ecks:\\
	Wir nehmen an, dass wir regelmäßiges 7--Eck mit Zirkel und Lineal konstruieren können. Dann ist $\rho = e^{\frac{2\pi i}{7}}$ aus $\left\{0, 1\right\}$ konstruierbar, also auch $\eta loneqq \rho + \rho^{-1} = 2\cos{\frac{2\pi}{7}}$. Wir wissen
	\[\begin{array}{lcl}
		0 & = & \rho^7 - 1 = \underbrace{(\rho - 1)}_{\neq 0}(\rho^6 + \rho^5 + \rho^4 + \rho^3 + \rho^2 + \rho + 1)\text{, also}\\
		0 & = & \rho^{-3}(\rho^6 + \rho^5 + \rho^4 + \rho^3 + \rho^2 + \rho + 1)\\
		& = & \rho^3 + \rho^2 + \rho + 1 + \rho^{-1} + \rho^{-2} + \rho^{-3}\\
		& = & (\rho^3 + \rho^{-3}) + (\rho^2 + \rho^{-2}) + (\rho + \rho^{-1}) + 1
	\end{array}\]
	Wegen $\eta^2 = \rho^2 + \rho^{-2} + 2$ und $\eta^3 = (\rho^3 + \rho^{-3}) + 3(\rho + \rho^{-1})$ folgt
	\[0 = (\eta^3 - 3\eta) + (\eta^2 - 2) + \eta + 1 = \eta^3 + \eta^2 - 2\eta - 1\]
	Das Polynom $X^3 + X^2 - 2X - 1$ ist aber irreduzibel in $\Q\left[X\right]$, denn andernfalls hätte es eine Nullstelle $y \in \Q$. Schreibe $y = \frac{m}{n}$ mit $m, n \in \Z, \ \ggt(m, n) = 1$. Dann: $m^3 + m^2n - 2mn^2 - n^3 = 0$. Daher ist jeder Primteiler von $m$ einer von $n$ und umgekehrt. Also $y = \pm 1$, was aber auch nicht geht.\\
	Daher ist $X^3 + X^2 - 2X - 1$ tatsächlich das Minimalpolynom von $\eta$ über $\Q$. Also ist $\left[\Q(\eta):\Q\right] = 3$, und $\eta$ nicht aus $\left\{0, 1\right\}$ konstruierbar.
	
	\item Konstruktion des regelmäßigen 5--Ecks:\\
	Wir zeigen, dass $\rho = e^{\frac{2\pi i}{5}}$ aus $\left\{0, 1\right\}$ konstruierbar ist. Wegen
	\[\begin{array}{lcl}
		0 & = & \rho^5 - 1 = (\rho - 1)(\rho^4 + \rho^3 + \rho^2 + \rho + 1)\\
		0 & = & \rho^{-2}(\rho^4 + \rho^3 + \rho^2 + \rho + 1)\\
		& = & \rho^2 + \rho + 1 + \rho^{-1} + \rho^{-2}\\
		& = & (\rho^2 + \rho^{-2}) + (\rho + \rho^{-1}) + 1
	\end{array}\]
	Für $\eta = \rho + \rho^{-1} = 2\cos{\frac{2\pi}{5}}$ gilt $\eta^2 = \rho^2 + \rho^{-2} + 2$. Also gilt:
	\[0 = (\eta^2 - 2) + \eta + 1 = \eta^2 + \eta - 1 \text{,d.\,h. } \eta = -\frac{1}{2} + \frac{\sqrt{5}}{2}\]
	Daher ist $\eta$ aus $\left\{0, 1\right\}$ konstruierbar. Folglich ist auch $\rho$ aus $\left\{0, 1\right\}$ konstruierbar.
	Wegen $\eta^2 = \rho^2 + \overbrace{\rho^3}^{= \rho^{-2}\text{, da }\rho^5 = 1} + 2$ ist
	\[0 = \rho^2 + \rho(\rho^3 + \rho^2 + 1) + 1 = \rho^2 + \rho(\eta^2 - 1) + 1 = \rho^2 + \rho(\frac{1}{2} - \frac{\sqrt{5}}{2}) + 1\]
	Folglich: $\rho = \frac{1}{4}(-1 + \sqrt{5} + \sqrt{-10 - 2 \cdot \sqrt{5}})$ \randbem{$\rho \ \Rightarrow$ konstruierbar}
	
	\item Konstruktion regelmäßiger n--Ecke:\\
	\textsc{Gauß} zeigte, dass man ein regelmäßiges 17--Eck mit Zirkel und Lineal konstriuieren kann. 
	Allgemeiner zeigte er, dass man ein regelmäßiges n--Eck genau dann mit Zirkel und Lineal konstruieren kann, wenn $\varphi(n)$ [Eulerische $\varphi$--Funktion] eine 2er--Potenz ist. 
	Aus \autoref{satz:3.5} folgt leicht, dass dies genau dann der Fall ist, wenn eine Primfaktorzerlegung \[n = 2^a \cdot p_1 \cdot \ldots \cdot p_r\] mit paarweise verschiedenen "`Fermatschen"' Primzahlen $p_1, \ldots, p_r$ ist.
	Dabei heißt $p \in \PZ$ \highl{Fermat!Primzahl}{Fermatsche Primzahl}, wenn $p-1$ eine 2er--Potenz ist.
	Man sieht sofort, dass $p$ dann die folgende Form haben muss: \[p = 2^{2^k} + 1 \ (k \in \N_0).\]
	Für $k = 0, 1, 2, 3, 4$ erhält man die Primzahlen \[p = 3, 5, 17, 257, 65537.\]
	Bis heute kennt man keine weitere Fermat--Primzahlen. Nach Aufgabe 11 ist $2^{2^5} + 1 \notin \PZ$. Man weiß, dass für $k = 5, \ldots, 32$ keine Primzahl erhält. 
	Offen ist der Fall $k = 33$. Man weiß nicht, ob $2^{2^k} + 1$ für undendlich viele $k$ eine Primzahl (keine Primzahl) ist.
\end{enumerate}
\end{bem}

\begin{bem}[Quadratische Gleichungen]\label{bem:9.4}
Löse $X^2 + aX + b = 0$ $(a, b \in \C)$\\
\emph{Methode:} Quadratische Ergänzung\\
$0 = (X + \frac{a}{2})^2 + b - \frac{a^2}{4}$ $\Rightarrow$ dann Wurzel ziehen
\emph{Beispiel:} 
\[\begin{array}{lcl}
0 = X^2 + X + 1 = (X - \frac{1}{2}) + 1 - \frac{1}{4} & \Leftrightarrow & X + \frac{1}{2} = \pm \sqrt{-\frac{3}{4}} = \pm \frac{\sqrt{3}}{2}i\\
& \Leftrightarrow & X = -\frac{1}{2} \pm \frac{\sqrt{3}}{2}i
\end{array}\]
Wegen $(X-1)(X^2 + X + 1) = X^3 - 1$ sind $1, -\frac{1}{2} \pm \frac{\sqrt{3}}{2}i$ genau die Lösungen von $X^3 = 1$ in $\C$.
\end{bem}

\begin{bem}[Kubische Gleichungen]\label{bem:9.5}
Löse $X^3 + aX^2 + bX + c = 0$ $(a, b, c \in \C)$ mit Hilfe von $+, -, \cdot, /, \sqrt{ \ }$.\\
\begin{itemize}
	\item[\emph{Methode:}] \textsc{S. del Ferro} (1465--1526, geheim)\\
	\textsc{G. Cardano} (1501--1576, ohne Genehmigung veröffentlicht)
	
	\item[\emph{Heute:}] Cardanosche Formel\\
	Zunächst kubische Ergänzung
	\[\begin{array}{lcl}
		0 & = & (X + \frac{a}{3} - \frac{a}{3})^3 + a(X + \frac{a}{3} - \frac{a}{3})^2 + b(X + \frac{a}{3} - \frac{a}{3}) + c\\
		& = & (X + \frac{a}{3})^3 - a(X + \frac{a}{3})^2 + \frac{a^2}{3}(X + \frac{a}{3}) - \frac{a^3}{27} + a(X + \frac{a}{3})^2\\
		& & - \frac{2a^2}{3}(X + \frac{a}{3}) + \frac{a^3}{9} + b(X + \frac{a}{3}) - \frac{ab}{3} + c\\
		& = & (X + \frac{a}{3})^3 + (b - \frac{a^2}{3})(X + \frac{a}{3}) + (\frac{2a^3}{27} - \frac{ab}{3} + c)
	\end{array}\]
	Daher genügt es Gleichungen der Form \[X^3 + pY + q = 0 \ \ (p, q \in \C)\] zu lösen.
	
	\item[\emph{Ansatz:}]
	\[\begin{array}{lcl}
		Y & = & U + V\\
		0 & = & (U + V)^3 + p(U + V) + q\\
		& = & U^3 + 3U^2V + 3UV^2 + V^3 + pU + pV + q\\
		& = & U^3 + V^3 + q + (U + V)(3UV + p)
	\end{array}\]
	Wir lösen jetzt die Gleichungen $U^3 + V^3 + q = 0$ und $3UV = -p$.
	Dazu verwenden wir die folgende qudratische Gleichung: \randbem{Lösung ergibt Nullstellen $U^3$ und $V^3$ $\Rightarrow$ Wurzelziehen noch notwendig!}
	\[0 = (T - U^3)(T - V^3) = T^2 - (U^3 + V^3)T + U^3V^3 = T^2 + qT - \frac{p^3}{27} \text{ (\highl{Resolvente!quadratisch}{Quadratische Resolvente})}.\]
	
	\item[\emph{Beispiel:}] Zu lösen: $X^3 - 18X - 35 = 0$\\
	Quadratische Resolvente: $T^2 - 35T + 216 = 0$\\
	Lösungen: $T_1 = 27$, $T_2 = 8$\\
	$\Rightarrow$ $U^3 = 27 \longrightarrow U \in \left\{3, \frac{3}{2}(-1 \pm \sqrt{3})\right\}$\\
	$3UV = 18 \longrightarrow \begin{array}[t]{lll}U_1 = 3 & V_1 = 2 & X_1 = 5\\ U_2 = \frac{3}{2}(-1 + \sqrt{3}i) & V_2 = -1 - \sqrt{3}i & X_2 = -\frac{5}{2} + \frac{\sqrt{3}}{2}i\\ U_3 = \frac{3}{2}(-1 - \sqrt{3}i) & V_3 = -1 + \sqrt{3}i & X_3 = -\frac{5}{2} - \frac{\sqrt{3}}{2}i\end{array}$
\end{itemize}
\end{bem}

\begin{bem}[Gleichungen 4. Grades]\label{bem:9.6}
\textsc{Ferrari} (1522--1565, Schüler von \textsc{Cardano})\\
Zunächst wird die allgemeine Gleichung $X^4 + aX^3 + bX^2 + cX + d = 0$ $(a, b, c, d \in \C)$ auf eine Gleichung der Form
\[Y^4 + pY^2 + qY + r = 0 \ \ (p, q, r \in C)\]
reduziert.
\begin{itemize}
	\item[\emph{Ansatz:}] Neue Variable $Z$:
	\[\begin{array}{lcl}
		(Y^2 + Z^2) & = & Y^4 + 2Y^2Z + Z^2\\
		& = & -pY^2 - qY - r + 2Y^2Z + Z^2\\
		& = & (2Z-p)Y^2 - qY + Z^2 - r\\
		& = & (2Z-p)(Y - \frac{q}{2(2Z-p)})^2 - \frac{q^2}{4(2Z - p)} + Z^2 - r
	\end{array}\]
	
	\item[\emph{Idee:}] Wähle $Z$ so, dass der absolute Koeffizient verschwindet.
	\[0 = -q^2 + 4(Z^2 - r)(2Z - p) = 8Z^3 - 4pZ^2 - 8rZ + 4rp - q^2 \text{ (\highl{Resolvente!kubisch}{Kubische Resolvente})}\]
	Sei $Z$ Lösung. Dann bleibt noch zu lösen:
	\[(Y^2 + Z)^2 = (2Z - p)(Y - \frac{q}{2(2Z - p)})^2\]
	Dies führt auf zwei quadratische Gleichungen.
	
	\item[\emph{Beispiel:}] Zu lösen: $X^4 - 10X^2 - 20X - 16 = 0$\\
	Kubische Resolvente: $0 = 8Z^3 + 40Z^2 + 128Z + 240$\\
	Äquivalent: $0 = Z^3 + 5Z^2 + 16Z + 30$. Eine Lösung ist $Z = -3$\\
	Man erhält: $(Y^2 - 3)^2 = 4(Y + \frac{20}{8})^2 = (2Y + 5)^2$\\
	Also: $Y^2 - 3 = 2Y + 5$ oder $Y^2 - 3 = -2Y - 5$\\
	d.\,h.: $Y^2 - 2Y - 8 = 0$ oder $Y^2 + 2Y + 2 = 0$, d.\,h. $Y \in \left\{-2, -4, -1 + i, -1 - i\right\}$
\end{itemize}
\end{bem}

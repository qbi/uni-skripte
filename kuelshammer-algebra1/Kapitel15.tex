\chapter{Gruppenoperationen}
%%%%%%%%%%%%%%%%%%%
% Definition 15.1
%%%%%%%%%%%%%%%%%%%
\begin{defi}\label{defi:15.1}
Eine \highl{Gruppe!Operation}{Operation (action)} einer Gruppe $G$ auf eine Menge $\Omega$ ist eine Abbildung $G\times\Omega\mapsto\Omega$, $(g,\alpha)\mapsto {}^g\alpha$, mit den folgenden Eigenschaften:\\

\begin{enumerate}[(i)]
	\item ${}^1\alpha = \alpha$ für alle $\alpha\in\Omega$
	\item ${}^g({}^h\alpha) = {}^{gh}\alpha$ für alle $\alpha \in \Omega, g,h \in G$
\end{enumerate}
Man sagt auch: "`$G$ operiert auf $\Omega$"' oder "`$\Omega$ ist eine $G$--Menge"'.
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Satz 15.1
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:15.1}
Seien $G$ eine Gruppe und $\Omega$ eine $G$--Menge.
Für $g\in G$ ist dann
\[\tau_g\colon\Omega\mapsto\Omega, \alpha\mapsto {}^g\alpha,\]
bijektiv, d.\,h. $\tau_g\in\Sym(\Omega)$.
Außerdem ist die Abbildung
\[\tau\colon G\mapsto\Sym(\Omega), g\mapsto\tau_g,\]
ein Homomorphismus.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Für $g,h\in G$, $\alpha\in\Omega$ ist $(\tau_g\circ\tau_h)(\alpha) = {}^g({}^h\alpha) = {}^{gh}\alpha = \tau_{gh}(\alpha)$ und $\tau_1(\alpha) = {}^1\alpha = 1$, d.\,h. $\tau_g\circ\tau_h=\tau_{gh}$ und $\tau_1=id_{\Omega}$.
Daher ist $\tau_g$ bijektiv.
Die Behauptung folgt.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Satz 15.2
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:15.2}
Seien $G$ eine Gruppe, $\Omega$ eine Menge und $\tau\colon G\mapsto\Sym(\Omega)$ ein Homomorphismus.\\
Dann erhält man durch
\[{}^g\alpha:=(\tau(g))(\alpha) \qquad \text{ für } g\in G,\alpha\in\Omega\]
eine Operation von $G$ auf $\Omega$.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Da $\tau$ ein Homomorphismus ist, gilt für alle $g,h\in G$, $\alpha\in\Omega$:
\[\begin{array}{lcl}
{}^g({}^h\alpha) & = & \tau(g)\left[(\tau(h))(\alpha)\right]\\
& = & \left[\tau(g)\circ\tau(h)\right](\alpha)\\
& = & \left[\tau(gh)\right](\alpha) = {}^{gh}\alpha\\
{}^1\alpha & = & (\tau(1))(\alpha) = id_{\Omega}(\alpha) = \alpha
\end{array}\]
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Bemerkung 15.1
%%%%%%%%%%%%%%%%%%%
\begin{bem}\label{bem:15.1}
Die Sätze \autoref{satz:15.1} und \autoref{satz:15.2} zeigen, dass Gruppenoperationen im Wesentlichen das gleiche sind wie Homomorphismen von Gruppen.
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Beispiel 15.1
%%%%%%%%%%%%%%%%%%%
\begin{bsp}\label{bsp:15.1}
Für jede Menge $\Omega$ operiert jede Untergruppe $G$ von $\Sym(\Omega)$ auf $\Omega$ durch
\[{}^g\alpha:= g(\alpha) \qquad \text{ für } g\in G, \alpha\in\Omega.\]
(Dies ist der Spezialfall $\tau\colon G\mapsto\Sym(\Omega), g\mapsto g$.)
\end{bsp}

%%%%%%%%%%%%%%%%%%%
% Definition 15.2
%%%%%%%%%%%%%%%%%%%
\begin{defi}\label{defi:15.2}
Sei $\Omega$ eine $G$--Menge und $\tau\colon G\mapsto\Sym(\Omega)$ der entsprechende Homomorphismus.
Dann heißt:
\[\Ker(\tau) = \left\{g\in G\colon\tau_g=id_{\Omega}\right\} = \left\{g\in G\colon {}^g\alpha=\alpha  \qquad  \text{ für alle } \alpha\in\Omega\right\}\]
der \highl{Gruppe!Operation!Kern}{Kern} der Operation.

Ist $\Ker(\tau) = G$ [d.\,h. ${}^g\alpha=\alpha$ für alle $g\in G$, $x\in\Omega$], so nennt man die Operation \highl{Gruppe!Operation!trivial}{trivial}.
Man sagt: "`$G$ operiert trivial auf $\Omega$"'.

Ist dagegen $\Ker(\tau) = \left\{1\right\}$ [d.\,h. $\tau$ ist injektiv], so nennt man die Operation \highl{Gruppe!Operation!treu}{treu}.
Gegebenenfalls ist $G \cong \tau(G) \leq \Sym(\Omega)$.
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Satz 15.3
%%%%%%%%%%%%%%%%%%%
\begin{satz}[Cayley]\label{satz:15.3}
Jede Gruppe $G$ ist zu eine Untergruppe einer symmetrischen Gruppe isomorph.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
$G$ operiert auf sich selbst durch Multiplikation: ${}^gx:= g\cdot x$ $(g,x\in G)$.
Diese Operation ist treu, denn aus $gx = x$ für alle $x\in G$ folgt $g = 1$.\\
Also: $G \cong \tau(G) \leq\Sym(G)$.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Satz 15.4
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:15.4}
Seien $G$ eine Gruppe und $\Omega$ eine $G$--Menge.\\
Wir definieren eine Relation $\sim$ auf $\Omega$ durch
\[\alpha\sim\beta :\Leftrightarrow\exists g\in G\colon {}^g\alpha = \beta.\]
Dann ist $\sim$ eine Äquivalenzrelation.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
\begin{itemize}
	\item[\underline{Reflexivität:}] ${}^1\alpha = \alpha$
	
	\item[\underline{Symmetrie:}] $\alpha\sim\beta\Rightarrow\exists g\in G\colon {}^g\alpha=\beta\Rightarrow g^{-1}\in G$ und $g^{-1}\beta = {}^{g^{-1}}({}^g\alpha) = {}^{g^{-1}g}\alpha=\alpha$, d.\,h. $\beta\sim\alpha$
	
	\item[\underline{Transisivität:}] $\alpha\sim\beta\wedge\beta\sim\gamma\Rightarrow\exists g,h\in G\colon {}^g\alpha=\beta\wedge {}^h\beta=\gamma \Rightarrow h\cdot g\in G\wedge {}^{hg}\alpha = {}^h\beta=\gamma\Rightarrow\alpha\sim\beta$.
\end{itemize}
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Bemerkung 15.2
%%%%%%%%%%%%%%%%%%%
\begin{bem}\label{bem:15.2}
\begin{enumerate}[(i)]
	\item Für $\alpha\in\Omega$ ist also
	\[\Orb_G(\alpha)=\left\{\beta\in\Omega\colon\alpha\sim\beta\right\} = \left\{{}^g\alpha\colon g\in G\right\}\]
	die Äquivalenzklasse von $\alpha$ bezüglich $\sim$.\\
	$\Orb_G(\alpha)$ heißt \highl{Gruppe!Bahn}{Bahn} von $\alpha$ unter $G$ (\highl{Gruppe!Orbit}{Orbit}).
	
	\item Sind $\Omega_1,\ldots,\Omega_n$ die verschiedenen Bahnen von $G$ auf $\Omega$, so ist
	\[\begin{array}{lcl}
	\Omega & = & \bigcup\limits_{i=1}^{n}\Omega_i \text{, also}\\
	\abs{\Omega} & = & \abs{\Omega_1} + \dotsc + \abs{\Omega_n} \quad \text{ (\highl{Gruppe!Bahn!Bahnengleichung}{Bahnengleichung})}
	\end{array}.\]
	Für $i=1,\ldots,n$ heißt $\abs{\Omega_i}$ die \highl{Gruppe!Bahn!Länge}{Länge} der Bahn $\Omega_i$.
\end{enumerate}
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Satz 15.5
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:15.5}
Für jede Gruppe $G$, jede $G$--Menge $\Omega$ und jedes $\alpha\in\Omega$ ist
\[\Stb_G(\alpha):=G_{\alpha}:=\left\{g\in G\colon {}^g\alpha=\alpha\right\} \leq G\]
und $\abs{G:G_{\alpha}} = \abs{\Orb_G(\alpha)}$; insbesondere ist $\abs{\Orb_G(\alpha)}\Bigl|\abs{G}$ im Fall $\abs{G}<\infty$.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Definition 15.3
%%%%%%%%%%%%%%%%%%%
\begin{defi}\label{defi:15.3}
$G_\alpha$ heißt \highl{Gruppe!Stabilisator}{Stabilisator} von $\alpha$ in $G$.
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Wegen ${}^1\alpha=\alpha$ ist $1\in G_{\alpha}$.
Für $g,h\in G_\alpha$ ist ${}^g\alpha=\alpha={}^h\alpha$.
Daher ist ${}^{g^{-1}h}\alpha = g^{-1}({}^h\alpha) = {}^{g^{-1}}({}^g\alpha) = {}^{g^{-1}g}\alpha = {}^1\alpha = \alpha$, d.\,h. $g^{-1}h\in G_\alpha$.
Also: $G_\alpha \leq G$.\\
Für $g,h\in G$ gilt
\[{}^g\alpha={}^h\alpha\Leftrightarrow\alpha={}^{g^{-1}h}\alpha\Leftrightarrow g^{-1}h\in G_\alpha\Leftrightarrow gG_\alpha=hG_\alpha.\]
Daher ist die Abbildung $f\colon G/G_\alpha\mapsto\Orb_G(\alpha)$, $gG_\alpha\mapsto {}^g\alpha$ wohldefiniert und injektiv.
Sicher ist $f$ surjektiv, also auch bijektiv.\\
Insbesondere ist $\abs{G:G_\alpha}=\abs{\Orb_G(\alpha)}$.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Beispiel 15.2
%%%%%%%%%%%%%%%%%%%
\begin{bsp}\label{bsp:15.2}
Jede Gruppe $G$ operiert auf sich selbst durch Konjugation:
\[{}^gx = gxg^{-1} \qquad (g,x\in G),\]
denn ${}^1x = 1x1^{-1} = x$ und ${}^g({}^hx) = {}^g(hxh^{-1}) = ghxh^{-1}g^{-1} = (gh)x(gh)^{-1} = {}^{gh}x \qquad (g,h,x\in G)$

Die Bahn von $x\in G$ ist dabei die Konjugationsklasse $K_x = \left\{gxg^{-1}\colon g\in G\right\}$.
Liegen $x,y\in G$ in der gleichen Konjugationsklasse [d.\,h. es existiert ein $g\in G$ mit $y=gxg^{-1}$], so heißen $x,y$ konjugiert in $G$.

Der Stabilisator von $x\in G$ unter Konjugationsoperation ist
\[\left\{g\in G\colon gxg^{-1} = x\right\} = \left\{g\in G\colon gx=xg\right\} =: C_G(x);\]
$C_G(x)$ heißt \highl{Gruppe!Zentralisator}{Zentralisator} von $x$ in $G$ (Übungsaufgabe 31).
Nach \autoref{satz:15.5} ist $G_g(x)\leq G$, und $\abs{G:C_G(x)}$ ist die Länge der Konjugationsklasse von $G$ und ist $x_i\in K_i$ für $i=1,\ldots,n$, so gilt nach der Bahngleichung:
\[\abs{G} = \abs{k_1} + \dotsc+\abs{K_n}=\abs{G:C_G(x_1)}+\ldots+\abs{G:C_G(x_n)} \qquad \text{ (Klassengleichung)}\]
Dabei gilt: $\begin{array}[t]{lcl}\abs{K_i} = 1 & \Leftrightarrow & K_i = \left\{x_i\right\}\\
& \Leftrightarrow & gx_ig^{-1} = x_i \text{ für alle } x\in G\\
& \Leftrightarrow & gx_i = x_ig \text{ für alle } g\in G\\
& \Leftrightarrow & x_i\in Z(G)
\end{array}.$
\end{bsp}

%%%%%%%%%%%%%%%%%%%
% Definition 15.4
%%%%%%%%%%%%%%%%%%%
\begin{defi}\label{defi:15.4}
Sei $p\in\PZ$.
Eine endliche Gruppe $P$ mit $\abs{P}=p^n$ für $n\in\N_0$ heißt \highl{Gruppe!p--Gruppe}{$p$--Gruppe}.
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Satz 15.6
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:15.6}
Für $p\in\PZ$ und jede endliche $p$--Gruppe $P \neq \left\{1\right\}$ gilt: $Z(P) \neq \left\{1\right\}$.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Seien $K_1,\ldots,K_t$ die Konjugationklassen von $P$, \OE{} $\abs{K_i} = 1$ für $i=1\ldots,s$ und $\abs{K_i} > 1$ für $i=s+1,\ldots,t$.\\
Dann: $K_1\cup\dotsc\cup K_s = Z(P)$ und $\abs{K_i} \equiv 0\pmod{p}$ für $i=s+1,\ldots,t$.\\
Also: $0\equiv\abs{P}=\abs{K_1}+\dotsc+\abs{K_t} =\abs{Z(P)}+\abs{K_{s+1}}+\dotsc+\abs{K_t}\equiv\abs{Z(P)}\pmod{p}$.\\
Daher: $Z(P) \neq \left\{1\right\}$.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Satz 15.7
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:15.7}
Für jede Primzahl $p$ ist jede endliche $p$--Gruppe $P$ auflösbar.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
(Induktion nach $\abs{P}$)\\
Im Fall $\abs{P}=1$ ist alles klar.\\
Sei also $\abs{P}>1$.
Nach \autoref{satz:15.6} ist $\left\{1\right\}\neq Z(P)\unlhd P$ und $\abs{P/Z(P)}<\abs{P}$.
Nach Induktion ist die $p$--Gruppe $P/Z(P)$ auflösbar.
Da $Z(P)$ abelsch, also auch auflösbar ist, folgt: $P$ auflösbar.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Satz 15.8
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:15.8}
Für $p\in\PZ$ ist jede Gruppe $P$ der Ordnung $p^2$ abelsch.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Nach \autoref{satz:15.6} ist $\abs{Z(P)} > 1$, also $\abs{P/Z(P)} \in \left\{1,p\right\}$;
insbesondere ist $P/Z(P)$ zyklisch.
Nach Übungsafgabe 31(ix) ist $P$ abelsch.
\end{beweis}

\chapter{Auflösbarkeit von Gruppen und Gleichungen}
%%%%%%%%%%%%%%%%%%%
% Definition 14.1
%%%%%%%%%%%%%%%%%%%
\begin{defi}\label{defi:14.1}
Eine Gruppe $G$ heißt \highl{Gruppe!auflösbar}{auflösbar}, wenn eine Folge
\[G = G_0 \unrhd G_1 \unrhd \ldots \unrhd G_n = \left\{1\right\}\]
von Untergruppen von $G$ mit folgender Eigenschaft: $G_i \unlhd G_{i-1}$ und $G_{i-1}/G_i$ abelsch für $i = 1, \ldots, n$ existiert.
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Satz 14.1
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:14.1}
Seien $G$ eine Gruppe, $H\leq G$ und $N\unlhd G$.
Dann gilt:
\begin{enumerate}[(i)]
	\item $G$ auflösbar $\Rightarrow$ $H$ auflösbar
	
	\item $G$ auflösbar $\Rightarrow$ $G/N$ auflösbar
	
	\item $N$ auflösbar und $G/N$ auflösbar $\Rightarrow$ $G$ auflösbar
\end{enumerate}
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
\begin{enumerate}[(i)]
	\item Sei $G$ auflösbar und $G=G_0\unrhd G_1\unrhd\dotsc\unrhd G_n = \left\{1\right\}$ wie oben.
	Dann ist $H = H \cap G_0 \geq H\cap G_1\geq\dotsc\geq H\cap G_n = \left\{1\right\}$ eine Folge von Untergruppen von $H$.
	Für $i=1,\ldots,n$ gilt nach dem 1. Isomorphiesatz (\autoref{satz:10.6}): 
	\[\begin{array}{lcl}
	H\cap G_i & = & (H\cap G_{i-1})\cap G_i\\
	& \unlhd & H\cap G_{i-1}, \text{ und}\\ 
	(H\cap G_{i-1})/(H\cap G_i) & = & (H\cap G_{i-1})/((H\cap G_{i-1})\cap G)\\ 
	& \cong & ((H\cap G_{i-1})G_i)/G_i\\
	& \leq & G_{i-1}/G_i.
	\end{array}\]
	Mit $G_{i-1}/G_i$ ist auch $H\cap G_{i-1}/H\cap G_i$ abelsch.
	
	\item Sei $G$ auflösbar und $G=G_0\unrhd G_1\unrhd\dotsc\unrhd G_n = \left\{1\right\}$ wie oben.\\
	Nach dem 1. Isomorphiesatz (\autoref{satz:10.6}) ist
	\[G/N = G_0N/N \geq G_1N/N\geq\dotsc\geq G_nN/N = \left\{1\right\}\]
	eine Folge von Untergruppen von $G/N_0$.
	Für $i=1,\ldots,n$ hat jedes Element in $G_iN/N$ die Form $xN$ mit $x\in G_i$ und jedes Element in $G_{i-1}N/N$ die From $yN$ mit $y\in G_{i-1}$.
	Dabei: $(yN)(xN)\underbrace{(yN)^{-1}}_{y^{-1}N} = \underbrace{yxy^{-1}}_{\in G_i}N\in G_iN/N$.\\
	Also: $G_iN/N \unlhd G_{i-1}N/N$.
	Nach den Isomorphiesätzen (\autoref{satz:10.6} \autoref{satz:10.7}) gilt:
	\[\begin{array}{lcl}
	(G_{i-1}N/N)/(G_iN/N) & \cong & G_{i-1}N/G_iN\\
	& = & G_{i-1}(G_iN)/(G_iN)\\
	& \cong & G_{i-1}/(G_{i-1}\cap G_iN)\\
	& \cong & \frac{(G_{i-1}/G_i)}{(G_{i-1}\cap G_iN)/G_i}
	\end{array}\]
	Mit $G_{i-1}/G_i$ ist auch $(G_{i-1}/N)/(G_iN/N)$ abelsch.
	
	\item Seien $N$ und $G/N$ auflösbar.
	Dann existieren Folgen von Untergruppen $N = N_0 \unrhd N_1 \unrhd \dotsc \unrhd N_k = \left\{1\right\}$ und $G/N = G_0/N \unrhd G_1/N \unrhd \dotsc \unrhd G_l/N = \left\{1\right\}$ mit folgenden Eigenschaften:\\
	$N_i\unrhd N_{i-1}$ und $N_{i-1}/N_i$ abelsch $(i=1,\ldots,k)$, $G_j/N \unlhd G_{j-1}/N$ und $\underbrace{(G_{j-1}/N)/(G_j/N)}_{\cong G_{j-1}/G_j}$ abelsch $(j=1,\ldots,l)$.\\
	Also: $G_j \unlhd G_{j-1}$ und $G_{j-1}/G_j$ abelsch.\\
	Wir erhalten eine Folge von Untergruppen 
	\[G = G_0 \unrhd G_1 \unrhd \dotsc \unrhd G_l = N = N_0 \unrhd N_1 \unrhd \dotsc \unrhd N_k = \left\{1\right\}\]
	mit den gewünschten Eigenschaften.\\
	Also: $G$ auflösbar.
\end{enumerate}
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Bemerkung 14.1
%%%%%%%%%%%%%%%%%%%
\begin{bem}\label{bem:14.1}
Für $n\in\N$ ist nach Übungsaufgabe 34 die Abbildung $\epsilon\colon\Sym(n)\mapsto(\left\{\pm 1\right\}, \cdot)$, $g\mapsto\prod\limits_{1\leq i<j\leq n}\frac{g(j)-g(i)}{j-i}$, ein Homomorphismus.
Für $g\in\Sym(n)$ nennt man $\epsilon(g)$ das Vorzeichen von $g$.\\
Ist $\epsilon(g) = 1$, so heißt $g$ gerade, sonst ungerade.\\
Dann ist \[\Alt(n)\coloneqq\Ker(\epsilon) = \left\{g\in\Sym(n)\colon g \text{ gerade}\right\}\unlhd\Sym(n).\]
$\Alt(n)$ heißt \highl{Gruppe!alternierend}{alternierende Gruppe} des Grades $n$.

Ist $n>1$ und $t\in\Sym(n)$ derart, dass $t$ die Zahlen $1$ und $2$ vertauscht und $3,\ldots,n$ festhält, so ist $\epsilon(t) = -1$, insbesondere ist $\epsilon$ surjektiv.
Nach dem Homomorphiesatz (\autoref{satz:10.5}) ist also $\Sym(n)/\Alt(n) \cong (\left\{\pm 1\right\}, \cdot)$.\\
Daher ist $\abs{\Sym(n):\Alt(n)} = 2$ und $\abs{\Alt(n)} = \frac{n!}{2}$.
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Beispiel 14.1
%%%%%%%%%%%%%%%%%%%
\begin{bsp}\label{bsp:14.1}
\begin{enumerate}[(a)]
	\item Jede abelsche Gruppe ist auflösbar.
	Betrachte $G \unrhd \left\{1\right\}$.
	
	\item $\Sym(3)$ auflösbar:
	Betrachte $\Sym(3) \unrhd \Alt(3) \unrhd \left\{1\right\}$.
	
	\item $\Alt(4)$ auflösbar:
	Betrachte $\Alt(4) \unrhd V_4 \unrhd \left\{1\right\}$ mit
	\[V_4 \coloneqq \left\{
	\underbrace{\begin{pmatrix}1 & 2 & 3 & 4\\ 1 & 2 & 3 & 4\end{pmatrix}}_{=id}, 
	\underbrace{\begin{pmatrix}1 & 2 & 3 & 4\\ 2 & 1 & 4 & 3\end{pmatrix}}_{=a}, 
	\underbrace{\begin{pmatrix}1 & 2 & 3 & 4\\ 3 & 4 & 1 & 2\end{pmatrix}}_{=b},
	\underbrace{\begin{pmatrix}1 & 2 & 3 & 4\\ 4 & 3 & 2 & 1\end{pmatrix}}_{=c}\right\}\]
	\begin{tabular}{c|cccc}
	$V_4$ & $1$ & $a$ & $b$ & $c$\\
	\hline
	$1$ & $1$ & $a$ & $b$ & $c$\\
	$a$ & $a$ & $1$ & $c$ & $b$\\
	$b$ & $b$ & $c$ & $1$ & $a$\\
	$c$ & $c$ & $b$ & $a$ & $1$
	\end{tabular}\\
	Also: $V_4$ abelsche Untergruppe von $\Alt(4)$.
	Man nennt $V_4$ \highl{Kleinsche Vierergruppe}.
	Die nichttrivialen Nebenklassen sind:
	\[\begin{array}{lcl}
	\begin{pmatrix}1 & 2 & 3 & 4\\ 2 & 3 & 1 & 4\end{pmatrix}V_4 & = & \left\{
	\begin{pmatrix}1 & 2 & 3 & 4\\ 2 & 3 & 1 & 4\end{pmatrix}, 
	\begin{pmatrix}1 & 2 & 3 & 4\\ 2 & 3 & 4 & 1\end{pmatrix}, 
	\begin{pmatrix}1 & 2 & 3 & 4\\ 1 & 4 & 2 & 3\end{pmatrix},
	\begin{pmatrix}1 & 2 & 3 & 4\\ 4 & 1 & 3 & 2\end{pmatrix}\right\}\\
	& = & V_4\begin{pmatrix}1 & 2 & 3 & 4\\ 2 & 3 & 1 & 4\end{pmatrix}\\
	& & \\
	\begin{pmatrix}1 & 2 & 3 & 4\\ 3 & 1 & 2 & 4\end{pmatrix}V_4 & = &
	\begin{pmatrix}1 & 2 & 3 & 4\\ 2 & 3 & 1 & 4\end{pmatrix}^2V_4\\
	& = & \begin{pmatrix}1 & 2 & 3 & 4\\ 2 & 3 & 1 & 4\end{pmatrix}V_4\begin{pmatrix}1 & 2 & 3 & 4\\ 2 & 3 & 1 & 4\end{pmatrix}\\
	& = & V_4\begin{pmatrix}1 & 2 & 3 & 4\\ 2 & 3 & 1 & 4\end{pmatrix}^2\\
	& = & V_4\begin{pmatrix}1 & 2 & 3 & 4\\ 3 & 1 & 2 & 4\end{pmatrix}
	\end{array}\]
	Also: $V_4 \unlhd \Alt(4)$.
	
	\item $\Sym(4)/\Alt(4) \cong (\left\{\pm 1\right\},\cdot)$ abelsch und Daher auflösbar.
	Wegen (c) und (iii) ist $\Sym(4)$ auflösbar.
\end{enumerate}
\end{bsp}

%%%%%%%%%%%%%%%%%%%
% Satz 14.2
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:14.2}
$n \geq 5 \Rightarrow \Sym(n)$ nicht auflösbar.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Für paarweise verschiedene $a, b, c, \in \left\{1,\ldots,n\right\}$ sei $(a, b, c) \in \Sym(n)$ die Permutation, die $a$ auf $b$ abbildet, $b$ auf $c$, $c$ auf $a$ und die übrigen Zahlen fest läßt.\\
Offenbar $(a,b,c)^{-1} = (c, b, a)$.
Daher bilden die Elemente der Form 
\[(a_1,b_1,c_1)(a_2,b_2,c_2)\ldots(a_k,b_k,c_k) \qquad (k\in\N_0, a_i, b_i, c_i \in \left\{1,\ldots,n\right\}\text{ paarweise verschieden})\] eine Untergruppe $G$ von $\Sym(n)$.\\
Wir nehmen an: $\Sym(n)$ auflösbar.
Dann ist $G$ auch auflösbar.
Es existiert also eine Folge von Untergruppen $g = G_0 \unrhd G_1 \unrhd G_2\unrhd \dotsc \unrhd G_n = \left\{1\right\}$ wie in \autoref{defi:14.1}.
\OE{} $G \neq G_1$.\\
Für paarweise verschiedene $a,b,c,d,e \in \left\{1,\ldots,n\right\}$ gilt:
\[(a,b,c)=(a,b,d)(c,e,a)(a,b,d)^{-1}(c,e,a)^{-1}.\]
Da $G/G_1$ abelsch ist, folgt:
\[\begin{array}{lcl}
(a,b,c)G_1 & = & \left[(a,b,d)(c,e,a)(a,b,d)^{-1}(c,e,a)^{-1}\right]G_1\\
& = & \left[(a,b,d)G_1\right]\left[(c,e,a)G_1\right]\left[(a,b,d)G_1\right]\left[(c,e,a)G_1\right]\\
& = & 1
\end{array}\]
Also: $(a,b,c) \in G_1$.\\
Daher: $G \leq G_1$, d.\,h. $G=G_1$ $\Rightarrow$ Widerspruch!
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Bemerkung 14.2
%%%%%%%%%%%%%%%%%%%
\begin{bem}\label{bem:14.2}
\begin{enumerate}[(i)]
	\item Wegen $\Alt(n) \unlhd \Sym(n)$ ist auch $\Alt(n)$ für $n \geq 5$ nicht auflösbar.
	Man kann zeigen, dass $\Alt(5)$ die kleinste nicht auflösbare Gruppe ist (Ordnung $60$).
	
	\item Man zeigt leicht, dass Gruppen von Primzahlpotenzordnung auflösbar sind (Algebra 2).
	
	\item \textsc{Burnside}(1901) hat bewiesen, dass Gruppen der Ordnung $p^aq^b \qquad (p,q\in \PZ, a,b \in \N)$ auflösbar sind (Algebra 2).
	
	\item Feit und Thomson haben (1963) bewiesen, dass Gruppen ungerader Ordnung immer auflösbar sind (Odd Order Theorem).
	Der Beweis ist ca. 225 Seiten lang (Bücher von Bender-Glaubermann und Peterfalvi).
\end{enumerate}
\end{bem}

%%%%%%%%%%%%%%%%%%%
% Satz 14.3
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:14.3}
Eine endliche Gruppe $G$ ist genau dann auflösbar, wenn eine Folge von Untergruppen $G = G_0 \unrhd G_1 \unrhd \dotsc \unrhd G_0 = \left\{1\right\}$ mit der folgenden Eigenschaft:\\
Für $i=1,\ldots,n$ ist $G_i\unlhd G_{i-1}$ und $G_{i-1}/G_i$ zyklisch (von Primzahlordnung).
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
\begin{itemize}
	\item["`$\Leftarrow$"':] Klar, da jede zyklische Gruppe abelsch ist und da Gruppen von Primzahlordnung zyklisch sind.
	
	\item["`$\Rightarrow$"':] Sei $G$ auflösbar.
	Dann existiert eine Folge von Untergruppen $G = G_0 \unrhd G_1 \unrhd \dotsc \unrhd G_0 = \left\{1\right\}$ mit der Eigenschaft $G_i \unlhd G_{i-1}$ und $G_{i-1}/G_i$ abelsch für $i=1,\ldots,n$.\\
	Nach \autoref{satz:13.7} existiert eine Folge von Untergruppen
	\[G_{i-1}/G_i = G_{i-1,0}/G_i \geq G_{i-1,1}/G_i\geq \dotsc \geq G_{i-1,k_{i-1}}/G_i = \left\{1\right\}\]
	mit $\underbrace{\abs{G_{i-1,j-1}/G_i:G_{i-1,j}/G_i}}_{=\abs{G_{i-1,j-1}:G_{i-1,j}}} \in \PZ$ für alle $j$.
	Daher ist
	\[(\ast) \qquad G_{i-1} = G_{i-1,0} \geq G_{i-1,1} \geq \dotsc \geq G_{i-1,k_{i-1}} = G_i\]
	eine Folge von Untergruppen von $G_{i-1}$.\\
	Da $G_{i-1}/G_i$ abelsch ist, folgt $G_{i-1}/G_i\unlhd G_{i-1,j-1}/G_i$ und $G_{i-1,j} \unlhd G_{i-1,j-1}$ für alle $j$.
	Wir wissen auch, dass Gruppen von Primzahlordnung zyklisch sind.
	Setzt man also die Folgen $(\ast)$ zusammen, so folgt die Behauptung.\footnote{Verfeinerung der ursprünglichen Folge.}
\end{itemize}
\end{beweis}

% Definition 14.2
%%%%%%%%%%%%%%%%%%%
\begin{defi}\label{defi:14.2}
\begin{enumerate}[(i)]
	\item Eine Körpererweiterung heißt \highl{Körpererweiterng!Radikalerweiterung}{Radikalerweiterung}, wenn eine Folge von Teilkörpern $K = K_0 \subseteq K_1 \subseteq \dotsc \subseteq K_r = L$ und $b_i \in K_i$, $n_i\in\N$ existiert mit\footnote{$b_i$ ist $n_i$-te Wurzel}
	\[K_i = K_{i-1}(b_i) \text{ und } b_i^{n_i} \in K_{i-1} \text{ für } i=1,\ldots,r\]
	
	\item Ein Polynom $\phi\in K\left[X\right]$ heißt über $K$ \highl{Polynom!auflösbar, Radikale}{durch Radikale auflösbar}, wenn eine Radikalerweiterung $L\mid K$ derart existiert, dass $\phi$ in $L\left[X\right]$ in Lineafaktoren zerfällt.
\end{enumerate}
\end{defi}

\begin{bem}\label{bem:14.3}
Dies präzisiert den Begriff der Auflösung der Gleichung $\phi(X) = 0$ durch die 4 Grundrechenarten und das Wurzelziehen.
\end{bem}

% Satz 14.4
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:14.4}
Sei $L\mid K$ eine Radikalerweiterung und $\chr K = 0$.
Dann existiert eine Radikal- und Galoiserweiterung $M\mid K$ mit $L\subseteq M$.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Sei $K = K_0 \subseteq K_1 \subseteq \dotsc \subseteq K_r = L$ mit $K_i = K_{i-1}(b_i)$ und $b_i^{n_i} \in K_{i-1}$ wie oben.\\
Dann: $L = K(b_1,\ldots,b_r)$ und $\left[L:K\right] < \infty$. Für $i=1,\ldots,r$ sie $\mu_i$ das Minimalpolynom von $b_i$ über $K$.
Außerdem sei $N$ Zerfällungskörper von $\mu \coloneqq \mu_1\ldots\mu_r$ über $L$.
Dann ist $N$ auch Zerfällungskörper von $\mu$ über $K$.
Wegen $\chr K = 0$ ist also $N\mid K$ eine Galoiserweiterung.\\
Schreibe: $G\coloneqq \Gal(N\mid K) = \left\{\sigma_1 = id_N,\sigma_2,\ldots,\sigma_s\right\}$.\\
Setze: $K_{ij} \coloneqq K(\sigma_1(b_1),\ldots,\sigma_s(b_1),\ldots,\sigma_1(b_{i-1}),\ldots,\sigma_s(b_{i-1}),\sigma_1(b_i),\ldots,\sigma_j(b_i))$ $(i=1,\ldots,r,j=1,\ldots,s)$.\\
Dann: $\begin{array}[t]{lclclclcl}
	K & \subseteq & K_{11} & \subseteq & K_{12} & \subseteq & \dotsc & \subseteq & K_{1s}\\
	& \subseteq & K_{21} & \subseteq & K_{22} & \subseteq & \dotsc & \subseteq & K_{2s}\\
	& \vdots &&&&&&\vdots&\\
	& \subseteq & K_{r1} & \subseteq & K_{r2} & \subseteq & \dotsc & \subseteq & K_{rs} = M
	\end{array}$\\
	Für $i = 1,\ldots,r$ ist $K_{is} = K(\sigma(b_k)\colon k=1,\ldots,i,\sigma\in G)$.\\
	Für $\tau\in G$ ist $\tau(K_{is}) \subseteq K_{is}$.
	Nach \autoref{satz:11.5} ist $K_{is}\mid K$ eine Galoiserweiterung, insbesondere ist $M\mid K$ eine Galoiserweiterung.\\
	Offenbar ist $K_1 = K(b_1) = K(\sigma_1(b_1)) = K_{11}$.
	Sei nun $i>1$ und bereits $K_{i-1} \subseteq K_{i-1,1}$ bewiesen.\\
	Dann: $K_i = K_{i-1}(b_i) \subseteq K_{i-1,1}(b_i) \subseteq K_{i-1,s}(b_i) = K_{i1}$, und $b_i^{n_i} \in K_{i-1} \subseteq K_{i-1,1} \subseteq K_{i-1,s}$.
	Daher: $K_i\subseteq K_{i1}$ und $K_{i1} = K_{i-1,s}(b_i)$ und $b_i^{n_i} \in K_{i-1,s}$.
	Insbesondere: $L = K_r \subseteq K_{r1} \subseteq K_{rs} = M$.\\
	Für $j>1$ ist $K_{ij} = K_{i,j-1}(\sigma_j(b_i))$ mit
	\[\sigma_j(b_i)^{ni} = \sigma_j(b_i^{n_i}) \in \sigma_j(K_{i-1,s}) = K_{i-1,s} \subseteq K_{i1}\subseteq K_{i,j-1}.\]
	Daher: $M\mid K$ Radikalerweiterung.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Satz 14.5
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:14.5}
Sei $L\mid K$ eine Radikal- und Galoiserweiterung mit $\chr K = 0$.\\
Dann ist $\Gal(K\mid K)$ auflösbar.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Die Aussagen von $\Gal(M\mid K)$ sind schwer. Deswegen wenden wir den
Umweg über Adjunktion von $\zeta$ an und wenden dann das vorherige
Kapitel an.
Sei $K = K_0 \subseteq K_1 \subseteq \dotsc \subseteq K_r = L$ mit $K_i = K_{i-1}(b_i)$ und $b_i^{n_i}\in K_{i-1}$ wie oben.
Sei $n \coloneqq n_1\ldots n_r$ und $M$ Zerfällungskörper von $X^n-1$ über $L$.
Wie üblich bilden die Nullstellen von $X^n-1$ in $M$ eine zyklische Gruppe $\left\langle \zeta\right\rangle$ der Ordnung $n$.
Wegen $K\subseteq K(\zeta)\subseteq K_1(\zeta) \subseteq \dotsc \subseteq K_r(\zeta) = L(\zeta) = M$, $\zeta^n = 1 \in K$ und $K_i(\zeta) = K_{i-1}(\zeta,b_i)$ und $b_i^{n_i} \in K_{i-1} \subseteq K_{i-1}(\zeta)$ für alle $i$ ist $M\mid K$ Radikalerweiterung.

Da $L\mid K$ Galoiserweiterung ist, ist $L$ Zerfällungskörper eines Polynomes $\phi\in K\left[X\right]$ über $K$.
Dann ist $M\mid K$Zerfällungskörper von $(X^n-1)\phi \in K\left[X\right]$ über $K$.
Wegen $\chr K = 0$ ist $M\mid K$ Galoiserweiterung.

Es genügt zu zeigen, dass $\Gal(M\mid K)$ auflösbar ist;
denn dann ist auch $\Gal(L\mid K) \stackrel{\text{\autoref{satz:11.4}}}{\cong} \Gal(M\mid K) / \Gal(M\mid L)$.

Zunächst ist $K(\zeta)\mid K$ eine Galoiserweiterung, da $K(\zeta)$ Zerfällungskörper von $X^n-1$ über $K$ ist.
Für $\sigma\in\Gal(K(\zeta)\mid K)$ ist $\sigma(\zeta)^n = \sigma(\zeta^n) = \sigma(1) = 1$, d.\,h. $\sigma(\zeta) = \zeta^i$ für ein $i\in\Z$ mit $\ggt(i, n) = 1$.
Dabei ist $i$ modulo $n$ eindeutig bestimmt.\\
Durch $\sigma\mapsto i+n\Z$ erhalten wir eine wohldefinierte Abbildung
\[f\colon\Gal(K(\zeta)\mid K)\mapsto U(\ZmZ{n}).\]
Diese ist injektiv, da $\sigma$ durch $\sigma(\zeta) = \zeta^i$ eindeutig bestimmt ist.\\
Sei nun auch $\tau\in\Gal(K(\sigma)\mid K)$, und sei $j\in\Z$ mit $\ggt(j,n) = 1$ und $\tau(\zeta) = \zeta^j$.
Dann: $(\sigma\circ\tau)(\zeta) = \sigma(\tau(\zeta)) = \sigma(\zeta^j) = \sigma(\zeta)^j = (\zeta^i)^j = \zeta^{ij}$, d.\,h. $f(\sigma\circ\tau) = ij+n\Z = (i+n\Z)(j+n\Z) = f(\sigma)f(\tau)$.
Also ist $f$ ein Monomorphismus, insbesondere ist $\Gal(K(\zeta)\mid K) \cong \Bld(f) \leq U(\ZmZ{n})$ abelsch.
Daher ist $\Gal(M\mid K(\zeta)) \unlhd \Gal(M\mid K)$ und $\Gal(M\mid K) / \Gal(M\mid K(\zeta)) \cong \Gal(K(\zeta)\mid K)$ abelsch.

Für $i = 1,\ldots,r$ ist $K_i(\zeta) = K_{i-1}(\zeta,b_i)$ und $b_i^{n_i} \in K_{i-1}(\zeta)$.
Da das Produkt $X^{n_i}-1$ in $K_{i-1}(\zeta)$ in Linearfaktoren zerfällt, ist $K_i(\zeta)\mid K_{i-1}(\zeta)$ nach \autoref{satz:13.8} eine Galoiserweiterung mit zyklischer Galoisgruppe $\Gal(K_i(\zeta)\mid K_{i-1}(\zeta))$.\\
Nach \autoref{satz:11.4} ist $\Gal(M\mid K_i(\zeta)) \unlhd \Gal(M\mid K_{i-1}(\zeta))$, und 
\[\Gal(M\mid K_{i-1}(\zeta))/\Gal(M\mid K_i(\zeta)) \cong \Gal(K_i(\zeta)\mid K_{i-1}(\zeta))\]
ist zyklisch.\\
Wir erhalten so eine Folge von Untergruppen
\[\Gal(M\mid K) \geq \Gal(M\mid K(\zeta)) \geq \Gal(M\mid K_1(\zeta)) \geq \dotsc \geq \Gal(M\mid \underbrace{K_r(\zeta)}_{=M}) = \left\{1\right\}\]
dabei ist jedes Glied Normalteiler im vorangehenden Glied, und die entsprechende Faktorgruppe ist abelsch.
Daher ist $\Gal(M\mid K)$ auflösbar.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
\begin{satz}[Eisenstein Kriterium]\label{satz:14.6}
Sei $\alpha = X^n + a_{n-1}X^{n-1} + \dotsc + a_1X+a_0\in\Z\left[X\right]$, und sei $p\in\PZ$ mit $p\mid a_0,\ldots,p\mid a_{n-1},p^2\nmid a_0$.\\
Dann ist $\alpha$ irreduzibel in $\Q\left[X\right]$.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Annahme: $\alpha$ reduzibel in $\Q\left[X\right]$.\\
Dann existiert $\beta,\gamma \in\Z\left[X\right]$ mit $\alpha = \beta\gamma$ und $\deg\beta < n > \deg\gamma$, \OE{}$\beta,\gamma$ normiert.
Schreibe
\[\begin{array}{lcl}
\beta & = & X^k + b_{k-1}X^{k-1} + \dotsc + b_1X + b_0,\\
\gamma & = & X^l + c_{l-1}X^{l-1} + \dotsc + c_1X + c_0 \qquad (b_i,c_j\in\Z).
\end{array}\]
Wegen $p^2\nmid a_0 = b_0c_0$ ist $p\nmid b_0$ oder $p\nmid c_0$, \OE{} $p\nmid c_0$.
Wegen $p\nmid a_0 = b_0c_0$ ist $p\nmid b_0$.
Sei $p\mid b_1,\ldots,p\mid b_{i-1},p\nmid b$ $(1 \leq i \leq k < n)$.
Dann
\[p\mid a_i = \underbrace{b_0c_i + b_1c_{i-1}+\dotsc}_{\equiv 0\pmod{p}}+\underbrace{b_ic_0}_{\not\equiv 0\pmod{p}}\]
Widerspruch!
\end{beweis}

% Satz 14.7
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:14.7}
Sei $f\colon K\mapsto\widetilde{K}$, $a\mapsto\widetilde{a}$, ein Isomorphismus von Körpern.\\
Sei 
\[\phi = a_nX^n+a_{n-1}X^{n-1}+\dotsc+a_1X + a_0\in K\left[X\right]\] 
und sei 
\[\widetilde{\phi}\coloneqq\widetilde{a}_nX^n + \widetilde{a}_{n-1}X^{n-1}+\dotsc+\widetilde{a}_1X+\widetilde{a}_0 \in \widetilde{K}\left[X\right]\] 
das entsprechende Polynom in $\widetilde{K}\left[X\right]$.\\
Sei $L$ Zerfällungskörper von $\phi$ über $K$, und sei $\widetilde{L}$ Zerfällungskörper von $\widetilde{\phi}$ über $\widetilde{K}$.\\
Dann kann man $f$ zu einem Isomorphismus $g\colon L\mapsto\widetilde{L}$\footnote{d.\,h. Zerfällungskörper sind isomorph.} fortsetzen.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
(Induktion nach $\left[L:K\right] =: m$)\\
Offenbar ist die Abbildung $F\colon K\left[X\right]\mapsto\widetilde{K}\left[X\right]$, $\sum\limits_{i=0}^{\infty}c_iX^i\mapsto\sum\limits_{i=0}^{\infty}X^i$, ein Isomorphismus von Ringen.

Im Fall $m = 1$ ist $L=K$.
Also ist $\phi = a_n(X-b_1)\ldots(X-b_n)$ mit $b_1,\ldots,b_n\in K$.
Daher ist $\widetilde{\phi} = F(\phi) = \widetilde{a}(X-\widetilde{b}_1)\ldots(X-\widetilde{b}_n)$ mit $\widetilde{b}_1,\ldots,\widetilde{b}_n \in\widetilde{K}$.\\
Also ist $\widetilde{L} = \widetilde{K}(\widetilde{b}_1,\ldots,\widetilde{b}_n) = \widetilde{K}$, und die Sache ist klar.

Im Fall $m>1$ und $\pi$ Primfaktor von $\phi$ in $K\left[X\right]$ mit $\deg\pi > 1$.
Dann ist $\widetilde{\pi} = F(\pi)$ ein Primfaktor von $F(\phi) = \widetilde{\phi}$.
Mit $\phi$ zerfällt $\pi$ in $L\left[X\right]$ in Linearfaktoren, hat also mindestens eine Nullstelle $a\in L$.
Analog existiert eine Nullstelle $\widetilde{a}$ von $\widetilde{\pi}$ in $\widetilde{L}$.
Nach dem Fortsetzungssatz (\autoref{satz:11.6}) können wir $f$ zu einem Isomorphismus $h\colon K(a)\mapsto\widetilde{K}(\widetilde{a})$ mit $h(a) = \widetilde{a}$ fortsetzen.
Offenbar ist $L$ auch Zerfällungskörper von $\phi$ über $K(a)$, und $\widetilde{L}$ ist Zerfällungskörper von $\widetilde{\phi}$ über $\widetilde{K}(\widetilde{a})$.
Nach dem Gradsatz (\autoref{satz:6.3}) ist
\[\left[L:K(a)\right] = \left[L:K\right]/\underbrace{\left[K(a):K\right]}_{=\deg\pi > 1} < \left[L:K\right].\]
Nach Induktion kann man also $f$ zu einem Isomorphismus $G\colon L\mapsto\widetilde{L}$ fortsetzen.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Bemerkung 14.5
%%%%%%%%%%%%%%%%%%%
\begin{bem}\label{bem:14.5}
Sind also $L,L'$ Zerfällungskörper von $\phi$ über $K$, so existiert ein Isomorphismus $G\colon L\mapsto L'$ mit $g(x) = x$ für $x\in K$.\footnote{Spezialfall von \autoref{satz:14.7} für $f=id$ und $K=\widetilde{K}$}
\end{bem}

% Satz 14.8
%%%%%%%%%%%%%%%%%%%
\begin{satz}[Cauchy]\label{satz:14.8}
Sei $G$ eine endliche Gruppe und $p$ ein Primteiler von $\abs{G}$.\\
Dann enthält $G$ ein Element der Ordnung $p$.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Sei $M\coloneqq \left\{(g_1,\ldots,g_p)\in G^p\colon g_1\cdot\dotsc\cdot g_p = 1\right\}$.\\
Dann: $\abs{M} = \abs{G}^{p-1}\equiv 0\pmod{p}$.\\
Wir definieren eine Relation $\sim$ auf $M$ durch:
\[(g_1,\ldots,g_p)\sim(h_1,\ldots,h_p):\Leftrightarrow \exists i\in\left\{1,\ldots,p\right\}\colon (h_1,\ldots,h_p)=(g_i,\ldots,g_p,g_1,\ldots,g_{i-1})\]
Offenbar ist $\sim$ eine Äquivalenzrelation.
Jede Äquivalenzklasse enthält $1$ oder $p$ Elemente.
Daher ist die Anzahl der $1$-elementigen Äquivalenzklassen durch $p$ teilbar.
Sicher ist $\left\{(1,\ldots,1)\right\}$ eine einelementige Äquivalenzklasse.
Also existiert mindestens eine weitere einelementige Äauivalenzklasse $\left\{(a,\ldots,a)\right\} \qquad (a\neq 1)$.
Wegen $a^p = 1$ hat $a$ die Ordnung $p$.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Definition 14.3
%%%%%%%%%%%%%%%%%%%
\begin{defi}\label{defi:14.3}
Sei $n\in\N$.
Für $i=1,\ldots,n-1$ sei $s_i\in\Sym(n)$ die Permutation, die $i$ und $i+1$ vertauscht und die übrigen Zahlen fest lässt.
Man nennt $s_1,\ldots,s_{n-1}$ \highl{Basistranspositionen} in $\Sym(n)$.
\end{defi}

%%%%%%%%%%%%%%%%%%%
% Satz 14.9
%%%%%%%%%%%%%%%%%%%
\begin{satz}\label{satz:14.9}
Jedes $g\in \Sym(n)$ lässt sich als Produkt von Basistranspositionen schreiben.
\end{satz}

%%%%%%%%%%%%%%%%%%%
% Beweis
%%%%%%%%%%%%%%%%%%%
\begin{beweis}
Sei $\Phi(g) \coloneqq \left\{\left\{i,j\right\}\colon 1\leq i<j\leq n, g(i)>g(j)\right\}$ und $l(g) \coloneqq \abs{\Phi(g)}$.
(Die Elemente in $\Phi(g)$ heißen \highl{Inversionen} oder \highl{Fehlstände} und $l(g)$ heißt die \highl{L"ange}{Länge} von $g$.)
Wir argumentieren nach $l(g)$.

Im Fall $l(g) = 0$ ist $\Phi(g) = \varnothing$, d.\,h. $g(1)<g(2)<g(3)<\dotsc<g(n)$ und daher $g=1$.

Sei $l(g) > 0$.
Im Fall $\left\{k, k+1\right\} \notin \Phi(g)$ für $k=1,\ldots,n-1$ wäre wieder $g(1)<g(2)<\dotsc<g(n)$, d.\,h. $g=1$ und $l(g) = 0$. Widerspruch!

Also existiert ein $k\in\left\{1,\ldots,n-1\right\}$ mit $\left\{k,k+1\right\} \in \Phi(g)$.
Man zeigt leicht:
\[\Phi(g\cdot s_k) = \left\{\left\{s_k(i),s_k(j)\right\}\colon \left\{ij\right\}\in\Phi(g)\right\}\backslash\left\{\left\{k,k+1\right\}\right\}\]
Dann ist $l(g\cdot s_k) = l(g)-1$.
Daher ist $g\cdot s_k$ ein Produkt von Basistranspositionen, also auch $g\cdot s_k \cdot s_k = g$.
\end{beweis}

%%%%%%%%%%%%%%%%%%%
% Beispiel 14.3
%%%%%%%%%%%%%%%%%%%
\begin{bsp}\label{bsp:14.3}
$g = \begin{pmatrix}1 & 2 & 3\\ 3 & 2 & 1\end{pmatrix} \Rightarrow gs_1 = \begin{pmatrix}1 & 2 & 3\\ 2 & 3 & 1\end{pmatrix} \Rightarrow gs_1s_2 = \begin{pmatrix}1 & 2 & 3\\ 2 & 1 & 3\end{pmatrix} \Rightarrow gs_1s_2s_1 = \begin{pmatrix}1 & 2 & 3\\ 1 & 2 & 3\end{pmatrix} = id$ $\Rightarrow g = s_1^{-1}s_2^{-1}s_1^{-1} = s_1s_2s_1$.
\end{bsp}

% Beispiel 14.4
%%%%%%%%%%%%%%%%%%%
\begin{bsp}\label{bsp:14.4}
Wir wollen zeigen, dass das Polynom $\phi = X^5-4X+2$ über $\Q$ nicht durch Radikale auflösbar ist.
Dazu nehmen wir das Gegenteil an.

Dann existiert eine Radikalerweiterung $M\mid\Q$, in der $\phi$ in Linearfaktoren zerfällt.
Nach \autoref{satz:14.4} können wir annehmen, dass $M\mid\Q$ auch Galoiserweiterung ist.\footnote{Notfalls muss $M$ vergrößert werden.}
Seien $a_1,\ldots,a_5$ die Nullstellen von $\phi$ in $M$.
Dann ist $L = \Q(a_1,\ldots,a_5)$ Zerfällungskörper von $\phi$ über $\Q$, insbesondere ist $L\mid\Q$ Galoiserweiterung.
Nach \autoref{satz:11.4} ist $\Gal(M\mid L) \unlhd \Gal(M\mid\Q)$, und $\Gal(M\mid\Q)/\Gal(M\mid L) \cong \Gal(L\mid\Q)$.\\
Nach \autoref{satz:14.5} ist $\Gal(M\mid\Q)$ auflösbar, also auch $\Gal(L\mid\Q)$.
Nach Eisenstein (\autoref{satz:14.6}) ist $\phi$ irreduzibel in $\Q\left[X\right]$.
Da $\C$ algebraisch abgeschlossen ist, zerfällt $\phi$ in $\C\left[X\right]$ in Linearfaktoren.
Wegen $\chr\C = 0$ sind die Nullstellen $b_1,\ldots,b_5$ von $\phi$ in $\C$ paarweise verschieden.
Daher ist $L'\coloneqq\Q(b_1,\ldots,b_5)$ ein Zerfällungskörper von $\phi$ über $\Q$.
Nach \autoref{satz:14.7} existiert ein Isomorphismus $f\colon L\mapsto L'$ mit $f(x) = x$ für $x\in\Q$.\\
Daher ist $G\coloneqq\Gal(L'\mid\Q) \cong \Gal(L\mid\Q)$ auflösbar.
Wir werden zeigen, dass andererseits $G\cong\Sym(5)$ ist.
\autoref{satz:14.2} liefert dann den gesuchten Widerspruch.

\underline{Kurvendiskussion:} $\phi' = 5X^4-4$ hat die Nullstelle $\pm\sqrt[4]{\frac{4}{5}}$.
Wegen $\phi(0) = 2$ und $\phi(1) = -1$ hat $\phi$ genau $3$ reelle Nullstellen (und zwei konjugiert komplexe nicht reelle Nullstellen).\\
Jedes $\sigma\in G$ permutiert die Nullstellen $b_1,\ldots,b_5$ von $\phi$.
Auf diese Weise erhalten wir einen Monomorphismus: 
\[\Gamma\colon G\mapsto\Sym(5)\]
dabei ist $5 = \deg\phi = \left[\Q(b_1):\Q\right]\mid\left[L':\Q\right] = \abs{G}$.
Nach Cauchy (\autoref{satz:14.8}) entahlten $G$ und $\Gamma(G)$ ein Element der Ordnung $5$.
Man zeigt leicht, dass die Elemente der Ordnung $5$ in $\Sym(5)$ die folgende Form haben:\\
% \InitGraph{3}{3}{0.5}{0.5}{0.5cm}
% \TextAt(0, 1.2)[c]{e}
% \TextAt(0.3, 0)[c]{d}
% \TextAt(1.7, 0)[c]{c}
% \TextAt(1, 2)[c]{a}
% \TextAt(2, 1.2)[c]{b}
% \ArrowAt(1.2, 1.8, 1.8, 1.4) % a -> b
% \ArrowAt(2, 0.9, 1.8, 0.3) % b -> c
% \ArrowAt(1.4, 0.1, 0.6, 0.1) % c -> d
% \ArrowAt(0.2, 0.2, 0.1, 0.9) % d -> e
% \ArrowAt(0.2, 1.4, 0.8, 1.9) % e -> a
% \CloseGraph
% $=:(a, b, c, d, e)$, dabei ist $\left\{a, b, c, d, e\right\} = \left\{1, 2, 3, 4, 5\right\}$\\
Sei also $(a, b, c, d, e) \in \Gamma(G)$.
Die komplexe Konjugation liefert ein weiteres Element in $G$.
Das entsprechende Element in $\Gamma(G)$ hat die Form:
\[x \rightleftarrows y =: (x, y) \qquad (x, y \in \left\{1,\ldots,5\right\},x\neq y).\]
Bei passender Numerierung können wir $x=1, y=2$ annehmen.
Wegen
\[(a,b,c,d,e) = (b,c,d,e,a)=(c,d,e,a,b)=(d,e,a,b,c)=(e,a,b,c,d)\]
können wir $a=1$ annehmen.
Wegen
\[\begin{array}{lcl}
(a,b,c,d,e)^2 & = & (a,c,e,b,d)\\
(a,b,c,d,e)^3 & = & (a,d,b,e,c)\\
(a,b,c,d,e)^4 & = & (a,e,d,c,b) = (a,b,c,d,e)^{-1}
\end{array}\]
können wir $b=2$ annehmen.
Also können wir \OE{} annehmen:
\[(1,2),(1,2,3,4,5)\in\Gamma(G).\]
Wegen 
\[\begin{array}[t]{lcl}
(1,2,3,4,5)(1,2)(1,2,3,4,5)^{-1} & = & (2,3)\\
(1,2,3,4,5)(2,3)(1,2,3,4,5)^{-1} & = & (3,4)\\
(1,2,3,4,5)(3,4)(1,2,3,4,5)^{-1} & = & (4,5)
\end{array}.\]
Also enthält $\Gamma(G)$ alle Basistranspositionen.
Daher ist $\underbrace{\Gamma(G)}_{\cong G} = \Sym(5)$.
\end{bsp}
